45 available are {64, 128, 256, 512, 1024}.

66 Dictionary-like object, the interest attributes are : 

67

69 - 'labels': Numpy recarray containing the labels of the regions. 

70 - 'description': str, general description of the dataset. 

71 

72 References 

73 ---------- 

74 .. footbibliography:: 

75 

76 """

402def _fetch_atlases_fsl(atlas_source, atlas_name, 

403 data_dir=None, symmetric_split=False, 

404 resume=True, verbose=1): 

405 

406 atlas_sources = {"HarvardOxford", "Juelich"} 

407 

408 urls = {'HarvardOxford': 

409 'http://www.nitrc.org/frs/download.php/9902/HarvardOxford.tgz', 

410 'Juelich': 

411 'https://www.nitrc.org/frs/download.php/12096/Juelich.tgz'} 

412 

413 atlas_items = {'HarvardOxford': ( 

414 "cort-maxprob-thr0-1mm", "cort-maxprob-thr0-2mm", 

415 "cort-maxprob-thr25-1mm", "cort-maxprob-thr25-2mm", 

416 "cort-maxprob-thr50-1mm", "cort-maxprob-thr50-2mm", 

417 "cort-prob-1mm", "cort-prob-2mm", 

418 "cortl-maxprob-thr0-1mm", "cortl-maxprob-thr0-2mm", 

419 "cortl-maxprob-thr25-1mm", "cortl-maxprob-thr25-2mm", 

420 "cortl-maxprob-thr50-1mm", "cortl-maxprob-thr50-2mm", 

421 "cortl-prob-1mm", "cortl-prob-2mm", 

422 "sub-maxprob-thr0-1mm", "sub-maxprob-thr0-2mm", 

423 "sub-maxprob-thr25-1mm", "sub-maxprob-thr25-2mm", 

424 "sub-maxprob-thr50-1mm", "sub-maxprob-thr50-2mm", 

425 "sub-prob-1mm", "sub-prob-2mm"), 

426 'Juelich': ( 

427 "maxprob-thr0-1mm", "maxprob-thr0-2mm", 

428 "maxprob-thr25-1mm", "maxprob-thr25-2mm", 

429 "maxprob-thr50-1mm", "maxprob-thr50-2mm", 

430 "prob-1mm", "prob-2mm")} 

431 

432 if atlas_source not in atlas_sources: 

433 raise ValueError("Invalid atlas source: {0}. " 

434 "Please choose an atlas source " 

435 "among:\n{1}".format(atlas_source, 

436 '\n'.join(atlas_sources))) 

437 

438 if atlas_name not in atlas_items[atlas_source]: 

439 raise ValueError("Invalid atlas name: {0}. Please choose an atlas " 

440 "among:\n{1}".format(atlas_name, 

441 '\n'.join(atlas_items[ 

442 atlas_source]))) 

443 

444 if atlas_name in ("cortl-prob-1mm", "cortl-prob-2mm", 

445 "cort-prob-1mm", "cort-prob-2mm", 

446 "sub-prob-1mm", "sub-prob-2mm", 

447 "prob-1mm", "prob-2mm") and symmetric_split: 

448 raise ValueError("Region splitting not supported for probabilistic " 

449 "atlases") 

450 

451 # For practical reasons, we mimic the FSL data directory here. 

452 dataset_name = 'fsl' 

453 data_dir = _get_dataset_dir(dataset_name, data_dir=data_dir, 

454 verbose=verbose) 

455 opts = {'uncompress': True} 

456 root = os.path.join('data', 'atlases') 

457 

458 if atlas_source == 'HarvardOxford': 

459 if atlas_name[0] == 'c': 

460 if ('cort-maxprob' in atlas_name and symmetric_split 

461 or 'cortl-maxprob' in atlas_name): 

462 split_name = atlas_name.split('-', 1) 

463 atlas_name = 'cortl-{}'.format(split_name[1]) 

464 label_file = 'HarvardOxford-Cortical-Lateralized.xml' 

465 lateralized = True 

466 else: 

467 label_file = 'HarvardOxford-Cortical.xml' 

468 lateralized = False 

469 else: 

470 label_file = 'HarvardOxford-Subcortical.xml' 

471 lateralized = False 

472 else: 

473 label_file = "Juelich.xml" 

474 lateralized = False 

475 

476 label_file = os.path.join(root, label_file) 

477 

478 atlas_file = os.path.join(root, atlas_source, 

479 '{}-{}.nii.gz'.format(atlas_source, atlas_name)) 

480 

481 atlas_img, label_file = _fetch_files( 

482 data_dir, 

483 [(atlas_file, urls[atlas_source], opts), 

484 (label_file, urls[atlas_source], opts)], 

485 resume=resume, verbose=verbose) 

486 

487 names = {} 

488 from xml.etree import ElementTree 

489 names[0] = 'Background' 

490 for label in ElementTree.parse(label_file).findall('.//label'): 

491 names[int(label.get('index')) + 1] = label.text 

492 names = list(names.values()) 

493 

494 if not symmetric_split and atlas_source == "Juelich": 

495 atlas_img = check_niimg(atlas_img) 

496 atlas = get_data(atlas_img) 

497 

498 labels = np.unique(atlas) 

499 new_label = 0 

500 new_atlas = atlas.copy() 

501 # Assumes that the background label is zero. 

502 new_names = [names[0]] 

503 for label, name in zip(labels[1:], names[1:]): 

504 new_label += 1 

505 if name.endswith('R'): 

506 new_label -= 1 

507 new_atlas[atlas == label] = new_label 

508 continue 

509 else: 

510 new_atlas[atlas == label] = new_label 

511 if name.endswith('L'): 

512 new_names.append(name.rsplit(" ", 1)[0]) 

513 else: 

514 new_names.append(name) 

515 atlas_img = new_img_like(atlas_img, new_atlas, atlas_img.affine) 

516 return Bunch(maps=atlas_img, labels=new_names) 

517 atlas_img = check_niimg(atlas_img) 

518 if lateralized or not symmetric_split: 

519 return Bunch(maps=atlas_img, labels=names) 

520 

521 atlas = get_data(atlas_img) 

522 

523 labels = np.unique(atlas) 

524 # Build a mask of both halves of the brain 

525 middle_ind = (atlas.shape[0] - 1) // 2 

526 # Put zeros on the median plane 

527 atlas[middle_ind, ...] = 0 

528 # Split every zone crossing the median plane into two parts. 

529 left_atlas = atlas.copy() 

530 left_atlas[middle_ind:, ...] = 0 

531 right_atlas = atlas.copy() 

532 right_atlas[:middle_ind, ...] = 0 

533 

534 new_label = 0 

535 new_atlas = atlas.copy() 

536 # Assumes that the background label is zero. 

537 new_names = [names[0]] 

538 for label, name in zip(labels[1:], names[1:]): 

539 new_label += 1 

540 left_elements = (left_atlas == label).sum() 

541 right_elements = (right_atlas == label).sum() 

542 n_elements = float(left_elements + right_elements) 

543 if (left_elements / n_elements < 0.05 

544 or right_elements / n_elements < 0.05): 

545 new_atlas[atlas == label] = new_label 

546 new_names.append(name) 

547 continue 

548 new_atlas[right_atlas == label] = new_label 

549 new_names.append(name + ', left part') 

550 new_label += 1 

551 new_atlas[left_atlas == label] = new_label 

552 new_names.append(name + ', right part') 

553 

554 atlas_img = new_img_like(atlas_img, new_atlas, atlas_img.affine) 

555 return Bunch(maps=atlas_img, labels=new_names)

1339def _separate_talairach_levels(atlas_img, labels, verbose=1): 

1340 """Separate the multiple annotation levels in talairach raw atlas. 

1341 

1342 The Talairach atlas has five levels of annotation: hemisphere, lobe, gyrus, 

1343 tissue, brodmann area. They are mixed up in the original atlas: each label 

1344 in the atlas corresponds to a 5-tuple containing, for each of these levels, 

1345 a value or the string '*' (meaning undefined, background). 

1346 

1347 This function disentangles the levels, and stores each on an octet in an 

1348 int64 image (the level with most labels, ba, has 72 labels). 

1349 This way, any subset of these levels can be accessed by applying a bitwise 

1350 mask. 

1351 

1352 In the created image, the least significant octet contains the hemisphere, 

1353 the next one the lobe, then gyrus, tissue, and ba. Background is 0. 

1354 The labels contain 

1355 [('level name', ['labels', 'for', 'this', 'level' ...]), ...], 

1356 where the levels are in the order mentionned above. 

1357 

1358 The label '*' is replaced by 'Background' for clarity. 

1359 

1360 """ 

1361 labels = np.asarray(labels) 

1362 if verbose: 

1363 print( 

1364 'Separating talairach atlas levels: {}'.format(_TALAIRACH_LEVELS)) 

1365 levels = [] 

1366 new_img = np.zeros(atlas_img.shape, dtype=np.int64) 

1367 for pos, level in enumerate(_TALAIRACH_LEVELS): 

1368 if verbose: 

1369 print(level) 

1370 level_img = np.zeros(atlas_img.shape, dtype=np.int64) 

1371 level_labels = {'*': 0} 

1372 for region_nb, region in enumerate(labels[:, pos]): 

1373 level_labels.setdefault(region, len(level_labels)) 

1374 level_img[get_data(atlas_img) == region_nb] = level_labels[ 

1375 region] 

1376 # shift this level to its own octet and add it to the new image 

1377 level_img <<= 8 * pos 

1378 new_img |= level_img 

1379 # order the labels so that image values are indices in the list of 

1380 # labels for each level 

1381 level_labels = list(list( 

1382 zip(*sorted(level_labels.items(), key=lambda t: t[1])))[0]) 

1383 # rename '*' -> 'Background' 

1384 level_labels[0] = 'Background' 

1385 levels.append((level, level_labels)) 

1386 new_img = new_img_like(atlas_img, data=new_img) 

1387 return new_img, levels

1390def _get_talairach_all_levels(data_dir=None, verbose=1): 

1391 """Get the path to Talairach atlas and labels 

1392 

1393 The atlas is downloaded and the files are created if necessary. 

1394 

1395 The image contains all five levels of the atlas, each encoded on 8 bits 

1396 (least significant octet contains the hemisphere, the next one the lobe, 

1397 then gyrus, tissue, and ba). 

1398 

1399 The labels json file contains 

1400 [['level name', ['labels', 'for', 'this', 'level' ...]], ...], 

1401 where the levels are in the order mentionned above. 

1402 

1403 """

1409 return img_file, labels_file

1411 temp_dir = mkdtemp() 

1412 try: 

1413 temp_file = _fetch_files( 

1414 temp_dir, [('talairach.nii', atlas_url, {})], verbose=verbose)[0]
