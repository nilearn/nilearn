699 A ``metadata`` dictionary will be blocked by the filter if it 

700 does not respect ``metadata[key] == value`` for all 

701 ``key``, ``value`` pairs in `query_terms`. If ``None``, the 

702 empty dictionary is used. 

703 

704 callable_filter : callable, optional 

705 A ``metadata`` dictionary will be blocked by the filter if 

706 `callable_filter` does not return ``True`` for ``metadata``. 

707 Default=empty_filter 

708 

709 As an alternative to the `query_terms` dictionary parameter, 

710 key, value pairs can be passed as keyword arguments. 

711 

712 Attributes 

713 ---------- 

714 query_terms_ : dict 

715 In order to pass through the filter, metadata must verify 

716 ``metadata[key] == value`` for each ``key``, ``value`` pair in 

717 `query_terms_`. 

718 

719 callable_filters_ : list of callables 

720 In addition to ``(key, value)`` pairs, we can use this 

721 attribute to specify more elaborate requirements. Called with 

722 a dict representing metadata for an image or collection, each 

723 element of this list returns ``True`` if the metadata should 

724 pass through the filter and ``False`` otherwise. 

725 

726 A dict of metadata will only pass through the filter if it 

727 satisfies all the `query_terms` AND all the elements of 

728 `callable_filters_`. 

729 

730 See Also 

731 -------- 

732 nilearn.datasets.neurovault.IsNull, 

733 nilearn.datasets.neurovault.NotNull, 

734 nilearn.datasets.neurovault.NotEqual, 

735 nilearn.datasets.neurovault.GreaterOrEqual, 

736 nilearn.datasets.neurovault.GreaterThan, 

737 nilearn.datasets.neurovault.LessOrEqual, 

738 nilearn.datasets.neurovault.LessThan, 

739 nilearn.datasets.neurovault.IsIn, 

740 nilearn.datasets.neurovault.NotIn, 

741 nilearn.datasets.neurovault.Contains, 

742 nilearn.datasets.neurovault.NotContains, 

743 nilearn.datasets.neurovault.Pattern. 

744 

745 Examples 

746 -------- 

747 >>> from nilearn.datasets.neurovault import ResultFilter 

748 >>> filt = ResultFilter(a=0).AND(ResultFilter(b=1).OR(ResultFilter(b=2))) 

749 >>> filt({'a': 0, 'b': 1}) 

750 True 

751 >>> filt({'a': 0, 'b': 0}) 

752 False 

753 

754 """ 

755 def __init__(self, query_terms=None, 

756 callable_filter=_empty_filter, **kwargs): 

757 if query_terms is None: 

758 query_terms = {} 

759 query_terms = dict(query_terms, **kwargs) 

760 self.query_terms_ = query_terms 

761 self.callable_filters_ = [callable_filter] 

762 

763 def __call__(self, candidate): 

764 """Return True if candidate satisfies the requirements. 

765 

766 Parameters 

767 ---------- 

768 candidate : dict 

769 A dictionary representing metadata for a file or a 

770 collection, to be filtered. 

771 

772 Returns 

773 ------- 

774 bool 

775 ``True`` if `candidate` passes through the filter and ``False`` 

776 otherwise. 

777 

778 """ 

779 for key, value in self.query_terms_.items(): 

780 if not (value == candidate.get(key)): 

781 return False 

782 for callable_filter in self.callable_filters_: 

783 if not callable_filter(candidate): 

784 return False 

785 return True

788 filt1, filt2 = deepcopy(self), deepcopy(other_filter) 

789 new_filter = ResultFilter( 

790 callable_filter=lambda r: filt1(r) or filt2(r)) 

791 return new_filter

793 def AND(self, other_filter): 

794 filt1, filt2 = deepcopy(self), deepcopy(other_filter) 

795 new_filter = ResultFilter( 

796 callable_filter=lambda r: filt1(r) and filt2(r)) 

797 return new_filter

799 def XOR(self, other_filter): 

800 filt1, filt2 = deepcopy(self), deepcopy(other_filter) 

801 new_filter = ResultFilter( 

802 callable_filter=lambda r: filt1(r) != filt2(r)) 

803 return new_filter

805 def NOT(self): 

806 filt = deepcopy(self) 

807 new_filter = ResultFilter( 

808 callable_filter=lambda r: not filt(r)) 

809 return new_filter

930def _get_batch(query, prefix_msg='', timeout=10., verbose=3): 

931 """Given an URL, get the HTTP response and transform it to python dict. 

932 

933 The URL is used to send an HTTP GET request and the response is 

934 transformed into a dictionary. 

935 

936 Parameters 

937 ---------- 

938 query : str 

939 The URL from which to get data. 

940 

941 prefix_msg : str, optional 

942 Prefix for all log messages. 

943 Default=''. 

944 

945 timeout : float, optional 

946 Timeout in seconds. Default=10. 

947 

948 verbose : int, optional 

949 An integer in [0, 1, 2, 3] to control the verbosity level. 

950 Default=3. 

951 

952 Returns 

953 ------- 

954 batch : dict 

955 Python dict representing the response's content. 

956 

957 Raises 

958 ------ 

959 requests.RequestException 

960 If there was a problem opening the URL. 

961 

962 ValueError 

963 If the response could not be decoded, was not json, or did not contain 

964 either 'id' (single result), or 'results' and 'count' (actual batch). 

965 

966 """ 

967 session = _requests_session() 

968 req = requests.Request( 

969 method="GET", url=query, headers={"Connection": "Keep-Alive"}) 

970 prepped = session.prepare_request(req) 

971 _print_if('{0}getting new batch: {1}'.format( 

972 prefix_msg, query), _DEBUG, verbose) 

973 try: 

974 resp = session.send(prepped, timeout=timeout) 

975 resp.raise_for_status() 

976 batch = resp.json() 

977 except Exception: 

978 _print_if('Could not get batch from {0}'.format(query), 

979 _ERROR, verbose, with_traceback=True) 

980 raise 

981 if 'id' in batch: 

982 batch = {'count': 1, 'results': [batch]} 

983 for key in ['results', 'count']: 

984 if batch.get(key) is None: 

985 msg = ('Could not find required key "{0}" ' 

986 'in batch retrieved from {1}'.format(key, query)) 

987 _print_if(msg, _ERROR, verbose) 

988 raise ValueError(msg) 

989 

990 return batch

993def _scroll_server_results(url, local_filter=_empty_filter, 

994 query_terms=None, max_results=None, 

995 batch_size=None, prefix_msg='', verbose=3): 

996 """Download list of metadata from Neurovault. 

997 

998 Parameters 

999 ---------- 

1000 url : str 

1001 The base url (without the filters) from which to get data. 

1002 

1003 local_filter : callable, optional 

1004 Used to filter the results based on their metadata: 

1005 must return True if the result is to be kept and False otherwise. 

1006 Is called with the dict containing the metadata as sole argument. 

1007 Default=_empty_filter. 

1008 

1009 query_terms : dict, sequence of pairs or None, optional 

1010 Key-value pairs to add to the base url in order to form query. 

1011 If ``None``, nothing is added to the url. 

1012 

1013 max_results : int or None, optional 

1014 Maximum number of results to fetch; if ``None``, all available data 

1015 that matches the query is fetched. 

1016 

1017 batch_size : int or None, optional 

1018 Neurovault returns the metadata for hits corresponding to a query 

1019 in batches. batch_size is used to choose the (maximum) number of 

1020 elements in a batch. If None, ``_DEFAULT_BATCH_SIZE`` is used. 

1021 

1022 prefix_msg : str, optional 

1023 Prefix for all log messages. Default=''. 

1024 

1025 verbose : int, optional (default=3) 

1026 An integer in [0, 1, 2, 3] to control the verbosity level. 

1027 

1028 Yields 

1029 ------ 

1030 result : dict 

1031 A result in the retrieved batch. 

1032 

1033 None 

1034 Once for each batch that could not be downloaded or decoded, 

1035 to indicate a failure. 

1036 

1037 """ 

1038 query = _append_filters_to_query(url, query_terms) 

1039 if batch_size is None: 

1040 batch_size = _DEFAULT_BATCH_SIZE 

1041 query = '{0}{1}limit={2}&offset={{0}}'.format( 

1042 query, ('&' if '?' in query else '?'), batch_size) 

1043 downloaded = 0 

1044 n_available = None 

1045 while(max_results is None or downloaded < max_results): 

1046 new_query = query.format(downloaded) 

1047 try: 

1048 batch = _get_batch(new_query, prefix_msg, verbose=verbose) 

1049 except Exception: 

1050 yield None 

1051 batch = None 

1052 if batch is not None: 

1053 batch_size = len(batch['results']) 

1054 downloaded += batch_size 

1055 _print_if('{0}batch size: {1}'.format(prefix_msg, batch_size), 

1056 _DEBUG, verbose) 

1057 if n_available is None: 

1058 n_available = batch['count'] 

1059 max_results = (n_available if max_results is None 

1060 else min(max_results, n_available)) 

1061 for result in batch['results']: 

1062 if local_filter(result): 

1063 yield result

1970def _scroll(download_params): 

1971 """Iterate over Neurovault data. 

1972 

1973 Relevant images and collections are loaded from local disk, then 

1974 from neurovault.org 

1975 

1976 Parameters 

1977 ---------- 

1978 download_params : dict 

1979 General information about download session, containing e.g. the 

1980 data directory (see `_read_download_params` and 

1981 `_prepare_download_params for details`) 

1982 

1983 Yields 

1984 ------ 

1985 image : dict 

1986 Metadata for an image. 

1987 

1988 collection : dict 

1989 Metadata for the corresponding collection. 

1990 

1991 Notes 

1992 ----- 

1993 Stops if: 

1994 - All available images have been fetched. 

1995 - Or a max number of images has been specified by user and 

1996 reached. 

1997 - Or too many downloads have failed in a row. 

1998 

1999 """ 

2000 scroll_modes = {'filtered': _scroll_filtered, 'explicit': _scroll_explicit} 

2001 if download_params['max_images'] == 0: 

2002 return 

2003 found = 0 

2004 

2005 if download_params['download_mode'] != 'overwrite': 

2006 for image, collection in _scroll_local(download_params): 

2007 found = len(download_params['visited_images']) 

2008 _print_progress(found, download_params, _DEBUG) 

2009 yield image, collection 

2010 if found == download_params['max_images']: 

2011 break 

2012 _print_if('{0} image{1} found on local disk.'.format( 

2013 ('No' if not found else found), ('s' if found > 1 else '')), 

2014 _INFO, download_params['verbose']) 

2015 

2016 if download_params['download_mode'] == 'offline': 

2017 return 

2018 if found == download_params['max_images']: 

2019 return 

2020 server_data = scroll_modes[download_params['scroll_mode']](download_params) 

2021 n_consecutive_fails = 0 

2022 for image, collection in server_data: 

2023 if image is None or collection is None: 

2024 n_consecutive_fails += 1 

2025 else: 

2026 n_consecutive_fails = 0 

2027 found += 1 

2028 _print_progress(found, download_params) 

2029 yield image, collection 

2030 

2031 if n_consecutive_fails >= download_params['max_consecutive_fails']: 

2032 warnings.warn('Neurovault download stopped early: ' 

2033 'too many downloads failed in a row ({0})'.format( 

2034 n_consecutive_fails)) 

2035 return 

2036 if found == download_params['max_images']: 

2037 return
