26def _standardize(signals, detrend=False, standardize='zscore'): 

27 """ Center and standardize a given signal (time is along first axis) 

28 

29 Parameters 

30 ---------- 

31 signals: numpy.ndarray 

32 Timeseries to standardize 

33 

34 detrend: bool 

35 if detrending of timeseries is requested 

36 

37 standardize: {'zscore', 'psc', True, False}, default is 'zscore' 

38 Strategy to standardize the signal. 

39 'zscore': the signal is z-scored. Timeseries are shifted 

40 to zero mean and scaled to unit variance. 

41 'psc': Timeseries are shifted to zero mean value and scaled 

42 to percent signal change (as compared to original mean signal). 

43 True : the signal is z-scored. Timeseries are shifted 

44 to zero mean and scaled to unit variance. 

45 False : Do not standardize the data. 

46 

47 Returns 

48 ------- 

49 std_signals: numpy.ndarray 

50 copy of signals, standardized. 

51 """ 

52 

53 if standardize not in [True, False, 'psc', 'zscore']: 

54 raise ValueError('{} is no valid standardize strategy.' 

55 .format(standardize)) 

56 

57 if detrend: 

58 signals = _detrend(signals, inplace=False) 

59 else: 

60 signals = signals.copy() 

61 

62 if standardize: 

63 if signals.shape[0] == 1: 

64 warnings.warn('Standardization of 3D signal has been requested but ' 

65 'would lead to zero values. Skipping.') 

66 return signals 

67 

68 elif (standardize == 'zscore') or (standardize is True): 

69 if not detrend: 

70 # remove mean if not already detrended 

71 signals = signals - signals.mean(axis=0) 

72 

73 std = signals.std(axis=0) 

74 std[std < np.finfo(np.float64).eps] = 1. # avoid numerical problems 

75 signals /= std 

76 

77 elif standardize == 'psc': 

78 mean_signal = signals.mean(axis=0) 

79 invalid_ix = np.absolute(mean_signal) < np.finfo(np.float64).eps 

80 signals = (signals - mean_signal) / np.absolute(mean_signal) 

81 signals *= 100 

82 

83 if np.any(invalid_ix): 

84 warnings.warn('psc standardization strategy is meaningless ' 

85 'for features that have a mean of 0. ' 

86 'These time series are set to 0.') 

87 signals[:, invalid_ix] = 0 

88 

89 return signals

92def _mean_of_squares(signals, n_batches=20): 

93 """Compute mean of squares for each signal. 

94 This function is equivalent to 

95 

96 var = np.copy(signals) 

97 var **= 2 

98 var = var.mean(axis=0) 

99 

100 but uses a lot less memory. 

101 

102 Parameters 

103 ---------- 

104 signals : numpy.ndarray, shape (n_samples, n_features) 

105 signal whose mean of squares must be computed. 

106 

107 n_batches : int, optional 

108 number of batches to use in the computation. Tweaking this value 

109 can lead to variation of memory usage and computation time. The higher 

110 the value, the lower the memory consumption. 

111 

112 """ 

113 # No batching for small arrays 

114 if signals.shape[1] < 500: 

115 n_batches = 1 

116 

117 # Fastest for C order 

118 var = np.empty(signals.shape[1]) 

119 for batch in gen_even_slices(signals.shape[1], n_batches): 

120 tvar = np.copy(signals[:, batch]) 

121 tvar **= 2 

122 var[batch] = tvar.mean(axis=0) 

123 

124 return var

127def _row_sum_of_squares(signals, n_batches=20): 

128 """Compute sum of squares for each signal. 

129 This function is equivalent to 

130 

131 signals **= 2 

132 signals = signals.sum(axis=0) 

133 

134 but uses a lot less memory. 

135 

136 Parameters 

137 ---------- 

138 signals : numpy.ndarray, shape (n_samples, n_features) 

139 signal whose sum of squares must be computed. 

140 

141 n_batches : int, optional 

142 number of batches to use in the computation. Tweaking this value 

143 can lead to variation of memory usage and computation time. The higher 

144 the value, the lower the memory consumption. 

145 

146 """ 

147 # No batching for small arrays 

148 if signals.shape[1] < 500: 

149 n_batches = 1 

150 

151 # Fastest for C order 

152 var = np.empty(signals.shape[1]) 

153 for batch in gen_even_slices(signals.shape[1], n_batches): 

154 var[batch] = np.sum(signals[:, batch] ** 2, 0) 

155 

156 return var

159def _detrend(signals, inplace=False, type="linear", n_batches=10): 

160 """Detrend columns of input array. 

161 

162 Signals are supposed to be columns of `signals`. 

163 This function is significantly faster than scipy.signal.detrend on this 

164 case and uses a lot less memory. 

165 

166 Parameters 

167 ---------- 

168 signals : numpy.ndarray 

169 This parameter must be two-dimensional. 

170 Signals to detrend. A signal is a column. 

171 

172 inplace : bool, optional 

173 Tells if the computation must be made inplace or not (default 

174 False). 

175 

176 type : str, optional 

177 Detrending type ("linear" or "constant"). 

178 See also scipy.signal.detrend. 

179 

180 n_batches : int, optional 

181 number of batches to use in the computation. Tweaking this value 

182 can lead to variation of memory usage and computation time. The higher 

183 the value, the lower the memory consumption. 

184 

185 Returns 

186 ------- 

187 detrended_signals: numpy.ndarray 

188 Detrended signals. The shape is that of 'signals'. 

189 

190 Notes 

191 ----- 

192 

193 If a signal of lenght 1 is given, it is returned unchanged. 

194 

195 """ 

196 signals = as_float_array(signals, copy=not inplace) 

197 if signals.shape[0] == 1: 

198 warnings.warn('Detrending of 3D signal has been requested but ' 

199 'would lead to zero values. Skipping.') 

200 return signals 

201 

202 signals -= np.mean(signals, axis=0) 

203 if type == "linear": 

204 # Keeping "signals" dtype avoids some type conversion further down, 

205 # and can save a lot of memory if dtype is single-precision. 

206 regressor = np.arange(signals.shape[0], dtype=signals.dtype) 

207 regressor -= regressor.mean() 

208 std = np.sqrt((regressor ** 2).sum()) 

209 # avoid numerical problems 

210 if not std < np.finfo(np.float64).eps: 

211 regressor /= std 

212 regressor = regressor[:, np.newaxis] 

213 

214 # No batching for small arrays 

215 if signals.shape[1] < 500: 

216 n_batches = 1 

217 

218 # This is fastest for C order. 

219 for batch in gen_even_slices(signals.shape[1], n_batches): 

220 signals[:, batch] -= np.dot(regressor[:, 0], signals[:, batch] 

221 ) * regressor 

222 return signals

225def _check_wn(btype, freq, nyq): 

226 wn = freq / float(nyq) 

227 if wn >= 1.: 

228 # results looked unstable when the critical frequencies are 

229 # exactly at the Nyquist frequency. See issue at SciPy 

230 # https://github.com/scipy/scipy/issues/6265. Before, SciPy 1.0.0 ("wn 

231 # should be btw 0 and 1"). But, after ("0 < wn < 1"). Due to unstable 

232 # results as pointed in the issue above. Hence, we forced the 

233 # critical frequencies to be slightly less than 1. but not 1. 

234 wn = 1 - 10 * np.finfo(1.).eps 

235 warnings.warn( 

236 'The frequency specified for the %s pass filter is ' 

237 'too high to be handled by a digital filter (superior to ' 

238 'nyquist frequency). It has been lowered to %.2f (nyquist ' 

239 'frequency).' % (btype, wn)) 

240 

241 if wn < 0.0: # equal to 0.0 is okay 

242 wn = np.finfo(1.).eps 

243 warnings.warn( 

244 'The frequency specified for the %s pass filter is ' 

245 'too low to be handled by a digital filter (must be non-negative).' 

246 ' It has been set to eps: %.5e' % (btype, wn)) 

247 

248 return wn

251def butterworth(signals, sampling_rate, low_pass=None, high_pass=None, 

252 order=5, copy=False): 

253 """ Apply a low-pass, high-pass or band-pass Butterworth filter 

254 

255 Apply a filter to remove signal below the `low` frequency and above the 

256 `high` frequency. 

257 

258 Parameters 

259 ---------- 

260 signals: numpy.ndarray (1D sequence or n_samples x n_sources) 

261 Signals to be filtered. A signal is assumed to be a column 

262 of `signals`. 

263 

264 sampling_rate: float 

265 Number of samples per time unit (sample frequency) 

266 

267 low_pass: float, optional 

268 If specified, signals above this frequency will be filtered out 

269 (low pass). This is -3dB cutoff frequency. 

270 

271 high_pass: float, optional 

272 If specified, signals below this frequency will be filtered out 

273 (high pass). This is -3dB cutoff frequency. 

274 

275 order: integer, optional 

276 Order of the Butterworth filter. When filtering signals, the 

277 filter has a decay to avoid ringing. Increasing the order 

278 sharpens this decay. Be aware that very high orders could lead 

279 to numerical instability. 

280 

281 copy: bool, optional 

282 If False, `signals` is modified inplace, and memory consumption is 

283 lower than for copy=True, though computation time is higher. 

284 

285 Returns 

286 ------- 

287 filtered_signals: numpy.ndarray 

288 Signals filtered according to the parameters 

289 """ 

290 if low_pass is None and high_pass is None: 

291 if copy: 

292 return signals.copy() 

293 else: 

294 return signals 

295 

296 if low_pass is not None and high_pass is not None \ 

297 and high_pass >= low_pass: 

298 raise ValueError( 

299 "High pass cutoff frequency (%f) is greater or equal" 

300 "to low pass filter frequency (%f). This case is not handled " 

301 "by this function." 

302 % (high_pass, low_pass)) 

303 

304 nyq = sampling_rate * 0.5 

305 

306 critical_freq = [] 

307 if high_pass is not None: 

308 btype = 'high' 

309 critical_freq.append(_check_wn(btype, high_pass, nyq)) 

310 

311 if low_pass is not None: 

312 btype = 'low' 

313 critical_freq.append(_check_wn(btype, low_pass, nyq)) 

314 

315 if len(critical_freq) == 2: 

316 btype = 'band' 

317 else: 

318 critical_freq = critical_freq[0] 

319 

320 b, a = sp_signal.butter(order, critical_freq, btype=btype, output='ba') 

321 if signals.ndim == 1: 

322 # 1D case 

323 output = sp_signal.filtfilt(b, a, signals) 

324 if copy: # filtfilt does a copy in all cases. 

325 signals = output 

326 else: 

327 signals[...] = output 

328 else: 

329 if copy: 

330 # No way to save memory when a copy has been requested, 

331 # because filtfilt does out-of-place processing 

332 signals = sp_signal.filtfilt(b, a, signals, axis=0) 

333 else: 

334 # Lesser memory consumption, slower. 

335 for timeseries in signals.T: 

336 timeseries[:] = sp_signal.filtfilt(b, a, timeseries) 

337 

338 # results returned in-place 

339 

340 return signals

343def high_variance_confounds(series, n_confounds=5, percentile=2., 

344 detrend=True): 

345 """ Return confounds time series extracted from series with highest 

346 variance. 

347 

348 Parameters 

349 ---------- 

350 series: numpy.ndarray 

351 Timeseries. A timeseries is a column in the "series" array. 

352 shape (sample number, feature number) 

353 

354 n_confounds: int, optional 

355 Number of confounds to return 

356 

357 percentile: float, optional 

358 Highest-variance series percentile to keep before computing the 

359 singular value decomposition, 0. <= `percentile` <= 100. 

360 series.shape[0] * percentile / 100 must be greater than n_confounds 

361 

362 detrend: bool, optional 

363 If True, detrend timeseries before processing. 

364 

365 Returns 

366 ------- 

367 v: numpy.ndarray 

368 highest variance confounds. Shape: (samples, n_confounds) 

369 

370 Notes 

371 ----- 

372 This method is related to what has been published in the literature 

373 as 'CompCor' (Behzadi NeuroImage 2007). 

374 

375 The implemented algorithm does the following: 

376 

377 - compute sum of squares for each time series (no mean removal) 

378 - keep a given percentile of series with highest variances (percentile) 

379 - compute an svd of the extracted series 

380 - return a given number (n_confounds) of series from the svd with 

381 highest singular values. 

382 

383 See also 

384 -------- 

385 nilearn.image.high_variance_confounds 

386 """ 

387 

388 if detrend: 

389 series = _detrend(series) # copy 

390 

391 # Retrieve the voxels|features with highest variance 

392 

393 # Compute variance without mean removal. 

394 var = _mean_of_squares(series) 

395 var_thr = np.nanpercentile(var, 100. - percentile) 

396 series = series[:, var > var_thr] # extract columns (i.e. features) 

397 # Return the singular vectors with largest singular values 

398 # We solve the symmetric eigenvalue problem here, increasing stability 

399 s, u = linalg.eigh(series.dot(series.T) / series.shape[0]) 

400 ix_ = np.argsort(s)[::-1] 

401 u = u[:, ix_[:n_confounds]].copy() 

402 return u

411 data = data.astype(np.float32) 

412 return data

415@rename_parameters({'sessions': 'runs'}, '0.9.0') 

416def clean(signals, runs=None, detrend=True, standardize='zscore', 

417 confounds=None, standardize_confounds=True, filter='butterworth', 

418 low_pass=None, high_pass=None, t_r=2.5, ensure_finite=False): 

419 """Improve SNR on masked fMRI signals. 

420 

421 This function can do several things on the input signals, in 

422 the following order: 

423 

424 - detrend 

425 - low- and high-pass filter 

426 - remove confounds 

427 - standardize 

428 

429 Low-pass filtering improves specificity. 

430 

431 High-pass filtering should be kept small, to keep some 

432 sensitivity. 

433 

434 Filtering is only meaningful on evenly-sampled signals. 

435 

436 According to Lindquist et al. (2018), removal of confounds will be done 

437 orthogonally to temporal filters (low- and/or high-pass filters), if both 

438 are specified. 

439 

440 Parameters 

441 ---------- 

442 signals: numpy.ndarray 

443 Timeseries. Must have shape (instant number, features number). 

444 This array is not modified. 

445 

446 runs : numpy array, optional 

447 Add a run level to the cleaning process. Each run will be 

448 cleaned independently. Must be a 1D array of n_samples elements. 

449 'runs' replaces 'sessions' after release 0.9.0. 

450 Using 'session' will result in an error after release 0.9.0. 

451 

452 confounds: numpy.ndarray, str, DataFrame or list of 

453 Confounds timeseries. Shape must be 

454 (instant number, confound number), or just (instant number,) 

455 The number of time instants in signals and confounds must be 

456 identical (i.e. signals.shape[0] == confounds.shape[0]). 

457 If a string is provided, it is assumed to be the name of a csv file 

458 containing signals as columns, with an optional one-line header. 

459 If a list is provided, all confounds are removed from the input 

460 signal, as if all were in the same array. 

461 

462 t_r: float 

463 Repetition time, in second (sampling period). Set to None if not. 

464 

465 filter: {'butterworth', 'cosine', False} 

466 Filtering methods. 

467 'butterworth': perform butterworth filtering. 

468 'cosine': generate discrete cosine transformation drift terms. 

469 False : Do not perform filtering. 

470 

471 low_pass, high_pass: float 

472 Respectively high and low cutoff frequencies, in Hertz. 

473 `low_pass` is not implemented for filter='cosine'. 

474 

475 detrend: bool 

476 If detrending should be applied on timeseries (before 

477 confound removal) 

478 

479 standardize: {'zscore', 'psc', False}, default is 'zscore' 

480 Strategy to standardize the signal. 

481 'zscore': the signal is z-scored. Timeseries are shifted 

482 to zero mean and scaled to unit variance. 

483 'psc': Timeseries are shifted to zero mean value and scaled 

484 to percent signal change (as compared to original mean signal). 

485 False : Do not standardize the data. 

486 

487 standardize_confounds: boolean, optional, default is True 

488 If standardize_confounds is True, the confounds are z-scored: 

489 their mean is put to 0 and their variance to 1 in the time dimension. 

490 

491 ensure_finite: bool 

492 If True, the non-finite values (NANs and infs) found in the data 

493 will be replaced by zeros. 

494 

495 Returns 

496 ------- 

497 cleaned_signals: numpy.ndarray 

498 Input signals, cleaned. Same shape as `signals`. 

499 

500 Notes 

501 ----- 

502 Confounds removal is based on a projection on the orthogonal 

503 of the signal space. See `Friston, K. J., A. P. Holmes, 

504 K. J. Worsley, J.-P. Poline, C. D. Frith, et R. S. J. Frackowiak. 

505 "Statistical Parametric Maps in Functional Imaging: A General 

506 Linear Approach". Human Brain Mapping 2, no 4 (1994): 189-210. 

507 <http://dx.doi.org/10.1002/hbm.460020402>`_ 

508 

509 Orthogonalization between temporal filters and confound removal is based on 

510 suggestions in `Lindquist, M., Geuter, S., Wager, T., & Caffo, B. (2018). 

511 Modular preprocessing pipelines can reintroduce artifacts into fMRI data. 

512 bioRxiv, 407676. <http://dx.doi.org/10.1101/407676>`_ 

513 

514 See Also 

515 -------- 

516 nilearn.image.clean_img 

517 """ 

518 # Read confounds and signals 

519 signals, confounds = _sanitize_inputs(signals, confounds, ensure_finite) 

520 use_filter = _check_filter_parameters(filter, low_pass, high_pass, t_r) 

521 # Restrict the signal to the orthogonal of the confounds 

522 if runs is not None: 

523 signals = _process_runs(signals, runs, detrend, standardize, 

524 confounds, low_pass, high_pass, t_r) 

525 

526 # Detrend 

527 # Detrend and filtering should apply to confounds, if confound presents 

528 # keep filters orthogonal (according to Lindquist et al. (2018)) 

529 if detrend: 

530 mean_signals = signals.mean(axis=0) 

531 signals = _standardize(signals, standardize=False, detrend=detrend) 

532 if confounds is not None: 

533 confounds = _standardize(confounds, standardize=False, 

534 detrend=detrend) 

535 if use_filter: 

536 # check if filter parameters are satisfied and filter according to the strategy 

537 signals, confounds = _filter_signal(signals, confounds, filter, 

538 low_pass, high_pass, t_r) 

539 

540 # Remove confounds 

541 if confounds is not None: 

542 confounds = _standardize(confounds, standardize=standardize_confounds, 

543 detrend=False) 

544 if not standardize_confounds: 

545 # Improve numerical stability by controlling the range of 

546 # confounds. We don't rely on _standardize as it removes any 

547 # constant contribution to confounds. 

548 confound_max = np.max(np.abs(confounds), axis=0) 

549 confound_max[confound_max == 0] = 1 

550 confounds /= confound_max 

551 

552 # Pivoting in qr decomposition was added in scipy 0.10 

553 Q, R, _ = linalg.qr(confounds, mode='economic', pivoting=True) 

554 Q = Q[:, np.abs(np.diag(R)) > np.finfo(np.float64).eps * 100.] 

555 signals -= Q.dot(Q.T).dot(signals) 

556 

557 # Standardize 

558 if detrend and (standardize == 'psc'): 

559 # If the signal is detrended, we have to know the original mean 

560 # signal to calculate the psc. 

561 signals = _standardize(signals + mean_signals, standardize=standardize, 

562 detrend=False) 

563 else: 

564 signals = _standardize(signals, standardize=standardize, 

565 detrend=False) 

566 

567 return signals

570def _filter_signal(signals, confounds, filter, low_pass, high_pass, t_r): 

571 '''Filter signal based on provided strategy.''' 

572 if filter == 'butterworth': 

573 signals = butterworth(signals, sampling_rate=1. / t_r, 

574 low_pass=low_pass, high_pass=high_pass) 

575 if confounds is not None: 

576 # Apply low- and high-pass filters to keep filters orthogonal 

577 # (according to Lindquist et al. (2018)) 

578 confounds = butterworth(confounds, sampling_rate=1. / t_r, 

579 low_pass=low_pass, high_pass=high_pass) 

580 elif filter == 'cosine': 

581 from .glm.first_level.design_matrix import _cosine_drift 

582 frame_times = np.arange(signals.shape[0]) * t_r 

583 cosine_drift = _cosine_drift(high_pass, frame_times) 

584 if confounds is None: 

585 confounds = cosine_drift.copy() 

586 else: 

587 confounds = np.hstack((confounds, cosine_drift)) 

588 return signals, confounds

591def _process_runs(signals, runs, detrend, standardize, confounds, 

592 low_pass, high_pass, t_r): 

593 """Process each run independently.""" 

594 if len(runs) != len(signals): 

595 raise ValueError( 

596 ( 

597 'The length of the session vector (%i) ' 

598 'does not match the length of the signals (%i)' 

599 ) % (len(runs), len(signals)) 

600 ) 

601 for run in np.unique(runs): 

602 session_confounds = None 

603 if confounds is not None: 

604 session_confounds = confounds[runs == run] 

605 signals[runs == run, :] = \ 

606 clean(signals[runs == run], 

607 detrend=detrend, standardize=standardize, 

608 confounds=session_confounds, low_pass=low_pass, 

609 high_pass=high_pass, t_r=t_r) 

610 return signals

613def _sanitize_inputs(signals, confounds, ensure_finite): 

614 """Clean up signals and confounds before processing.""" 

615 signals = _sanitize_signals(signals, ensure_finite) 

616 

617 if not isinstance(confounds, 

618 (list, tuple, str, np.ndarray, pd.DataFrame, 

619 type(None))): 

620 raise TypeError("confounds keyword has an unhandled type: %s" 

621 % confounds.__class__) 

622 if confounds is None: 

623 return signals, confounds 

624 

625 if not isinstance(confounds, (list, tuple)): 

626 confounds = (confounds, ) 

627 all_confounds = [] 

628 n_time = signals.shape[0] 

629 for confound in confounds: 

630 confound = _sanitize_confound_dtype(n_time, confound) 

631 all_confounds.append(confound) 

632 confounds = np.hstack(all_confounds) 

633 return signals, _ensure_float(confounds)

636def _sanitize_confound_dtype(n_signal, confound): 

637 """Check confound is the correct datatype.""" 

638 if isinstance(confound, pd.DataFrame): 

639 confound = confound.values 

640 if isinstance(confound, str): 

641 filename = confound 

642 confound = csv_to_array(filename) 

643 if np.isnan(confound.flat[0]): 

644 # There may be a header 

645 confound = csv_to_array(filename, skip_header=1) 

646 if confound.shape[0] != n_signal: 

647 raise ValueError( 

648 "Confound signal has an incorrect length" 

649 "Signal length: {0}; confound length: {1}".format( 

650 n_signal, confound.shape[0]) 

651 ) 

652 elif isinstance(confound, np.ndarray): 

653 if confound.ndim == 1: 

654 confound = np.atleast_2d(confound).T 

655 elif confound.ndim != 2: 

656 raise ValueError("confound array has an incorrect number " 

657 "of dimensions: %d" % confound.ndim) 

658 if confound.shape[0] != n_signal: 

659 raise ValueError( 

660 "Confound signal has an incorrect length" 

661 "Signal length: {0}; confound length: {1}".format(

671def _check_filter_parameters(filter, low_pass, high_pass, t_r): 

672 """Check all filter related parameters are set correcly.""" 

673 if not filter: 

674 if any(isinstance(item, float) for item in [low_pass, high_pass]): 

675 warnings.warn( 

676 "No filter type selected but cutoff frequency provided." 

677 "Will not perform filtering." 

678 )

680 elif filter in availiable_filters: 

681 if filter == 'cosine' and not all(isinstance(item, float) 

682 for item in [t_r, high_pass]):

686 "t_r='{0}', high_pass='{1}'".format(t_r, high_pass) 

687 )

697 raise TypeError( 

698 "high/low pass must be float or None but you provided " 

699 "high_pass='{0}', low_pass='{1}'" 

700 .format(high_pass, low_pass) 

701 ) 

702 return True 

703 else: 

704 raise ValueError("Filter method {} not implemented.".format(filter))

709 if not isinstance(ensure_finite, bool): 

710 raise ValueError("'ensure_finite' must be boolean type True or False " 

711 "but you provided ensure_finite={0}"
