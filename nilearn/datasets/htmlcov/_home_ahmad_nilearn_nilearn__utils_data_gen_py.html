24def generate_mni_space_img(n_scans=1, res=30, random_state=0, mask_dilation=2):

49def generate_regions_ts(n_features, n_regions, 

50 overlap=0, 

51 rand_gen=None, 

52 window="boxcar"): 

53 """Generate some regions as timeseries. 

54 

55 Parameters 

56 ---------- 

57 n_features : int 

58 Number of features. 

59 

60 n_regions : int 

61 Number of regions. 

62 

63 overlap : int, optional 

64 Number of overlapping voxels between two regions (more or less). 

65 Default=0. 

66 

67 window : str, optional 

68 Name of a window in scipy.signal. e.g. "hamming". 

69 Default='boxcar'. 

70 

71 Returns 

72 ------- 

73 regions : numpy.ndarray 

74 Regions, nepresented as signals. 

75 shape (n_features, n_regions) 

76 

77 """ 

78 if rand_gen is None: 

79 rand_gen = np.random.RandomState(0) 

80 if window is None: 

81 window = "boxcar" 

82 

83 assert(n_features > n_regions) 

84 

85 # Compute region boundaries indices. 

86 # Start at 1 to avoid getting an empty region 

87 boundaries = np.zeros(n_regions + 1) 

88 boundaries[-1] = n_features 

89 boundaries[1:-1] = rand_gen.permutation(np.arange(1, n_features) 

90 )[:n_regions - 1] 

91 boundaries.sort() 

92 

93 regions = np.zeros((n_regions, n_features), order="C") 

94 overlap_end = int((overlap + 1) / 2.) 

95 overlap_start = int(overlap / 2.) 

96 for n in range(len(boundaries) - 1): 

97 start = int(max(0, boundaries[n] - overlap_start)) 

98 end = int(min(n_features, boundaries[n + 1] + overlap_end)) 

99 win = scipy.signal.get_window(window, end - start) 

100 win /= win.mean() # unity mean 

101 regions[n, start:end] = win 

102 

103 return regions

106def generate_maps(shape, n_regions, overlap=0, border=1, 

107 window="boxcar", rand_gen=None, affine=np.eye(4)): 

108 """Generate a 4D volume containing several maps. 

109 

110 Parameters 

111 ---------- 

112 n_regions : int 

113 Number of regions to generate. 

114 

115 overlap : int, optional 

116 Approximate number of voxels common to two neighboring regions. 

117 Default=0. 

118 

119 window : str, optional 

120 Name of a window in scipy.signal. Used to get non-uniform regions. 

121 Default='boxcar'. 

122 

123 ran_gen : None or numpy.random.RandomState, optional 

124 Random generator to use for generation. Default=None. 

125 

126 affine : Numpy array, optional 

127 Affine transformation to use. Default=np.eye(4). 

128 

129 border : int, optional 

130 Number of background voxels on each side of the 3D volumes. 

131 Default=1. 

132 

133 Returns 

134 ------- 

135 maps : nibabel.Nifti1Image 

136 4D array, containing maps. 

137 

138 mask_img : nibabel.Nifti1Image 

139 3D, The mask outside of which maps are 0. 

140 

141 """ 

142 mask = np.zeros(shape, dtype=np.int8) 

143 mask[border:-border, border:-border, border:-border] = 1 

144 ts = generate_regions_ts(mask.sum(), n_regions, overlap=overlap, 

145 rand_gen=rand_gen, window=window) 

146 mask_img = nibabel.Nifti1Image(mask, affine) 

147 return masking.unmask(ts, mask_img), mask_img

150def generate_labeled_regions(shape, n_regions, rand_gen=None, labels=None, 

151 affine=np.eye(4), dtype=int): 

152 """Generate a 3D volume with labeled regions. 

153 

154 Parameters 

155 ---------- 

156 shape : tuple 

157 Shape of returned array. 

158 

159 n_regions : int 

160 Number of regions to generate. By default (if "labels" is None), 

161 add a background with value zero. 

162 

163 labels : iterable, optional 

164 Labels to use for each zone. If provided, n_regions is unused. 

165 

166 rand_gen : numpy.random.RandomState 

167 Random generator to use for generation. 

168 

169 affine : numpy.ndarray, optional 

170 Affine of returned image. Default=np.eye(4). 

171 

172 dtype : type, optional 

173 Data type of image. Default=int. 

174 

175 Returns 

176 ------- 

177 regions : nibabel.Nifti1Image 

178 Data has shape "shape", containing region labels. 

179 

180 """ 

181 n_voxels = shape[0] * shape[1] * shape[2] 

182 if labels is None: 

183 labels = range(0, n_regions + 1) 

184 n_regions += 1 

185 else: 

186 n_regions = len(labels) 

187 

188 regions = generate_regions_ts(n_voxels, n_regions, rand_gen=rand_gen) 

189 # replace weights with labels 

190 for n, row in zip(labels, regions): 

191 row[row > 0] = n 

192 data = np.zeros(shape, dtype=dtype) 

193 data[np.ones(shape, dtype=bool)] = regions.sum(axis=0).T 

194 return nibabel.Nifti1Image(data, affine)

212def generate_fake_fmri(shape=(10, 11, 12), length=17, kind="noise", 

213 affine=np.eye(4), n_blocks=None, block_size=None, 

214 block_type='classification', 

215 rand_gen=np.random.RandomState(0)): 

216 """Generate a signal which can be used for testing. 

217 

218 The return value is a 4D array, representing 3D volumes along time. 

219 Only the voxels in the center are non-zero, to mimic the presence of 

220 brain voxels in real signals. Setting n_blocks to an integer generates 

221 condition blocks, the remaining of the timeseries corresponding 

222 to 'rest' or 'baseline' condition. 

223 

224 Parameters 

225 ---------- 

226 shape : tuple, optional 

227 Shape of 3D volume. Default=(10, 11, 12). 

228 

229 length : int, optional 

230 Number of time instants. Default=17. 

231 

232 kind : string, optional 

233 Kind of signal used as timeseries. 

234 "noise": uniformly sampled values in [0..255] 

235 "step": 0.5 for the first half then 1. 

236 Default='noise'. 

237 

238 affine : numpy.ndarray, optional 

239 Affine of returned images. Default=np.eye(4). 

240 

241 n_blocks : int or None, optional 

242 Number of condition blocks. 

243 

244 block_size : int or None, optional 

245 Number of timepoints in a block. Used only if n_blocks is not 

246 None. Defaults to 3 if n_blocks is not None. 

247 

248 block_type : str, optional 

249 Defines if the returned target should be used for 

250 'classification' or 'regression'. 

251 Default='classification'. 

252 

253 rand_gen : numpy.random.RandomState, optional 

254 Random generator to use for generation. 

255 Default=np.random.RandomState(0). 

256 

257 Returns 

258 ------- 

259 fmri : nibabel.Nifti1Image 

260 Fake fmri signal. 

261 shape: shape + (length,) 

262 

263 mask : nibabel.Nifti1Image 

264 Mask giving non-zero voxels. 

265 

266 target : numpy.ndarray 

267 Classification or regression target. Shape of number of 

268 time points (length). Returned only if n_blocks is not None. 

269 

270 """ 

271 full_shape = shape + (length, ) 

272 fmri = np.zeros(full_shape) 

273 # Fill central voxels timeseries with random signals 

274 width = [s // 2 for s in shape] 

275 shift = [s // 4 for s in shape] 

276 

277 if kind == "noise": 

278 signals = rand_gen.randint(256, size=(width + [length])) 

279 elif kind == "step": 

280 signals = np.ones(width + [length]) 

281 signals[..., :length // 2] = 0.5 

282 else: 

283 raise ValueError("Unhandled value for parameter 'kind'") 

284 

285 fmri[shift[0]:shift[0] + width[0], 

286 shift[1]:shift[1] + width[1], 

287 shift[2]:shift[2] + width[2], 

288 :] = signals 

289 

290 mask = np.zeros(shape) 

291 mask[shift[0]:shift[0] + width[0], 

292 shift[1]:shift[1] + width[1], 

293 shift[2]:shift[2] + width[2]] = 1 

294 

295 if n_blocks is None: 

296 return (nibabel.Nifti1Image(fmri, affine), 

297 nibabel.Nifti1Image(mask, affine)) 

298 

299 block_size = 3 if block_size is None else block_size 

300 flat_fmri = fmri[mask.astype(bool)] 

301 flat_fmri /= np.abs(flat_fmri).max() 

302 target = np.zeros(length, dtype=int) 

303 rest_max_size = (length - (n_blocks * block_size)) // n_blocks 

304 if rest_max_size < 0: 

305 raise ValueError( 

306 '%s is too small ' 

307 'to put %s blocks of size %s' % ( 

308 length, n_blocks, block_size)) 

309 t_start = 0 

310 if rest_max_size > 0: 

311 t_start = rand_gen.randint(0, rest_max_size, 1)[0] 

312 for block in range(n_blocks): 

313 if block_type == 'classification': 

314 # Select a random voxel and add some signal to the background 

315 voxel_idx = rand_gen.randint(0, flat_fmri.shape[0], 1)[0] 

316 trials_effect = (rand_gen.random_sample(block_size) + 1) * 3. 

317 else: 

318 # Select the voxel in the image center and add some signal 

319 # that increases with each block 

320 voxel_idx = flat_fmri.shape[0] // 2 

321 trials_effect = ( 

322 rand_gen.random_sample(block_size) + 1) * block 

323 t_rest = 0 

324 if rest_max_size > 0: 

325 t_rest = rand_gen.randint(0, rest_max_size, 1)[0] 

326 flat_fmri[voxel_idx, t_start:t_start + block_size] += trials_effect 

327 target[t_start:t_start + block_size] = block + 1 

328 t_start += t_rest + block_size 

329 target = target if block_type == 'classification' \ 

330 else target.astype(np.float64) 

331 fmri = np.zeros(fmri.shape) 

332 fmri[mask.astype(bool)] = flat_fmri 

333 return (nibabel.Nifti1Image(fmri, affine), 

334 nibabel.Nifti1Image(mask, affine), target)

376def generate_signals_from_precisions(precisions, 

377 min_n_samples=50, max_n_samples=100, 

378 random_state=0): 

379 """Generate timeseries according to some given precision matrices. 

380 

381 Signals all have zero mean. 

382 

383 Parameters 

384 ---------- 

385 precisions : list of numpy.ndarray 

386 List of precision matrices. Every matrix must be square (with the same 

387 size) and positive definite. The output of 

388 generate_group_sparse_gaussian_graphs() can be used here. 

389 

390 min_samples, max_samples : int, optional 

391 The number of samples drawn for each timeseries is taken at random 

392 between these two numbers. Defaults are 50 and 100. 

393 

394 random_state : int, optional 

395 Random state. Default=0. 

396 

397 Returns 

398 ------- 

399 signals : list of numpy.ndarray 

400 Output signals. signals[n] corresponds to precisions[n], and has shape 

401 (sample number, precisions[n].shape[0]). 

402 

403 """ 

404 random_state = check_random_state(random_state) 

405 

406 signals = [] 

407 n_samples = random_state.randint(min_n_samples, high=max_n_samples, 

408 size=len(precisions)) 

409 

410 mean = np.zeros(precisions[0].shape[0]) 

411 for n, prec in zip(n_samples, precisions): 

412 signals.append(random_state.multivariate_normal(mean, 

413 np.linalg.inv(prec), 

414 (n,))) 

415 return signals

418def generate_group_sparse_gaussian_graphs( 

419 n_subjects=5, n_features=30, min_n_samples=30, max_n_samples=50, 

420 density=0.1, random_state=0, verbose=0): 

421 """Generate signals drawn from a sparse Gaussian graphical model. 

422 

423 Parameters 

424 ---------- 

425 n_subjects : int, optional 

426 Number of subjects. Default=5. 

427 

428 n_features : int, optional 

429 Number of signals per subject to generate. Default=30. 

430 

431 density : float, optional 

432 Density of edges in graph topology. Default=0.1. 

433 

434 min_n_samples, max_n_samples : int, optional 

435 Each subject have a different number of samples, between these two 

436 numbers. All signals for a given subject have the same number of 

437 samples. Defaults are 30 and 50. 

438 

439 random_state : int or numpy.random.RandomState instance, optional 

440 Random number generator, or seed. Default=0. 

441 

442 verbose : int, optional 

443 Verbosity level (0 means no message). Default=0. 

444 

445 Returns 

446 ------- 

447 subjects : list of numpy.ndarray, shape for each (n_samples, n_features) 

448 subjects[n] is the signals for subject n. They are provided as a numpy 

449 len(subjects) = n_subjects. n_samples varies according to the subject. 

450 

451 precisions : list of numpy.ndarray 

452 Precision matrices. 

453 

454 topology : numpy.ndarray 

455 Binary array giving the graph topology used for generating covariances 

456 and signals. 

457 

458 """ 

459 random_state = check_random_state(random_state) 

460 # Generate topology (upper triangular binary matrix, with zeros on the 

461 # diagonal) 

462 topology = np.empty((n_features, n_features)) 

463 topology[:, :] = np.triu(( 

464 random_state.randint(0, high=int(1. / density), 

465 size=n_features * n_features) 

466 ).reshape(n_features, n_features) == 0, k=1) 

467 

468 # Generate edges weights on topology 

469 precisions = [] 

470 mask = topology > 0 

471 for _ in range(n_subjects): 

472 

473 # See also sklearn.datasets.samples_generator.make_sparse_spd_matrix 

474 prec = topology.copy() 

475 prec[mask] = random_state.uniform(low=.1, high=.8, size=(mask.sum())) 

476 prec += np.eye(prec.shape[0]) 

477 prec = np.dot(prec.T, prec) 

478 

479 # Assert precision matrix is spd 

480 np.testing.assert_almost_equal(prec, prec.T) 

481 eigenvalues = np.linalg.eigvalsh(prec) 

482 if eigenvalues.min() < 0: 

483 raise ValueError("Failed generating a positive definite precision " 

484 "matrix. Decreasing n_features can help solving " 

485 "this problem.") 

486 precisions.append(prec) 

487 

488 # Returns the topology matrix of precision matrices. 

489 topology += np.eye(*topology.shape) 

490 topology = np.dot(topology.T, topology) 

491 topology = topology > 0 

492 assert(np.all(topology == topology.T)) 

493 logger.log("Sparsity: {0:f}".format( 

494 1. * topology.sum() / (topology.shape[0] ** 2)), 

495 verbose=verbose) 

496 

497 # Generate temporal signals 

498 signals = generate_signals_from_precisions(precisions, 

499 min_n_samples=min_n_samples, 

500 max_n_samples=max_n_samples, 

501 random_state=random_state) 

502 return signals, precisions, topology
