47def high_variance_confounds(imgs, n_confounds=5, percentile=2., 

48 detrend=True, mask_img=None): 

49 """ Return confounds signals extracted from input signals with highest 

50 variance. 

51 

52 Parameters 

53 ---------- 

54 imgs : Niimg-like object 

55 4D image. 

56 See http://nilearn.github.io/manipulating_images/input_output.html. 

57 

58 mask_img : Niimg-like object 

59 If not provided, all voxels are used. 

60 If provided, confounds are extracted from voxels inside the mask. 

61 See http://nilearn.github.io/manipulating_images/input_output.html. 

62 

63 n_confounds : :obj:`int`, optional 

64 Number of confounds to return. Default=5. 

65 

66 percentile : :obj:`float`, optional 

67 Highest-variance signals percentile to keep before computing the 

68 singular value decomposition, 0. <= `percentile` <= 100. 

69 `mask_img.sum() * percentile / 100` must be greater than `n_confounds`. 

70 Default=2. 

71 

72 detrend : :obj:`bool`, optional 

73 If True, detrend signals before processing. Default=True. 

74 

75 Returns 

76 ------- 

77 :class:`numpy.ndarray` 

78 Highest variance confounds. Shape: *(number_of_scans, n_confounds)*. 

79 

80 Notes 

81 ------ 

82 This method is related to what has been published in the literature 

83 as 'CompCor' (Behzadi NeuroImage 2007). 

84 

85 The implemented algorithm does the following: 

86 

87 - Computes the sum of squares for each signal (no mean removal). 

88 - Keeps a given percentile of signals with highest variance (percentile). 

89 - Computes an SVD of the extracted signals. 

90 - Returns a given number (n_confounds) of signals from the SVD with 

91 highest singular values. 

92 

93 See also 

94 -------- 

95 nilearn.signal.high_variance_confounds 

96 

97 """ 

98 from .. import masking 

99 

100 if mask_img is not None: 

101 sigs = masking.apply_mask(imgs, mask_img) 

102 else: 

103 # Load the data only if it doesn't need to be masked 

104 imgs = check_niimg_4d(imgs) 

105 sigs = as_ndarray(get_data(imgs)) 

106 # Not using apply_mask here saves memory in most cases. 

107 del imgs # help reduce memory consumption 

108 sigs = np.reshape(sigs, (-1, sigs.shape[-1])).T 

109 

110 return signal.high_variance_confounds(sigs, n_confounds=n_confounds, 

111 percentile=percentile, 

112 detrend=detrend)

115def _fast_smooth_array(arr): 

116 """Simple smoothing which is less computationally expensive than 

117 applying a Gaussian filter. 

118 

119 Only the first three dimensions of the array will be smoothed. The 

120 filter uses [0.2, 1, 0.2] weights in each direction and use a 

121 normalisation to preserve the local average value. 

122 

123 Parameters 

124 ---------- 

125 arr : :class:`numpy.ndarray` 

126 4D array, with image number as last dimension. 3D arrays are 

127 also accepted. 

128 

129 Returns 

130 ------- 

131 :class:`numpy.ndarray` 

132 Smoothed array. 

133 

134 Notes 

135 ----- 

136 Rather than calling this function directly, users are encouraged 

137 to call the high-level function :func:`smooth_img` with 

138 `fwhm='fast'`. 

139 

140 """ 

141 neighbor_weight = 0.2 

142 # 6 neighbors in 3D if not on an edge 

143 nb_neighbors = 6 

144 # This scale ensures that a uniform array stays uniform 

145 # except on the array edges 

146 scale = 1 + nb_neighbors * neighbor_weight 

147 

148 # Need to copy because the smoothing is done in multiple statements 

149 # and there does not seem to be an easy way to do it in place 

150 smoothed_arr = arr.copy() 

151 weighted_arr = neighbor_weight * arr 

152 

153 smoothed_arr[:-1] += weighted_arr[1:] 

154 smoothed_arr[1:] += weighted_arr[:-1] 

155 smoothed_arr[:, :-1] += weighted_arr[:, 1:] 

156 smoothed_arr[:, 1:] += weighted_arr[:, :-1] 

157 smoothed_arr[:, :, :-1] += weighted_arr[:, :, 1:] 

158 smoothed_arr[:, :, 1:] += weighted_arr[:, :, :-1] 

159 smoothed_arr /= scale 

160 

161 return smoothed_arr

164def _smooth_array(arr, affine, fwhm=None, ensure_finite=True, copy=True): 

165 """Smooth images by applying a Gaussian filter. 

166 

167 Apply a Gaussian filter along the three first dimensions of `arr`. 

168 

169 Parameters 

170 ---------- 

171 arr : :class:`numpy.ndarray` 

172 4D array, with image number as last dimension. 3D arrays are also 

173 accepted. 

174 

175 affine : :class:`numpy.ndarray` 

176 (4, 4) matrix, giving affine transformation for image. (3, 3) matrices 

177 are also accepted (only these coefficients are used). 

178 If `fwhm='fast'`, the affine is not used and can be None. 

179 

180 fwhm : scalar, :class:`numpy.ndarray`/:obj:`tuple`/:obj:`list`, 'fast' or None, optional 

181 Smoothing strength, as a full-width at half maximum, in millimeters. 

182 If a nonzero scalar is given, width is identical in all 3 directions. 

183 A :class:`numpy.ndarray`, :obj:`tuple`, or :obj:`list` must have 3 elements, 

184 giving the FWHM along each axis. 

185 If any of the elements is zero or None, smoothing is not performed 

186 along that axis. 

187 If `fwhm='fast'`, a fast smoothing will be performed with a filter 

188 [0.2, 1, 0.2] in each direction and a normalisation 

189 to preserve the local average value. 

190 If fwhm is None, no filtering is performed (useful when just removal 

191 of non-finite values is needed). 

192 

193 ensure_finite : :obj:`bool`, optional 

194 If True, replace every non-finite values (like NaNs) by zero before 

195 filtering. Default=True. 

196 

197 copy : :obj:`bool`, optional 

198 If True, input array is not modified. True by default: the filtering 

199 is not performed in-place. Default=True. 

200 

201 Returns 

202 ------- 

203 :class:`numpy.ndarray` 

204 Filtered `arr`. 

205 

206 Notes 

207 ----- 

208 This function is most efficient with arr in C order. 

209 

210 """ 

211 # Here, we have to investigate use cases of fwhm. Particularly, if fwhm=0. 

212 # See issue #1537 

213 if isinstance(fwhm, (int, float)) and (fwhm == 0.0): 

214 warnings.warn("The parameter 'fwhm' for smoothing is specified " 

215 "as {0}. Setting it to None " 

216 "(no smoothing will be performed)" 

217 .format(fwhm)) 

218 fwhm = None 

219 if arr.dtype.kind == 'i': 

220 if arr.dtype == np.int64: 

221 arr = arr.astype(np.float64) 

222 else: 

223 arr = arr.astype(np.float32) # We don't need crazy precision. 

224 if copy: 

225 arr = arr.copy() 

226 if ensure_finite: 

227 # SPM tends to put NaNs in the data outside the brain 

228 arr[np.logical_not(np.isfinite(arr))] = 0 

229 if isinstance(fwhm, str) and (fwhm == 'fast'): 

230 arr = _fast_smooth_array(arr) 

231 elif fwhm is not None: 

232 fwhm = np.asarray([fwhm]).ravel() 

233 fwhm = np.asarray([0. if elem is None else elem for elem in fwhm]) 

234 affine = affine[:3, :3] # Keep only the scale part. 

235 fwhm_over_sigma_ratio = np.sqrt(8 * np.log(2)) # FWHM to sigma. 

236 vox_size = np.sqrt(np.sum(affine ** 2, axis=0)) 

237 sigma = fwhm / (fwhm_over_sigma_ratio * vox_size) 

238 for n, s in enumerate(sigma): 

239 if s > 0.0: 

240 ndimage.gaussian_filter1d(arr, s, output=arr, axis=n) 

241 return arr

244def smooth_img(imgs, fwhm): 

245 """Smooth images by applying a Gaussian filter. 

246 

247 Apply a Gaussian filter along the three first dimensions of `arr`. 

248 In all cases, non-finite values in input image are replaced by zeros. 

249 

250 Parameters 

251 ---------- 

252 imgs : Niimg-like object or iterable of Niimg-like objects 

253 Image(s) to smooth (see 

254 http://nilearn.github.io/manipulating_images/input_output.html 

255 for a detailed description of the valid input types). 

256 

257 fwhm : scalar, :class:`numpy.ndarray`, 'fast' or None 

258 Smoothing strength, as a Full-Width at Half Maximum (FWHM), in 

259 millimeters. 

260 If a scalar is given, width is identical on all three 

261 directions. A :class:`numpy.ndarray` must have 3 elements, giving the FWHM 

262 along each axis. 

263 If `fwhm='fast'`, a fast smoothing will be performed with 

264 a filter [0.2, 1, 0.2] in each direction and a normalisation 

265 to preserve the scale. 

266 If `fwhm` is None, no filtering is performed (useful when just removal 

267 of non-finite values is needed). 

268 

269 In corner case situations, `fwhm` is simply kept to None when `fwhm` is 

270 specified as `fwhm=0`. 

271 

272 Returns 

273 ------- 

274 :class:`nibabel.nifti1.Nifti1Image` or list of 

275 Filtered input image. If `imgs` is an iterable, then `filtered_img` is a 

276 list. 

277 

278 """ 

279 

280 # Use hasattr() instead of isinstance to workaround a Python 2.6/2.7 bug 

281 # See http://bugs.python.org/issue7624 

282 if hasattr(imgs, "__iter__") \ 

283 and not isinstance(imgs, str): 

284 single_img = False 

285 else: 

286 single_img = True 

287 imgs = [imgs] 

288 

289 ret = [] 

290 for img in imgs: 

291 img = check_niimg(img) 

292 affine = img.affine 

293 filtered = _smooth_array(get_data(img), affine, fwhm=fwhm, 

294 ensure_finite=True, copy=True) 

295 ret.append(new_img_like(img, filtered, affine, copy_header=True)) 

296 

297 if single_img: 

298 return ret[0] 

299 else: 

300 return ret

303def _crop_img_to(img, slices, copy=True): 

304 """Crops an image to a smaller size. 

305 

306 Crop `img` to size indicated by slices and adjust affine accordingly. 

307 

308 Parameters 

309 ---------- 

310 img : Niimg-like object 

311 Image to be cropped. If slices has less entries than `img` has dimensions, 

312 the slices will be applied to the first `len(slices)` dimensions (See 

313 http://nilearn.github.io/manipulating_images/input_output.html). 

314 

315 slices : list of slices 

316 Defines the range of the crop. 

317 E.g. [slice(20, 200), slice(40, 150), slice(0, 100)] defines a cube. 

318 

319 copy : :obj:`bool`, optional 

320 Specifies whether cropped data is to be copied or not. Default=True. 

321 

322 Returns 

323 ------- 

324 Niimg-like object 

325 Cropped version of the input image. 

326 

327 offset : :obj:`list`, optional 

328 List of tuples representing the number of voxels removed (before, after) 

329 the cropped volumes, i.e.: 

330 *[(x1_pre, x1_post), (x2_pre, x2_post), ..., (xN_pre, xN_post)]* 

331 

332 """ 

333 img = check_niimg(img) 

334 

335 data = get_data(img) 

336 affine = img.affine 

337 

338 cropped_data = data[tuple(slices)] 

339 if copy: 

340 cropped_data = cropped_data.copy() 

341 

342 linear_part = affine[:3, :3] 

343 old_origin = affine[:3, 3] 

344 new_origin_voxel = np.array([s.start for s in slices]) 

345 new_origin = old_origin + linear_part.dot(new_origin_voxel) 

346 

347 new_affine = np.eye(4) 

348 new_affine[:3, :3] = linear_part 

349 new_affine[:3, 3] = new_origin 

350 

351 return new_img_like(img, cropped_data, new_affine)

354def crop_img(img, rtol=1e-8, copy=True, pad=True, return_offset=False): 

355 """Crops an image as much as possible. 

356 

357 Will crop `img`, removing as many zero entries as possible without 

358 touching non-zero entries. Will leave one voxel of zero padding 

359 around the obtained non-zero area in order to avoid sampling issues 

360 later on. 

361 

362 Parameters 

363 ---------- 

364 img : Niimg-like object 

365 Image to be cropped (see 

366 http://nilearn.github.io/manipulating_images/input_output.html 

367 for a detailed description of the valid input types). 

368 

369 rtol : :obj:`float`, optional 

370 relative tolerance (with respect to maximal absolute value of the 

371 image), under which values are considered negligeable and thus 

372 croppable. Default=1e-8. 

373 

374 copy : :obj:`bool`, optional 

375 Specifies whether cropped data is copied or not. Default=True. 

376 

377 pad : :obj:`bool`, optional 

378 Toggles adding 1-voxel of 0s around the border. Default=True. 

379 

380 return_offset : :obj:`bool`, optional 

381 Specifies whether to return a tuple of the removed padding. 

382 Default=False. 

383 

384 Returns 

385 ------- 

386 Niimg-like object or :obj:`tuple` 

387 Cropped version of the input image and, if `return_offset=True`, a tuple 

388 of tuples representing the number of voxels removed (before, after) the 

389 cropped volumes, i.e.: 

390 *[(x1_pre, x1_post), (x2_pre, x2_post), ..., (xN_pre, xN_post)]* 

391 

392 """ 

393 img = check_niimg(img) 

394 data = get_data(img) 

395 infinity_norm = max(-data.min(), data.max()) 

396 passes_threshold = np.logical_or(data < -rtol * infinity_norm, 

397 data > rtol * infinity_norm) 

398 

399 if data.ndim == 4: 

400 passes_threshold = np.any(passes_threshold, axis=-1) 

401 coords = np.array(np.where(passes_threshold)) 

402 

403 # Sets full range if no data are found along the axis 

404 if coords.shape[1] == 0: 

405 start, end = [0, 0, 0], list(data.shape) 

406 else: 

407 start = coords.min(axis=1) 

408 end = coords.max(axis=1) + 1 

409 

410 # pad with one voxel to avoid resampling problems 

411 if pad: 

412 start = np.maximum(start - 1, 0) 

413 end = np.minimum(end + 1, data.shape[:3]) 

414 

415 slices = [slice(s, e) for s, e in zip(start, end)][:3] 

416 cropped_im = _crop_img_to(img, slices, copy=copy) 

417 return cropped_im if not return_offset else (cropped_im, tuple(slices))

420def _pad_array(array, pad_sizes): 

421 """Pad an array with zeros. 

422 

423 Pads an array with zeros as specified in `pad_sizes`. 

424 

425 Parameters 

426 ---------- 

427 array : :class:`numpy.ndarray` 

428 Array to pad. 

429 

430 pad_sizes : :obj:`list` 

431 Padding quantity specified as 

432 *[x1minpad, x1maxpad, x2minpad,x2maxpad, x3minpad, ...]*. 

433 

434 Returns 

435 ------- 

436 :class:`numpy.ndarray` 

437 Padded array. 

438 

439 Raises 

440 ------ 

441 ValueError 

442 Inconsistent min/max padding quantities. 

443 

444 """ 

445 if len(pad_sizes) % 2 != 0: 

446 raise ValueError("Please specify as many max paddings as min" 

447 " paddings. You have specified %d arguments" % 

448 len(pad_sizes)) 

449 

450 all_paddings = np.zeros([array.ndim, 2], dtype=np.int64) 

451 all_paddings[:len(pad_sizes) // 2] = np.array(pad_sizes).reshape(-1, 2) 

452 

453 lower_paddings, upper_paddings = all_paddings.T 

454 new_shape = np.array(array.shape) + upper_paddings + lower_paddings 

455 

456 padded = np.zeros(new_shape, dtype=array.dtype) 

457 source_slices = [slice(max(-lp, 0), min(s + up, s)) 

458 for lp, up, s in zip(lower_paddings, 

459 upper_paddings, 

460 array.shape)] 

461 target_slices = [slice(max(lp, 0), min(s - up, s)) 

462 for lp, up, s in zip(lower_paddings, 

463 upper_paddings, 

464 new_shape)] 

465 

466 padded[tuple(target_slices)] = array[source_slices].copy() 

467 return padded

470def _compute_mean(imgs, target_affine=None, 

471 target_shape=None, smooth=False): 

472 from . import resampling 

473 input_repr = _repr_niimgs(imgs, shorten=True) 

474 

475 imgs = check_niimg(imgs) 

476 mean_data = _safe_get_data(imgs) 

477 affine = imgs.affine 

478 # Free memory ASAP 

479 imgs = None 

480 if mean_data.ndim not in (3, 4): 

481 raise ValueError('Computation expects 3D or 4D ' 

482 'images, but %i dimensions were given (%s)' 

483 % (mean_data.ndim, input_repr)) 

484 if mean_data.ndim == 4: 

485 mean_data = mean_data.mean(axis=-1) 

486 else: 

487 mean_data = mean_data.copy() 

488 mean_data = resampling.resample_img( 

489 nibabel.Nifti1Image(mean_data, affine), 

490 target_affine=target_affine, target_shape=target_shape, 

491 copy=False) 

492 affine = mean_data.affine 

493 mean_data = get_data(mean_data) 

494 

495 if smooth: 

496 nan_mask = np.isnan(mean_data) 

497 mean_data = _smooth_array(mean_data, affine=np.eye(4), fwhm=smooth, 

498 ensure_finite=True, copy=False) 

499 mean_data[nan_mask] = np.nan 

500 

501 return mean_data, affine

504def mean_img(imgs, target_affine=None, target_shape=None, 

505 verbose=0, n_jobs=1): 

506 """Compute the mean of the images over time or the 4th dimension. 

507 

508 Note that if list of 4D images are given, the mean of each 4D image is 

509 computed separately, and the resulting mean is computed after. 

510 

511 Parameters 

512 ---------- 

513 imgs : Niimg-like object or iterable of Niimg-like objects 

514 Images to be averaged over time (see 

515 http://nilearn.github.io/manipulating_images/input_output.html 

516 for a detailed description of the valid input types). 

517 

518 target_affine : :class:`numpy.ndarray`, optional 

519 If specified, the image is resampled corresponding to this new affine. 

520 target_affine can be a 3x3 or a 4x4 matrix. 

521 

522 target_shape : :obj:`tuple` or :obj:`list`, optional 

523 If specified, the image will be resized to match this new shape. 

524 len(target_shape) must be equal to 3. 

525 A target_affine has to be specified jointly with target_shape. 

526 

527 verbose : :obj:`int`, optional 

528 Controls the amount of verbosity: higher numbers give more messages 

529 (0 means no messages). Default=0. 

530 

531 n_jobs : :obj:`int`, optional 

532 The number of CPUs to use to do the computation (-1 means 

533 'all CPUs'). Default=1. 

534 

535 Returns 

536 ------- 

537 :class:`~nibabel.nifti1.Nifti1Image` 

538 Mean image. 

539 

540 See Also 

541 -------- 

542 nilearn.image.math_img : For more general operations on images. 

543 

544 """ 

545 is_str = isinstance(imgs, str) 

546 is_iterable = isinstance(imgs, collections.abc.Iterable) 

547 if is_str or not is_iterable: 

548 imgs = [imgs, ] 

549 

550 imgs_iter = iter(imgs) 

551 first_img = check_niimg(next(imgs_iter)) 

552 

553 # Compute the first mean to retrieve the reference 

554 # target_affine and target_shape if_needed 

555 n_imgs = 1 

556 running_mean, first_affine = _compute_mean(first_img, 

557 target_affine=target_affine, 

558 target_shape=target_shape) 

559 

560 if target_affine is None or target_shape is None: 

561 target_affine = first_affine 

562 target_shape = running_mean.shape[:3] 

563 

564 for this_mean in Parallel(n_jobs=n_jobs, verbose=verbose)( 

565 delayed(_compute_mean)(n, target_affine=target_affine, 

566 target_shape=target_shape) 

567 for n in imgs_iter): 

568 n_imgs += 1 

569 # _compute_mean returns (mean_img, affine) 

570 this_mean = this_mean[0] 

571 running_mean += this_mean 

572 

573 running_mean = running_mean / float(n_imgs) 

574 return new_img_like(first_img, running_mean, target_affine)

699def new_img_like(ref_niimg, data, affine=None, copy_header=False): 

700 """Create a new image of the same class as the reference image 

701 

702 Parameters 

703 ---------- 

704 ref_niimg : Niimg-like object 

705 Reference image. The new image will be of the same type. 

706 

707 data : :class:`numpy.ndarray` 

708 Data to be stored in the image. 

709 

710 affine : 4x4 :class:`numpy.ndarray`, optional 

711 Transformation matrix. 

712 

713 copy_header : :obj:`bool`, optional 

714 Indicated if the header of the reference image should be used to 

715 create the new image. Default=False. 

716 

717 Returns 

718 ------- 

719 Niimg-like object 

720 A loaded image with the same type (and header) as the reference image. 

721 

722 """ 

723 # Hand-written loading code to avoid too much memory consumption 

724 orig_ref_niimg = ref_niimg 

725 is_str = isinstance(ref_niimg, str) 

726 has_get_data = hasattr(ref_niimg, 'get_data') 

727 has_get_fdata = hasattr(ref_niimg, 'get_fdata') 

728 has_iter = hasattr(ref_niimg, '__iter__') 

729 has_affine = hasattr(ref_niimg, 'affine') 

730 if has_iter and not any([is_str, has_get_data, has_get_fdata]): 

731 ref_niimg = ref_niimg[0] 

732 is_str = isinstance(ref_niimg, str) 

733 has_get_data = hasattr(ref_niimg, 'get_data') 

734 has_get_fdata = hasattr(ref_niimg, 'get_fdata') 

735 has_iter = hasattr(ref_niimg, '__iter__') 

736 has_affine = hasattr(ref_niimg, 'affine') 

737 if not ((has_get_data or has_get_fdata) and has_affine): 

738 if is_str: 

739 ref_niimg = nibabel.load(ref_niimg) 

740 else: 

741 raise TypeError(('The reference image should be a niimg, %r ' 

742 'was passed') % orig_ref_niimg) 

743 

744 if affine is None: 

745 affine = ref_niimg.affine 

746 if data.dtype == bool: 

747 default_dtype = np.int8 

748 if isinstance(ref_niimg, nibabel.freesurfer.mghformat.MGHImage): 

749 default_dtype = np.uint8 

750 data = as_ndarray(data, dtype=default_dtype) 

751 header = None 

752 if copy_header: 

753 header = copy.deepcopy(ref_niimg.header) 

754 try: 

755 'something' in header 

756 except TypeError: 

757 pass 

758 else: 

759 if 'scl_slope' in header: 

760 header['scl_slope'] = 0. 

761 if 'scl_inter' in header: 

762 header['scl_inter'] = 0. 

763 # 'glmax' is removed for Nifti2Image. Modify only if 'glmax' is 

764 # available in header. See issue #1611 

765 if 'glmax' in header: 

766 header['glmax'] = 0. 

767 if 'cal_max' in header: 

768 header['cal_max'] = np.max(data) if data.size > 0 else 0. 

769 if 'cal_min' in header: 

770 header['cal_min'] = np.min(data) if data.size > 0 else 0. 

771 klass = ref_niimg.__class__ 

772 if klass is nibabel.Nifti1Pair: 

773 # Nifti1Pair is an internal class, without a to_filename, 

774 # we shouldn't return it 

775 klass = nibabel.Nifti1Image 

776 return klass(data, affine, header=header)

779def threshold_img(img, threshold, mask_img=None, copy=True): 

780 """Threshold the given input image, mostly statistical or atlas images. 

781 

782 Thresholding can be done based on direct image intensities or selection 

783 threshold with given percentile. 

784 

785 .. versionadded:: 0.2 

786 

787 Parameters 

788 ---------- 

789 img : a 3D/4D Niimg-like object 

790 Image contains of statistical or atlas maps which should be thresholded. 

791 

792 threshold : :obj:`float` or :obj:`str` 

793 If float, we threshold the image based on image intensities meaning 

794 voxels which have intensities greater than this value will be kept. 

795 The given value should be within the range of minimum and 

796 maximum intensity of the input image. 

797 If string, it should finish with percent sign e.g. "80%" and we threshold 

798 based on the score obtained using this percentile on the image data. The 

799 voxels which have intensities greater than this score will be kept. 

800 The given string should be within the range of "0%" to "100%". 

801 

802 mask_img : Niimg-like object, default None, optional 

803 Mask image applied to mask the input data. 

804 If None, no masking will be applied. 

805 

806 copy : :obj:`bool`, optional 

807 If True, input array is not modified. True by default: the filtering 

808 is not performed in-place. Default=True. 

809 

810 Returns 

811 ------- 

812 :class:`~nibabel.nifti1.Nifti1Image` 

813 Thresholded image of the given input image. 

814 

815 See also 

816 -------- 

817 nilearn.glm.threshold_stats_img 

818 

819 """ 

820 from . import resampling 

821 from .. import masking 

822 

823 img = check_niimg(img) 

824 img_data = _safe_get_data(img, ensure_finite=True, copy_data=copy) 

825 affine = img.affine 

826 

827 if mask_img is not None: 

828 mask_img = check_niimg_3d(mask_img) 

829 if not _check_same_fov(img, mask_img): 

830 mask_img = resampling.resample_img(mask_img, target_affine=affine, 

831 target_shape=img.shape[:3], 

832 interpolation="nearest") 

833 

834 mask_data, _ = masking._load_mask_img(mask_img) 

835 # Set as 0 for the values which are outside of the mask 

836 img_data[mask_data == 0.] = 0. 

837 

838 if threshold is None: 

839 raise ValueError("The input parameter 'threshold' is empty. " 

840 "Please give either a float value or a string as e.g. '90%'.") 

841 else: 

842 cutoff_threshold = check_threshold(threshold, img_data, 

843 percentile_func=scoreatpercentile, 

844 name='threshold') 

845 

846 img_data[np.abs(img_data) < cutoff_threshold] = 0. 

847 threshold_img = new_img_like(img, img_data, affine) 

848 

849 return threshold_img

852def math_img(formula, **imgs): 

853 """Interpret a numpy based string formula using niimg in named parameters. 

854 

855 .. versionadded:: 0.2.3 

856 

857 Parameters 

858 ---------- 

859 formula : :obj:`str` 

860 The mathematical formula to apply to image internal data. It can use 

861 numpy imported as 'np'. 

862 

863 imgs : images (:class:`~nibabel.nifti1.Nifti1Image` or file names) 

864 Keyword arguments corresponding to the variables in the formula as 

865 Nifti images. All input images should have the same geometry (shape, 

866 affine). 

867 

868 Returns 

869 ------- 

870 :class:`~nibabel.nifti1.Nifti1Image` 

871 Result of the formula as a Nifti image. Note that the dimension of the 

872 result image can be smaller than the input image. The affine is the 

873 same as the input image. 

874 

875 See Also 

876 -------- 

877 nilearn.image.mean_img : To simply compute the mean of multiple images 

878 

879 Examples 

880 -------- 

881 Let's load an image using nilearn datasets module:: 

882 

883 >>> from nilearn import datasets 

884 >>> anatomical_image = datasets.load_mni152_template() 

885 

886 Now we can use any numpy function on this image:: 

887 

888 >>> from nilearn.image import math_img 

889 >>> log_img = math_img("np.log(img)", img=anatomical_image) 

890 

891 We can also apply mathematical operations on several images:: 

892 

893 >>> result_img = math_img("img1 + img2", 

894 ... img1=anatomical_image, img2=log_img) 

895 

896 Notes 

897 ----- 

898 This function is the Python equivalent of ImCal in SPM or fslmaths 

899 in FSL. 

900 

901 """ 

902 try: 

903 # Check that input images are valid niimg and have a compatible shape 

904 # and affine. 

905 niimgs = [] 

906 for image in imgs.values(): 

907 niimgs.append(check_niimg(image)) 

908 _check_same_fov(*niimgs, raise_error=True) 

909 except Exception as exc: 

910 exc.args = (("Input images cannot be compared, you provided '{0}'," 

911 .format(imgs.values()),) + exc.args) 

912 raise 

913 

914 # Computing input data as a dictionary of numpy arrays. Keep a reference 

915 # niimg for building the result as a new niimg. 

916 niimg = None 

917 data_dict = {} 

918 for key, img in imgs.items(): 

919 niimg = check_niimg(img) 

920 data_dict[key] = _safe_get_data(niimg) 

921 

922 # Add a reference to numpy in the kwargs of eval so that numpy functions 

923 # can be called from there. 

924 data_dict['np'] = np 

925 try: 

926 result = eval(formula, data_dict) 

927 except Exception as exc: 

928 exc.args = (("Input formula couldn't be processed, you provided '{0}'," 

929 .format(formula),) + exc.args) 

930 raise 

931 

932 return new_img_like(niimg, result, niimg.affine)

935def clean_img(imgs, sessions=None, detrend=True, standardize=True, 

936 confounds=None, low_pass=None, high_pass=None, t_r=None, 

937 ensure_finite=False, mask_img=None): 

938 """Improve SNR on masked fMRI signals. 

939 

940 This function can do several things on the input signals, in 

941 the following order: 

942 

943 - detrend 

944 - low- and high-pass filter 

945 - remove confounds 

946 - standardize 

947 

948 Low-pass filtering improves specificity. 

949 

950 High-pass filtering should be kept small, to keep some sensitivity. 

951 

952 Filtering is only meaningful on evenly-sampled signals. 

953 

954 According to Lindquist et al. (2018), removal of confounds will be done 

955 orthogonally to temporal filters (low- and/or high-pass filters), if both 

956 are specified. 

957 

958 .. versionadded:: 0.2.5 

959 

960 Parameters 

961 ---------- 

962 imgs : Niimg-like object 

963 4D image. The signals in the last dimension are filtered (see 

964 http://nilearn.github.io/manipulating_images/input_output.html 

965 for a detailed description of the valid input types). 

966 

967 sessions : :class:`numpy.ndarray`, optional 

968 Add a session level to the cleaning process. Each session will be 

969 cleaned independently. Must be a 1D array of n_samples elements. 

970 

971 detrend : :obj:`bool`, optional 

972 If detrending should be applied on timeseries (before confound removal). 

973 Default=True. 

974 

975 standardize : :obj:`bool`, optional 

976 If True, returned signals are set to unit variance. Default=True. 

977 

978 confounds : :class:`numpy.ndarray`, :obj:`str` or :obj:`list` of 

979 Confounds timeseries. optional 

980 Shape must be (instant number, confound number), or just (instant number,) 

981 The number of time instants in signals and confounds must be 

982 identical (i.e. signals.shape[0] == confounds.shape[0]). 

983 If a string is provided, it is assumed to be the name of a csv file 

984 containing signals as columns, with an optional one-line header. 

985 If a list is provided, all confounds are removed from the input 

986 signal, as if all were in the same array. 

987 

988 low_pass : :obj:`float`, optional 

989 Low cutoff frequencies, in Hertz. 

990 

991 high_pass : :obj:`float`, optional 

992 High cutoff frequencies, in Hertz. 

993 

994 t_r : :obj:`float`, optional 

995 Repetition time, in second (sampling period). Set to None if not 

996 specified. Mandatory if used together with `low_pass` or `high_pass`. 

997 

998 ensure_finite : :obj:`bool`, optional 

999 If True, the non-finite values (NaNs and infs) found in the images 

1000 will be replaced by zeros. Default=False. 

1001 

1002 mask_img : Niimg-like object, optional 

1003 If provided, signal is only cleaned from voxels inside the mask. If 

1004 mask is provided, it should have same shape and affine as imgs. 

1005 If not provided, all voxels are used. 

1006 See http://nilearn.github.io/manipulating_images/input_output.html. 

1007 

1008 Returns 

1009 ------- 

1010 Niimg-like object 

1011 Input images, cleaned. Same shape as `imgs`. 

1012 

1013 Notes 

1014 ----- 

1015 Confounds removal is based on a projection on the orthogonal 

1016 of the signal space [:footcite:`friston1994statistical`]. 

1017 

1018 Orthogonalization between temporal filters and confound removal is based on 

1019 suggestions in [:footcite:`Lindquist407676`]. 

1020 

1021 References 

1022 ---------- 

1023 .. footbibliography:: 

1024 

1025 See Also 

1026 -------- 

1027 nilearn.signal.clean 

1028 

1029 """ 

1030 # Avoid circular import 

1031 from .image import new_img_like 

1032 from .. import masking 

1033 

1034 imgs_ = check_niimg_4d(imgs) 

1035 

1036 # Check if t_r is set, otherwise propose t_r from imgs header 

1037 if low_pass is not None or high_pass is not None: 

1038 if t_r is None: 

1039 

1040 # We raise an error, instead of using the header's t_r as this 

1041 # value is considered to be non-reliable 

1042 raise ValueError( 

1043 "Repetition time (t_r) must be specified for filtering. You " 

1044 "specified None. imgs header suggest it to be {0}".format( 

1045 imgs.header.get_zooms()[3])) 

1046 

1047 # Prepare signal for cleaning 

1048 if mask_img is not None: 

1049 signals = masking.apply_mask(imgs_, mask_img) 

1050 else: 

1051 signals = get_data(imgs_).reshape(-1, imgs_.shape[-1]).T 

1052 

1053 # Clean signal 

1054 data = signal.clean( 

1055 signals, sessions=sessions, detrend=detrend, standardize=standardize, 

1056 confounds=confounds, low_pass=low_pass, high_pass=high_pass, t_r=t_r, 

1057 ensure_finite=ensure_finite) 

1058 

1059 # Put results back into Niimg-like object 

1060 if mask_img is not None: 

1061 imgs_ = masking.unmask(data, mask_img) 

1062 else: 

1063 imgs_ = new_img_like( 

1064 imgs_, data.T.reshape(imgs_.shape), copy_header=True) 

1065 

1066 return imgs_
