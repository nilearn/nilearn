26def md5_hash(string):

28 m.update(string.encode('utf-8'))

39def _md5_sum_file(path): 

40 """ Calculates the MD5 sum of a file. 

41 """ 

42 with open(path, 'rb') as f: 

43 m = hashlib.md5() 

44 while True: 

45 data = f.read(8192) 

46 if not data: 

47 break 

48 m.update(data) 

49 return m.hexdigest()

52def _read_md5_sum_file(path): 

53 """ Reads a MD5 checksum file and returns hashes as a dictionary. 

54 """ 

55 with open(path, "r") as f: 

56 hashes = {} 

57 while True: 

58 line = f.readline() 

59 if not line: 

60 break 

61 h, name = line.rstrip().split(' ', 1) 

62 hashes[name] = h 

63 return hashes

77def _chunk_report_(bytes_so_far, total_size, initial_size, t0): 

78 """Show downloading percentage. 

79 

80 Parameters 

81 ---------- 

82 bytes_so_far : int 

83 Number of downloaded bytes. 

84 

85 total_size : int 

86 Total size of the file (may be 0/None, depending on download method). 

87 

88 t0 : int 

89 The time in seconds (as returned by time.time()) at which the 

90 download was resumed / started. 

91 

92 initial_size : int 

93 If resuming, indicate the initial size of the file. 

94 If not resuming, set to zero. 

95 

96 """ 

97 if not total_size: 

98 sys.stderr.write("\rDownloaded %d of ? bytes." % (bytes_so_far)) 

99 

100 else: 

101 # Estimate remaining download time 

102 total_percent = float(bytes_so_far) / total_size 

103 

104 current_download_size = bytes_so_far - initial_size 

105 bytes_remaining = total_size - bytes_so_far 

106 dt = time.time() - t0 

107 download_rate = current_download_size / max(1e-8, float(dt)) 

108 # Minimum rate of 0.01 bytes/s, to avoid dividing by zero. 

109 time_remaining = bytes_remaining / max(0.01, download_rate) 

110 

111 # Trailing whitespace is to erase extra char when message length 

112 # varies 

113 sys.stderr.write( 

114 "\rDownloaded %d of %d bytes (%.1f%%, %s remaining)" 

115 % (bytes_so_far, total_size, total_percent * 100, 

116 _format_time(time_remaining)))

119def _chunk_read_(response, local_file, chunk_size=8192, report_hook=None, 

120 initial_size=0, total_size=None, verbose=1): 

121 """Download a file chunk by chunk and show advancement 

122 

123 Parameters 

124 ---------- 

125 response : urllib.response.addinfourl 

126 Response to the download request in order to get file size. 

127 

128 local_file : file 

129 Hard disk file where data should be written. 

130 

131 chunk_size : int, optional 

132 Size of downloaded chunks. Default=8192. 

133 

134 report_hook : bool, optional 

135 Whether or not to show downloading advancement. Default: None 

136 

137 initial_size : int, optional 

138 If resuming, indicate the initial size of the file. 

139 Default=0. 

140 

141 total_size : int, optional 

142 Expected final size of download (None means it is unknown). 

143 

144 verbose : int, optional 

145 Verbosity level (0 means no message). Default=1. 

146 

147 Returns 

148 ------- 

149 data : string 

150 The downloaded file. 

151 

152 """ 

153 try: 

154 if total_size is None: 

155 total_size = response.headers.get('Content-Length').strip() 

156 total_size = int(total_size) + initial_size 

157 except Exception as e: 

158 if verbose > 2: 

159 print("Warning: total size could not be determined.") 

160 if verbose > 3: 

161 print("Full stack trace: %s" % e) 

162 total_size = None 

163 bytes_so_far = initial_size 

164 

165 t0 = time_last_display = time.time() 

166 for chunk in response.iter_content(chunk_size): 

167 bytes_so_far += len(chunk) 

168 time_last_read = time.time() 

169 if (report_hook and 

170 # Refresh report every second or when download is 

171 # finished. 

172 (time_last_read > time_last_display + 1. or not chunk)): 

173 _chunk_report_(bytes_so_far, 

174 total_size, initial_size, t0) 

175 time_last_display = time_last_read 

176 if chunk: 

177 local_file.write(chunk) 

178 else: 

179 break 

180 

181 return

316def _uncompress_file(file_, delete_archive=True, verbose=1): 

317 """Uncompress files contained in a data_set. 

318 

319 Parameters 

320 ---------- 

321 file_ : string 

322 Path of file to be uncompressed. 

323 

324 delete_archive : bool, optional 

325 Wheteher or not to delete archive once it is uncompressed. 

326 Default=True. 

327 

328 verbose : int, optional 

329 Verbosity level (0 means no message). Default=1. 

330 

331 Notes 

332 ----- 

333 This handles zip, tar, gzip and bzip files only. 

334 

335 """ 

336 if verbose > 0: 

337 sys.stderr.write('Extracting data from %s...' % file_) 

338 data_dir = os.path.dirname(file_) 

339 # We first try to see if it is a zip file 

340 try: 

341 filename, ext = os.path.splitext(file_) 

342 with open(file_, "rb") as fd: 

343 header = fd.read(4) 

344 processed = False 

345 if zipfile.is_zipfile(file_): 

346 z = zipfile.ZipFile(file_) 

347 z.extractall(path=data_dir) 

348 z.close() 

349 if delete_archive: 

350 os.remove(file_) 

351 file_ = filename 

352 processed = True 

353 elif ext == '.gz' or header.startswith(b'\x1f\x8b'): 

354 import gzip 

355 if ext == '.tgz': 

356 filename = filename + '.tar' 

357 elif ext == '': 

358 # For gzip file, we rely on the assumption that there is an extenstion 

359 shutil.move(file_, file_ + '.gz') 

360 file_ = file_ + '.gz' 

361 with gzip.open(file_) as gz: 

362 with open(filename, 'wb') as out: 

363 shutil.copyfileobj(gz, out, 8192) 

364 # If file is .tar.gz, this will be handle in the next case 

365 if delete_archive: 

366 os.remove(file_) 

367 file_ = filename 

368 processed = True 

369 if os.path.isfile(file_) and tarfile.is_tarfile(file_): 

370 with contextlib.closing(tarfile.open(file_, "r")) as tar: 

371 tar.extractall(path=data_dir) 

372 if delete_archive: 

373 os.remove(file_) 

374 processed = True 

375 if not processed: 

376 raise IOError( 

377 "[Uncompress] unknown archive file format: %s" % file_) 

378 

379 if verbose > 0: 

380 sys.stderr.write('.. done.\n') 

381 except Exception as e: 

382 if verbose > 0: 

383 print('Error uncompressing file: %s' % e) 

384 raise

387def _filter_column(array, col, criteria): 

388 """Return index array matching criteria 

389 

390 Parameters 

391 ---------- 

392 array : numpy array with columns 

393 Array in which data will be filtered. 

394 

395 col : string 

396 Name of the column. 

397 

398 criteria : integer (or float), pair of integers, string or list of these 

399 if integer, select elements in column matching integer 

400 if a tuple, select elements between the limits given by the tuple 

401 if a string, select elements that match the string 

402 

403 """ 

404 # Raise an error if the column does not exist. This is the only way to 

405 # test it across all possible types (pandas, recarray...) 

406 try: 

407 array[col] 

408 except: 

409 raise KeyError('Filtering criterion %s does not exist' % col) 

410 

411 if (not isinstance(criteria, str) and 

412 not isinstance(criteria, bytes) and 

413 not isinstance(criteria, tuple) and 

414 isinstance(criteria, collections.abc.Iterable)): 

415 filter = np.zeros(array.shape[0], dtype=bool) 

416 for criterion in criteria: 

417 filter = np.logical_or(filter, 

418 _filter_column(array, col, criterion)) 

419 return filter 

420 

421 if isinstance(criteria, tuple): 

422 if len(criteria) != 2: 

423 raise ValueError("An interval must have 2 values") 

424 if criteria[0] is None: 

425 return array[col] <= criteria[1] 

426 if criteria[1] is None: 

427 return array[col] >= criteria[0] 

428 filter = array[col] <= criteria[1] 

429 return np.logical_and(filter, array[col] >= criteria[0]) 

430 

431 # Handle strings with different encodings 

432 if isinstance(criteria, (str, bytes)): 

433 criteria = np.array(criteria).astype(array[col].dtype) 

434 

435 return array[col] == criteria

438def _filter_columns(array, filters, combination='and'): 

439 """Return indices of recarray entries that match criteria. 

440 

441 Parameters 

442 ---------- 

443 array : numpy array with columns 

444 Array in which data will be filtered. 

445 

446 filters : list of criteria 

447 See _filter_column. 

448 

449 combination : string {'and', 'or'}, optional 

450 String describing the combination operator. Possible values are "and" 

451 and "or". Default='and'. 

452 

453 """ 

454 if combination == 'and': 

455 fcomb = np.logical_and 

456 mask = np.ones(array.shape[0], dtype=bool) 

457 elif combination == 'or': 

458 fcomb = np.logical_or 

459 mask = np.zeros(array.shape[0], dtype=bool) 

460 else: 

461 raise ValueError('Combination mode not known: %s' % combination) 

462 

463 for column in filters: 

464 mask = fcomb(mask, _filter_column(array, column, filters[column])) 

465 return mask

490def _fetch_file(url, data_dir, resume=True, overwrite=False, 

491 md5sum=None, username=None, password=None, 

492 verbose=1, session=None): 

493 """Load requested file, downloading it if needed or requested. 

494 

495 Parameters 

496 ---------- 

497 url : string 

498 Contains the url of the file to be downloaded. 

499 

500 data_dir : string 

501 Path of the data directory. Used for data storage in the specified 

502 location. 

503 

504 resume : bool, optional 

505 If true, try to resume partially downloaded files. 

506 Default=True. 

507 

508 overwrite : bool, optional 

509 If true and file already exists, delete it. Default=False. 

510 

511 md5sum : string, optional 

512 MD5 sum of the file. Checked if download of the file is required. 

513 

514 username : string, optional 

515 Username used for basic HTTP authentication. 

516 

517 password : string, optional 

518 Password used for basic HTTP authentication. 

519 

520 verbose : int, optional 

521 Verbosity level (0 means no message). Default=1. 

522 

523 session : requests.Session, optional 

524 Session to use to send requests. 

525 

526 Returns 

527 ------- 

528 files : string 

529 Absolute path of downloaded file. 

530 

531 Notes 

532 ----- 

533 If, for any reason, the download procedure fails, all downloaded files are 

534 removed. 

535 

536 """ 

537 if session is None: 

538 with requests.Session() as session: 

539 session.mount("ftp:", _NaiveFTPAdapter()) 

540 return _fetch_file( 

541 url, data_dir, resume=resume, overwrite=overwrite, 

542 md5sum=md5sum, username=username, password=password, 

543 verbose=verbose, session=session) 

544 # Determine data path 

545 if not os.path.exists(data_dir): 

546 os.makedirs(data_dir) 

547 

548 # Determine filename using URL 

549 parse = urllib.parse.urlparse(url) 

550 file_name = os.path.basename(parse.path) 

551 if file_name == '': 

552 file_name = md5_hash(parse.path) 

553 

554 temp_file_name = file_name + ".part" 

555 full_name = os.path.join(data_dir, file_name) 

556 temp_full_name = os.path.join(data_dir, temp_file_name) 

557 if os.path.exists(full_name): 

558 if overwrite: 

559 os.remove(full_name) 

560 else: 

561 return full_name 

562 if os.path.exists(temp_full_name): 

563 if overwrite: 

564 os.remove(temp_full_name) 

565 t0 = time.time() 

566 local_file = None 

567 initial_size = 0 

568 

569 try: 

570 # Download data 

571 headers = {} 

572 auth = None 

573 if username is not None and password is not None: 

574 if not url.startswith('https'): 

575 raise ValueError( 

576 'Authentication was requested on a non secured URL (%s).' 

577 'Request has been blocked for security reasons.' % url) 

578 auth = (username, password) 

579 if verbose > 0: 

580 displayed_url = url.split('?')[0] if verbose == 1 else url 

581 print('Downloading data from %s ...' % displayed_url) 

582 if resume and os.path.exists(temp_full_name): 

583 # Download has been interrupted, we try to resume it. 

584 local_file_size = os.path.getsize(temp_full_name) 

585 # If the file exists, then only download the remainder 

586 headers["Range"] = "bytes={}-".format(local_file_size) 

587 try: 

588 req = requests.Request( 

589 method="GET", url=url, headers=headers, auth=auth) 

590 prepped = session.prepare_request(req) 

591 with session.send(prepped, stream=True, 

592 timeout=_REQUESTS_TIMEOUT) as resp: 

593 resp.raise_for_status() 

594 content_range = resp.headers.get('Content-Range') 

595 if (content_range is None or not content_range.startswith( 

596 'bytes {}-'.format(local_file_size))): 

597 raise IOError('Server does not support resuming') 

598 initial_size = local_file_size 

599 with open(local_file, "ab") as fh: 

600 _chunk_read_( 

601 resp, fh, report_hook=(verbose > 0), 

602 initial_size=initial_size, verbose=verbose) 

603 except Exception: 

604 if verbose > 0: 

605 print('Resuming failed, try to download the whole file.') 

606 return _fetch_file( 

607 url, data_dir, resume=False, overwrite=overwrite, 

608 md5sum=md5sum, username=username, password=password, 

609 verbose=verbose, session=session) 

610 else: 

611 req = requests.Request( 

612 method="GET", url=url, headers=headers, auth=auth) 

613 prepped = session.prepare_request(req) 

614 with session.send( 

615 prepped, stream=True, timeout=_REQUESTS_TIMEOUT) as resp: 

616 resp.raise_for_status() 

617 with open(temp_full_name, "wb") as fh: 

618 _chunk_read_(resp, fh, report_hook=(verbose > 0), 

619 initial_size=initial_size, verbose=verbose) 

620 shutil.move(temp_full_name, full_name) 

621 dt = time.time() - t0 

622 if verbose > 0: 

623 # Complete the reporting hook 

624 sys.stderr.write(' ...done. ({0:.0f} seconds, {1:.0f} min)\n' 

625 .format(dt, dt // 60)) 

626 except (requests.RequestException): 

627 sys.stderr.write("Error while fetching file %s; dataset " 

628 "fetching aborted." % (file_name)) 

629 raise 

630 if md5sum is not None: 

631 if (_md5_sum_file(full_name) != md5sum): 

632 raise ValueError("File %s checksum verification has failed." 

633 " Dataset fetching aborted." % local_file) 

634 return full_name

655def movetree(src, dst): 

656 """Move an entire tree to another directory. Any existing file is 

657 overwritten""" 

658 names = os.listdir(src) 

659 

660 # Create destination dir if it does not exist 

661 if not os.path.exists(dst): 

662 os.makedirs(dst) 

663 errors = [] 

664 

665 for name in names: 

666 srcname = os.path.join(src, name) 

667 dstname = os.path.join(dst, name) 

668 try: 

669 if os.path.isdir(srcname) and os.path.isdir(dstname): 

670 movetree(srcname, dstname) 

671 os.rmdir(srcname) 

672 else: 

673 shutil.move(srcname, dstname) 

674 except (IOError, os.error) as why: 

675 errors.append((srcname, dstname, str(why))) 

676 # catch the Error from the recursive movetree so that we can 

677 # continue with other files 

678 except Exception as err: 

679 errors.extend(err.args[0]) 

680 if errors: 

681 raise Exception(errors)

684def _fetch_files(data_dir, files, resume=True, verbose=1, session=None): 

685 """Load requested dataset, downloading it if needed or requested. 

686 

687 This function retrieves files from the hard drive or download them from 

688 the given urls. Note to developpers: All the files will be first 

689 downloaded in a sandbox and, if everything goes well, they will be moved 

690 into the folder of the dataset. This prevents corrupting previously 

691 downloaded data. In case of a big dataset, do not hesitate to make several 

692 calls if needed. 

693 

694 Parameters 

695 ---------- 

696 data_dir : string 

697 Path of the data directory. Used for data storage in a specified 

698 location. 

699 

700 files : list of (string, string, dict) 

701 List of files and their corresponding url with dictionary that contains 

702 options regarding the files. Eg. (file_path, url, opt). If a file_path 

703 is not found in data_dir, as in data_dir/file_path the download will 

704 be immediately cancelled and any downloaded files will be deleted. 

705 Options supported are: 

706 * 'move' if renaming the file or moving it to a subfolder is needed 

707 * 'uncompress' to indicate that the file is an archive 

708 * 'md5sum' to check the md5 sum of the file 

709 * 'overwrite' if the file should be re-downloaded even if it exists 

710 

711 resume : bool, optional 

712 If true, try resuming download if possible. Default=True. 

713 

714 verbose : int, optional 

715 Verbosity level (0 means no message). Default=1. 

716 

717 session : `requests.Session`, optional 

718 Session to use to send requests. 

719 

720 Returns 

721 ------- 

722 files : list of string 

723 Absolute paths of downloaded files on disk. 

724 

725 """ 

726 if session is None: 

727 with requests.Session() as session: 

728 session.mount("ftp:", _NaiveFTPAdapter()) 

729 return _fetch_files( 

730 data_dir, files, resume=resume, 

731 verbose=verbose, session=session) 

732 # There are two working directories here: 

733 # - data_dir is the destination directory of the dataset 

734 # - temp_dir is a temporary directory dedicated to this fetching call. All 

735 # files that must be downloaded will be in this directory. If a corrupted 

736 # file is found, or a file is missing, this working directory will be 

737 # deleted. 

738 files = list(files) 

739 files_pickle = pickle.dumps([(file_, url) for file_, url, _ in files]) 

740 files_md5 = hashlib.md5(files_pickle).hexdigest() 

741 temp_dir = os.path.join(data_dir, files_md5) 

742 

743 # Create destination dir 

744 if not os.path.exists(data_dir): 

745 os.makedirs(data_dir) 

746 

747 # Abortion flag, in case of error 

748 abort = None 

749 

750 files_ = [] 

751 for file_, url, opts in files: 

752 # 3 possibilities: 

753 # - the file exists in data_dir, nothing to do. 

754 # - the file does not exists: we download it in temp_dir 

755 # - the file exists in temp_dir: this can happen if an archive has been 

756 # downloaded. There is nothing to do 

757 

758 # Target file in the data_dir 

759 target_file = os.path.join(data_dir, file_) 

760 # Target file in temp dir 

761 temp_target_file = os.path.join(temp_dir, file_) 

762 # Whether to keep existing files 

763 overwrite = opts.get('overwrite', False) 

764 if (abort is None and (overwrite or (not os.path.exists(target_file) and not 

765 os.path.exists(temp_target_file)))): 

766 

767 # We may be in a global read-only repository. If so, we cannot 

768 # download files. 

769 if not os.access(data_dir, os.W_OK): 

770 raise ValueError('Dataset files are missing but dataset' 

771 ' repository is read-only. Contact your data' 

772 ' administrator to solve the problem') 

773 

774 if not os.path.exists(temp_dir): 

775 os.mkdir(temp_dir) 

776 md5sum = opts.get('md5sum', None) 

777 

778 dl_file = _fetch_file(url, temp_dir, resume=resume, 

779 verbose=verbose, md5sum=md5sum, 

780 username=opts.get('username', None), 

781 password=opts.get('password', None), 

782 overwrite=overwrite) 

783 if 'move' in opts: 

784 # XXX: here, move is supposed to be a dir, it can be a name 

785 move = os.path.join(temp_dir, opts['move']) 

786 move_dir = os.path.dirname(move) 

787 if not os.path.exists(move_dir): 

788 os.makedirs(move_dir) 

789 shutil.move(dl_file, move) 

790 dl_file = move 

791 if 'uncompress' in opts: 

792 try: 

793 _uncompress_file(dl_file, verbose=verbose) 

794 except Exception as e: 

795 abort = str(e) 

796 

797 if (abort is None and not os.path.exists(target_file) and not 

798 os.path.exists(temp_target_file)): 

799 warnings.warn('An error occured while fetching %s' % file_) 

800 abort = ("Dataset has been downloaded but requested file was " 

801 "not provided:\nURL: %s\n" 

802 "Target file: %s\nDownloaded: %s" % 

803 (url, target_file, dl_file)) 

804 if abort is not None: 

805 if os.path.exists(temp_dir): 

806 shutil.rmtree(temp_dir) 

807 raise IOError('Fetching aborted: ' + abort) 

808 files_.append(target_file) 

809 # If needed, move files from temps directory to final directory. 

810 if os.path.exists(temp_dir): 

811 # XXX We could only moved the files requested 

812 # XXX Movetree can go wrong 

813 movetree(temp_dir, data_dir) 

814 shutil.rmtree(temp_dir) 

815 return files_

818def _tree(path, pattern=None, dictionary=False): 

819 """Return a directory tree under the form of a dictionaries and list 

820 

821 Parameters: 

822 ----------- 

823 path : string 

824 Path browsed. 

825 

826 pattern : string, optional 

827 Pattern used to filter files (see fnmatch). 

828 

829 dictionary : boolean, optional 

830 If True, the function will return a dict instead of a list. 

831 Default=False. 

832 

833 """ 

834 files = [] 

835 dirs = [] if not dictionary else {} 

836 for file_ in os.listdir(path): 

837 file_path = os.path.join(path, file_) 

838 if os.path.isdir(file_path): 

839 if not dictionary: 

840 dirs.append((file_, _tree(file_path, pattern))) 

841 else: 

842 dirs[file_] = _tree(file_path, pattern) 

843 else: 

844 if pattern is None or fnmatch.fnmatch(file_, pattern): 

845 files.append(file_path) 

846 files = sorted(files) 

847 if not dictionary: 

848 return sorted(dirs) + files 

849 if len(dirs) == 0: 

850 return files 

851 if len(files) > 0: 

852 dirs['.'] = files 

853 return dirs

856def make_fresh_openneuro_dataset_urls_index( 

857 data_dir=None, 

858 dataset_version='ds000030_R1.0.4', 

859 verbose=1, 

860 ): 

861 """ONLY intended for Nilearn developers, not general users. 

862 Creates a fresh, updated OpenNeuro :term:`BIDS` dataset index from AWS, 

863 ready for upload to osf.io . 

864 

865 Crawls the server where OpenNeuro dataset is stored 

866 and makes a JSON file `nistats_fetcher_openneuro_dataset_urls.json' 

867 containing a fresh list of dataset file URLs. 

868 

869 Notes 

870 ----- 

871 Needs Python package `Boto3`. 

872 

873 Do NOT rename this file. 

874 

875 This file can now be uploaded to Quick-Files section 

876 of the Nilearn account on osf.io . 

877 

878 Then this file can be downloaded by 

879 :func:`datasets.fetch_openneuro_dataset_index` 

880 

881 Run this function and upload the new file if the URL index downloaded by 

882 :func:`datasets.fetch_openneuro_dataset_index` becomes outdated. 

883 

884 This approach is faster than crawling the servers anew every time 

885 the OpenNeuro dataset is downloaded, 

886 and circumvents `boto3` as a dependency for everyday use. 

887 

888 Parameters 

889 ---------- 

890 data_dir : string, optional 

891 Path to store the downloaded dataset. 

892 If None downloads to user's Desktop. 

893 

894 dataset_version : string, optional 

895 Dataset version name. Assumes it is of the form [name]_[version]. 

896 Default is `ds000030_R1.0.4`. 

897 

898 verbose : int, optional 

899 Verbosity level (0 means no message). Default=1. 

900 

901 Returns 

902 ------- 

903 urls_path : string 

904 Path to downloaded dataset index. 

905 

906 urls : list of string 

907 Sorted list of dataset directories. 

908 

909 """ 

910 import boto3 

911 from botocore.handlers import disable_signing 

912 if not data_dir: 

913 data_dir = os.path.expanduser('~/Desktop') 

914 data_prefix = '{}/{}/uncompressed'.format( 

915 dataset_version.split('_')[0], dataset_version) 

916 

917 data_dir = _get_dataset_dir(data_prefix, data_dir=data_dir, 

918 verbose=verbose) 

919 

920 # First we download the url list from the uncompressed dataset version 

921 urls_path = os.path.join(data_dir, 

922 'nistats_fetcher_openneuro_dataset_urls.json', 

923 ) 

924 urls = [] 

925 if os.path.exists(urls_path): 

926 with open(urls_path, 'r') as json_file: 

927 urls = json.load(json_file) 

928 existing_index_msg = ("There is an existing url index at `{}`. " 

929 "Aborting download of fresh index." 

930 .format(urls_path) 

931 ) 

932 print(existing_index_msg) 

933 else: 

934 resource = boto3.resource('s3') 

935 resource.meta.client.meta.events.register('choose-signer.s3.*', 

936 disable_signing) 

937 bucket = resource.Bucket('openneuro') 

938 

939 for obj in bucket.objects.filter(Prefix=data_prefix): 

940 # get url of files (keys of directories end with '/') 

941 if obj.key[-1] != '/': 

942 url = '{}/{}/{}'.format(bucket.meta.client.meta.endpoint_url, 

943 bucket.name, 

944 obj.key, 

945 ) 

946 urls.append(url) 

947 urls = sorted(urls) 

948 

949 with open(urls_path, 'w') as json_file: 

950 json.dump(urls, json_file) 

951 print("Saved updated url index to {}.\nUpload it with the same name " 

952 "to the quick-files section of osf.io using the Nilearn account " 

953 "to update the file without breaking the fetcher download link." 

954 .format(urls_path)) 

955 return urls_path, urls
