64def assert_memory_less_than(memory_limit, tolerance, 

65 callable_obj, *args, **kwargs): 

66 """Check memory consumption of a callable stays below a given limit. 

67 

68 Parameters 

69 ---------- 

70 memory_limit : int 

71 The expected memory limit in MiB. 

72 

73 tolerance : float 

74 As memory_profiler results have some variability, this adds some 

75 tolerance around memory_limit. Accepted values are in range [0.0, 1.0]. 

76 

77 callable_obj : callable 

78 The function to be called to check memory consumption. 

79 

80 """ 

81 mem_used = memory_used(callable_obj, *args, **kwargs) 

82 

83 if mem_used > memory_limit * (1 + tolerance): 

84 raise ValueError("Memory consumption measured ({0:.2f} MiB) is " 

85 "greater than required memory limit ({1} MiB) within " 

86 "accepted tolerance ({2:.2f}%)." 

87 "".format(mem_used, memory_limit, tolerance * 100)) 

88 

89 # We are confident in memory_profiler measures above 100MiB. 

90 # We raise an error if the measure is below the limit of 50MiB to avoid 

91 # false positive. 

92 if mem_used < 50: 

93 raise ValueError("Memory profiler measured an untrustable memory " 

94 "consumption ({0:.2f} MiB). The expected memory " 

95 "limit was {1:.2f} MiB. Try to bench with larger " 

96 "objects (at least 100MiB in memory).". 

97 format(mem_used, memory_limit))

114@contextlib.contextmanager 

115def write_tmp_imgs(*imgs, **kwargs): 

116 """Context manager for writing Nifti images. 

117 

118 Write nifti images in a temporary location, and remove them at the end of 

119 the block. 

120 

121 Parameters 

122 ---------- 

123 imgs : Nifti1Image 

124 Several Nifti images. Every format understood by nibabel.save is 

125 accepted. 

126 

127 create_files : bool 

128 If True, imgs are written on disk and filenames are returned. If 

129 False, nothing is written, and imgs is returned as output. This is 

130 useful to test the two cases (filename / Nifti1Image) in the same 

131 loop. 

132 

133 use_wildcards : bool 

134 If True, and create_files is True, imgs are written on disk and a 

135 matching glob is returned. 

136 

137 Returns 

138 ------- 

139 filenames : string or list of strings 

140 Filename(s) where input images have been written. If a single image 

141 has been given as input, a single string is returned. Otherwise, a 

142 list of string is returned. 

143 

144 """ 

145 valid_keys = set(("create_files", "use_wildcards")) 

146 input_keys = set(kwargs.keys()) 

147 invalid_keys = input_keys - valid_keys 

148 if len(invalid_keys) > 0: 

149 raise TypeError("%s: unexpected keyword argument(s): %s" % 

150 (sys._getframe().f_code.co_name, 

151 " ".join(invalid_keys))) 

152 create_files = kwargs.get("create_files", True) 

153 use_wildcards = kwargs.get("use_wildcards", False) 

154 

155 prefix = "nilearn_" 

156 suffix = ".nii" 

157 

158 if create_files: 

159 filenames = [] 

160 try: 

161 with warnings.catch_warnings(): 

162 warnings.simplefilter("ignore", RuntimeWarning) 

163 for img in imgs: 

164 fd, filename = tempfile.mkstemp(prefix=prefix, 

165 suffix=suffix, 

166 dir=None) 

167 os.close(fd) 

168 filenames.append(filename) 

169 img.to_filename(filename) 

170 del img 

171 

172 if use_wildcards: 

173 yield prefix + "*" + suffix 

174 else: 

175 if len(imgs) == 1: 

176 yield filenames[0] 

177 else: 

178 yield filenames 

179 finally: 

180 failures = [] 

181 # Ensure all created files are removed 

182 for filename in filenames: 

183 try: 

184 os.remove(filename) 

185 except FileNotFoundError: 

186 # ok, file already removed 

187 pass 

188 except OSError as e: 

189 # problem eg permission, or open file descriptor 

190 failures.append(e) 

191 if failures: 

192 raise OSError( 

193 "The following files could not be removed:\n".format( 

194 "\n".join(str(e) for e in failures) 

195 ) 

196 ) 

197 else: # No-op 

198 if len(imgs) == 1: 

199 yield imgs[0] 

200 else: 

201 yield imgs
