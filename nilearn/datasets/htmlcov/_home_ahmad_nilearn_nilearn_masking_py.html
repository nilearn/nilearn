29def _load_mask_img(mask_img, allow_empty=False): 

30 """Check that a mask is valid, ie with two values including 0 and load it. 

31 

32 Parameters 

33 ---------- 

34 mask_img: Niimg-like object 

35 See http://nilearn.github.io/manipulating_images/input_output.html 

36 The mask to check 

37 

38 allow_empty: boolean, optional 

39 Allow loading an empty mask (full of 0 values) 

40 

41 Returns 

42 ------- 

43 mask: numpy.ndarray 

44 boolean version of the mask 

45 """ 

46 mask_img = _utils.check_niimg_3d(mask_img) 

47 mask = _safe_get_data(mask_img, ensure_finite=True) 

48 values = np.unique(mask) 

49 

50 if len(values) == 1: 

51 # We accept a single value if it is not 0 (full true mask). 

52 if values[0] == 0 and not allow_empty: 

53 raise ValueError( 

54 'The mask is invalid as it is empty: it masks all data.') 

55 elif len(values) == 2: 

56 # If there are 2 different values, one of them must be 0 (background) 

57 if 0 not in values: 

58 raise ValueError('Background of the mask must be represented with' 

59 '0. Given mask contains: %s.' % values) 

60 elif len(values) != 2: 

61 # If there are more than 2 values, the mask is invalid 

62 raise ValueError('Given mask is not made of 2 values: %s' 

63 '. Cannot interpret as true or false' 

64 % values) 

65 

66 mask = _utils.as_ndarray(mask, dtype=bool) 

67 return mask, mask_img.affine

70def _extrapolate_out_mask(data, mask, iterations=1): 

71 """ Extrapolate values outside of the mask. 

72 """ 

73 if iterations > 1: 

74 data, mask = _extrapolate_out_mask(data, mask, 

75 iterations=iterations - 1) 

76 new_mask = ndimage.binary_dilation(mask) 

77 larger_mask = np.zeros(np.array(mask.shape) + 2, dtype=bool) 

78 larger_mask[1:-1, 1:-1, 1:-1] = mask 

79 # Use nans as missing value: ugly 

80 masked_data = np.zeros(larger_mask.shape + data.shape[3:]) 

81 masked_data[1:-1, 1:-1, 1:-1] = data.copy() 

82 masked_data[np.logical_not(larger_mask)] = np.nan 

83 outer_shell = larger_mask.copy() 

84 outer_shell[1:-1, 1:-1, 1:-1] = np.logical_xor(new_mask, mask) 

85 outer_shell_x, outer_shell_y, outer_shell_z = np.where(outer_shell) 

86 extrapolation = list() 

87 for i, j, k in [(1, 0, 0), (-1, 0, 0), 

88 (0, 1, 0), (0, -1, 0), 

89 (0, 0, 1), (0, 0, -1)]: 

90 this_x = outer_shell_x + i 

91 this_y = outer_shell_y + j 

92 this_z = outer_shell_z + k 

93 extrapolation.append(masked_data[this_x, this_y, this_z]) 

94 

95 extrapolation = np.array(extrapolation) 

96 extrapolation = (np.nansum(extrapolation, axis=0) / 

97 np.sum(np.isfinite(extrapolation), axis=0)) 

98 extrapolation[np.logical_not(np.isfinite(extrapolation))] = 0 

99 new_data = np.zeros_like(masked_data) 

100 new_data[outer_shell] = extrapolation 

101 new_data[larger_mask] = masked_data[larger_mask] 

102 return new_data[1:-1, 1:-1, 1:-1], new_mask

109def intersect_masks(mask_imgs, threshold=0.5, connected=True): 

110 """ Compute intersection of several masks 

111 

112 Given a list of input mask images, generate the output image which 

113 is the threshold-level intersection of the inputs 

114 

115 Parameters 

116 ---------- 

117 mask_imgs: list of Niimg-like objects 

118 See http://nilearn.github.io/manipulating_images/input_output.html 

119 3D individual masks with same shape and affine. 

120 

121 threshold: float, optional 

122 Gives the level of the intersection, must be within [0, 1]. 

123 threshold=1 corresponds to keeping the intersection of all 

124 masks, whereas threshold=0 is the union of all masks. 

125 

126 connected: bool, optional 

127 If true, extract the main connected component 

128 

129 Returns 

130 ------- 

131 grp_mask: 3D nibabel.Nifti1Image 

132 intersection of all masks. 

133 """ 

134 if len(mask_imgs) == 0: 

135 raise ValueError('No mask provided for intersection') 

136 grp_mask = None 

137 first_mask, ref_affine = _load_mask_img(mask_imgs[0], allow_empty=True) 

138 ref_shape = first_mask.shape 

139 if threshold > 1: 

140 raise ValueError('The threshold should be smaller than 1') 

141 if threshold < 0: 

142 raise ValueError('The threshold should be greater than 0') 

143 threshold = min(threshold, 1 - 1.e-7) 

144 

145 for this_mask in mask_imgs: 

146 mask, affine = _load_mask_img(this_mask, allow_empty=True) 

147 if np.any(affine != ref_affine): 

148 raise ValueError("All masks should have the same affine") 

149 if np.any(mask.shape != ref_shape): 

150 raise ValueError("All masks should have the same shape") 

151 

152 if grp_mask is None: 

153 # We use int here because there may be a lot of masks to merge 

154 grp_mask = _utils.as_ndarray(mask, dtype=int) 

155 else: 

156 # If this_mask is floating point and grp_mask is integer, numpy 2 

157 # casting rules raise an error for in-place addition. Hence we do 

158 # it long-hand. 

159 # XXX should the masks be coerced to int before addition? 

160 grp_mask += mask 

161 

162 grp_mask = grp_mask > (threshold * len(list(mask_imgs))) 

163 

164 if np.any(grp_mask > 0) and connected: 

165 grp_mask = largest_connected_component(grp_mask) 

166 grp_mask = _utils.as_ndarray(grp_mask, dtype=np.int8) 

167 return new_img_like(_utils.check_niimg_3d(mask_imgs[0]), grp_mask, 

168 ref_affine)

171def _post_process_mask(mask, affine, opening=2, connected=True, 

172 warning_msg=""): 

173 if opening: 

174 opening = int(opening) 

175 mask = ndimage.binary_erosion(mask, iterations=opening) 

176 mask_any = mask.any() 

177 if not mask_any: 

178 warnings.warn("Computed an empty mask. %s" % warning_msg, 

179 MaskWarning, stacklevel=2) 

180 if connected and mask_any: 

181 mask = largest_connected_component(mask) 

182 if opening: 

183 mask = ndimage.binary_dilation(mask, iterations=2 * opening) 

184 mask = ndimage.binary_erosion(mask, iterations=opening) 

185 return mask, affine

188def compute_epi_mask(epi_img, lower_cutoff=0.2, upper_cutoff=0.85, 

189 connected=True, opening=2, exclude_zeros=False, 

190 ensure_finite=True, 

191 target_affine=None, target_shape=None, 

192 memory=None, verbose=0,): 

193 """Compute a brain mask from fMRI data in 3D or 4D ndarrays. 

194 

195 This is based on an heuristic proposed by T.Nichols: 

196 find the least dense point of the histogram, between fractions 

197 lower_cutoff and upper_cutoff of the total image histogram. 

198 

199 In case of failure, it is usually advisable to increase lower_cutoff. 

200 

201 Parameters 

202 ---------- 

203 epi_img: Niimg-like object 

204 See http://nilearn.github.io/manipulating_images/input_output.html 

205 EPI image, used to compute the mask. 3D and 4D images are accepted. 

206 If a 3D image is given, we suggest to use the mean image 

207 

208 lower_cutoff: float, optional 

209 lower fraction of the histogram to be discarded. 

210 

211 upper_cutoff: float, optional 

212 upper fraction of the histogram to be discarded. 

213 

214 connected: bool, optional 

215 if connected is True, only the largest connect component is kept. 

216 

217 opening: bool or int, optional 

218 if opening is True, a morphological opening is performed, to keep 

219 only large structures. This step is useful to remove parts of 

220 the skull that might have been included. 

221 If opening is an integer `n`, it is performed via `n` erosions. 

222 After estimation of the largest connected constituent, 2`n` closing 

223 operations are performed followed by `n` erosions. This corresponds 

224 to 1 opening operation of order `n` followed by a closing operator 

225 of order `n`. 

226 Note that turning off opening (opening=False) will also prevent 

227 any smoothing applied to the image during the mask computation. 

228 

229 ensure_finite: bool 

230 If ensure_finite is True, the non-finite values (NaNs and infs) 

231 found in the images will be replaced by zeros 

232 

233 exclude_zeros: bool, optional 

234 Consider zeros as missing values for the computation of the 

235 threshold. This option is useful if the images have been 

236 resliced with a large padding of zeros. 

237 

238 target_affine: 3x3 or 4x4 matrix, optional 

239 This parameter is passed to image.resample_img. Please see the 

240 related documentation for details. 

241 

242 target_shape: 3-tuple of integers, optional 

243 This parameter is passed to image.resample_img. Please see the 

244 related documentation for details. 

245 

246 memory: instance of joblib.Memory or string 

247 Used to cache the function call: if this is a string, it 

248 specifies the directory where the cache will be stored. 

249 

250 verbose: int, optional 

251 Controls the amount of verbosity: higher numbers give 

252 more messages 

253 

254 Returns 

255 ------- 

256 mask: nibabel.Nifti1Image 

257 The brain mask (3D image) 

258 """ 

259 if verbose > 0: 

260 print("EPI mask computation") 

261 

262 # Delayed import to avoid circular imports 

263 from .image.image import _compute_mean 

264 mean_epi, affine = \ 

265 cache(_compute_mean, memory)(epi_img, target_affine=target_affine, 

266 target_shape=target_shape, 

267 smooth=(1 if opening else False)) 

268 

269 if ensure_finite: 

270 # Get rid of memmapping 

271 mean_epi = _utils.as_ndarray(mean_epi) 

272 # SPM tends to put NaNs in the data outside the brain 

273 mean_epi[np.logical_not(np.isfinite(mean_epi))] = 0 

274 sorted_input = np.sort(np.ravel(mean_epi)) 

275 if exclude_zeros: 

276 sorted_input = sorted_input[sorted_input != 0] 

277 lower_cutoff = int(np.floor(lower_cutoff * len(sorted_input))) 

278 upper_cutoff = min(int(np.floor(upper_cutoff * len(sorted_input))), 

279 len(sorted_input) - 1) 

280 

281 delta = sorted_input[lower_cutoff + 1:upper_cutoff + 1] \ 

282 - sorted_input[lower_cutoff:upper_cutoff] 

283 ia = delta.argmax() 

284 threshold = 0.5 * (sorted_input[ia + lower_cutoff] + 

285 sorted_input[ia + lower_cutoff + 1]) 

286 

287 mask = mean_epi >= threshold 

288 

289 mask, affine = _post_process_mask(mask, affine, opening=opening, 

290 connected=connected, 

291 warning_msg="Are you sure that input " 

292 "data are EPI images not detrended. ") 

293 return new_img_like(epi_img, mask, affine)

311 A list of arrays, each item being a subject or a session.

313 If 3D images is given, we suggest to use the mean image of each 

314 session 

315 

316 threshold: float, optional 

317 the inter-session threshold: the fraction of the 

318 total number of session in for which a voxel must be in the 

319 mask to be kept in the common mask. 

320 threshold=1 corresponds to keeping the intersection of all 

321 masks, whereas threshold=0 is the union of all masks. 

322 

323 lower_cutoff: float, optional 

324 lower fraction of the histogram to be discarded. 

325 

326 upper_cutoff: float, optional 

327 upper fraction of the histogram to be discarded. 

328 

329 connected: bool, optional 

330 if connected is True, only the largest connect component is kept. 

331 

332 exclude_zeros: boolean, optional 

333 Consider zeros as missing values for the computation of the 

334 threshold. This option is useful if the images have been 

335 resliced with a large padding of zeros.

338 This parameter is passed to image.resample_img. Please see the

344

352 Returns

355 The brain mask. 

356 """ 

357 if len(epi_imgs) == 0: 

358 raise TypeError('An empty object - %r - was passed instead of an ' 

359 'image or a list of images' % epi_imgs) 

360 masks = Parallel(n_jobs=n_jobs, verbose=verbose)(

362 lower_cutoff=lower_cutoff,

372 mask = intersect_masks(masks, connected=connected, threshold=threshold)

376def compute_background_mask(data_imgs, border_size=2, 

377 connected=False, opening=False, 

378 target_affine=None, target_shape=None, 

379 memory=None, verbose=0): 

380 """ Compute a brain mask for the images by guessing the value of the 

381 background from the border of the image. 

382 

383 Parameters 

384 ---------- 

385 data_imgs: Niimg-like object 

386 See http://nilearn.github.io/manipulating_images/input_output.html 

387 Images used to compute the mask. 3D and 4D images are accepted. 

388 If a 3D image is given, we suggest to use the mean image 

389 

390 border_size: integer, optional 

391 The size, in voxel of the border used on the side of the image 

392 to determine the value of the background. 

393 

394 connected: bool, optional 

395 if connected is True, only the largest connect component is kept. 

396 

397 opening: bool or int, optional 

398 if opening is True, a morphological opening is performed, to keep 

399 only large structures. This step is useful to remove parts of 

400 the skull that might have been included. 

401 If opening is an integer `n`, it is performed via `n` erosions. 

402 After estimation of the largest connected constituent, 2`n` closing 

403 operations are performed followed by `n` erosions. This corresponds 

404 to 1 opening operation of order `n` followed by a closing operator 

405 of order `n`. 

406 

407 target_affine: 3x3 or 4x4 matrix, optional 

408 This parameter is passed to image.resample_img. Please see the 

409 related documentation for details. 

410 

411 target_shape: 3-tuple of integers, optional 

412 This parameter is passed to image.resample_img. Please see the 

413 related documentation for details. 

414 

415 memory: instance of joblib.Memory or string 

416 Used to cache the function call. 

417 

418 verbose: int, optional 

419 

420 Returns 

421 ------- 

422 mask: nibabel.Nifti1Image 

423 The brain mask (3D image) 

424 """ 

425 if verbose > 0: 

426 print("Background mask computation") 

427 

428 data_imgs = _utils.check_niimg(data_imgs) 

429 

430 # Delayed import to avoid circular imports 

431 from .image.image import _compute_mean 

432 data, affine = cache(_compute_mean, memory)(data_imgs, 

433 target_affine=target_affine, 

434 target_shape=target_shape, 

435 smooth=False) 

436 

437 if np.isnan(get_border_data(data, border_size)).any(): 

438 # We absolutely need to catter for NaNs as a background: 

439 # SPM does that by default 

440 mask = np.logical_not(np.isnan(data)) 

441 else: 

442 background = np.median(get_border_data(data, border_size)) 

443 mask = data != background 

444 

445 mask, affine = _post_process_mask(mask, affine, opening=opening, 

446 connected=connected, 

447 warning_msg="Are you sure that input " 

448 "images have a homogeneous background.") 

449 return new_img_like(data_imgs, mask, affine)

575 opening=2, memory=None, verbose=0): 

576 """Compute the whole-brain mask. This mask is calculated through the

578

585

593

598 After estimation of the largest connected constituent, 2`n` closing

602

605

609

642 memory=None, verbose=0, n_jobs=1, **kwargs): 

643 """ Compute a mask corresponding to the gray matter part of the brain for 

644 a list of images. 

645 The gray matter part is calculated through the resampling of MNI152 

646 template gray matter mask onto the target image 

647 

648 Parameters 

649 ---------- 

650 target_imgs: list of Niimg-like object 

651 See http://nilearn.github.io/manipulating_images/input_output.html 

652 Images used to compute the mask. 3D and 4D images are accepted. 

653 The images in this list must be of same shape and affine. The mask is 

654 calculated with the first element of the list for only the shape/affine 

655 of the image is used for this masking strategy 

656 

657 threshold: float, optional 

658 The value under which the MNI template is cut off. 

659 Default value is 0.5 

660 

661 connected: bool, optional 

662 if connected is True, only the largest connect component is kept. 

663 Default is True 

664 

665 opening: bool or int, optional 

666 if opening is True, a morphological opening is performed, to keep 

667 only large structures. 

668 If opening is an integer `n`, it is performed via `n` erosions. 

669 After estimation of the largest connected constituent, 2`n` closing 

670 operations are performed followed by `n` erosions. This corresponds 

671 to 1 opening operation of order `n` followed by a closing operator 

672 of order `n`. 

673 

674 memory: instance of joblib.Memory or str 

675 Used to cache the function call. 

676 

677 n_jobs: integer, optional 

678 Argument not used but kept to fit the API 

679 

680 **kwargs: optional arguments 

681 arguments such as 'target_affine' are used in the call of other 

682 masking strategies, which then would raise an error for this function 

683 which does not need such arguments. 

684 

685 verbose: int, optional 

686 Controls the amount of verbosity: higher numbers give 

687 more messages 

688 

689 Returns 

690 ------- 

691 mask: nibabel.Nifti1Image 

692 The brain mask (3D image) 

693 

694 See also 

695 -------- 

696 nilearn.masking.compute_brain_mask 

697 """ 

698 if len(target_imgs) == 0: 

699 raise TypeError('An empty object - %r - was passed instead of an ' 

700 'image or a list of images' % target_imgs) 

701 

702 # Check images in the list have the same FOV without loading them in memory 

703 imgs_generator = _utils.check_niimg(target_imgs, return_iterator=True) 

704 for _ in imgs_generator: 

705 pass 

706 

707 mask = compute_brain_mask(target_imgs[0], threshold=threshold, 

708 connected=connected, opening=opening, 

709 memory=memory, verbose=verbose) 

710 return mask

717def apply_mask(imgs, mask_img, dtype='f', 

718 smoothing_fwhm=None, ensure_finite=True): 

719 """Extract signals from images using specified mask. 

720 

721 Read the time series from the given Niimg-like object, using the mask. 

722 

723 Parameters 

724 ----------- 

725 imgs: list of 4D Niimg-like objects 

726 See http://nilearn.github.io/manipulating_images/input_output.html 

727 Images to be masked. list of lists of 3D images are also accepted. 

728 

729 mask_img: Niimg-like object 

730 See http://nilearn.github.io/manipulating_images/input_output.html 

731 3D mask array: True where a voxel should be used. 

732 

733 dtype: numpy dtype or 'f' 

734 The dtype of the output, if 'f', any float output is acceptable 

735 and if the data is stored on the disk as floats the data type 

736 will not be changed. 

737 

738 smoothing_fwhm: float 

739 (optional) Gives the size of the spatial smoothing to apply to 

740 the signal, in voxels. Implies ensure_finite=True. 

741 

742 ensure_finite: bool 

743 If ensure_finite is True (default), the non-finite values (NaNs and 

744 infs) found in the images will be replaced by zeros. 

745 

746 Returns 

747 -------- 

748 session_series: numpy.ndarray 

749 2D array of series with shape (image number, voxel number) 

750 

751 Notes 

752 ----- 

753 When using smoothing, ensure_finite is set to True, as non-finite 

754 values would spread accross the image. 

755 """ 

756 mask_img = _utils.check_niimg_3d(mask_img) 

757 mask, mask_affine = _load_mask_img(mask_img) 

758 mask_img = new_img_like(mask_img, mask, mask_affine) 

759 return _apply_mask_fmri(imgs, mask_img, dtype=dtype, 

760 smoothing_fwhm=smoothing_fwhm, 

761 ensure_finite=ensure_finite)

764def _apply_mask_fmri(imgs, mask_img, dtype='f', 

765 smoothing_fwhm=None, ensure_finite=True): 

766 """Same as apply_mask(). 

767 

768 The only difference with apply_mask is that some costly checks on mask_img 

769 are not performed: mask_img is assumed to contain only two different 

770 values (this is checked for in apply_mask, not in this function). 

771 """ 

772 

773 mask_img = _utils.check_niimg_3d(mask_img) 

774 mask_affine = mask_img.affine 

775 mask_data = _utils.as_ndarray(get_data(mask_img), 

776 dtype=bool) 

777 

778 if smoothing_fwhm is not None: 

779 ensure_finite = True 

780 

781 imgs_img = _utils.check_niimg(imgs) 

782 affine = imgs_img.affine[:3, :3] 

783 

784 if not np.allclose(mask_affine, imgs_img.affine): 

785 raise ValueError('Mask affine: \n%s\n is different from img affine:' 

786 '\n%s' % (str(mask_affine), 

787 str(imgs_img.affine))) 

788 

789 if not mask_data.shape == imgs_img.shape[:3]: 

790 raise ValueError('Mask shape: %s is different from img shape:%s' 

791 % (str(mask_data.shape), str(imgs_img.shape[:3]))) 

792 

793 # All the following has been optimized for C order. 

794 # Time that may be lost in conversion here is regained multiple times 

795 # afterward, especially if smoothing is applied. 

796 series = _safe_get_data(imgs_img) 

797 

798 if dtype == 'f': 

799 if series.dtype.kind == 'f': 

800 dtype = series.dtype 

801 else: 

802 dtype = np.float32 

803 series = _utils.as_ndarray(series, dtype=dtype, order="C", 

804 copy=True) 

805 del imgs_img # frees a lot of memory 

806 

807 # Delayed import to avoid circular imports 

808 from .image.image import _smooth_array 

809 _smooth_array(series, affine, fwhm=smoothing_fwhm, 

810 ensure_finite=ensure_finite, copy=False) 

811 return series[mask_data].T

814def _unmask_3d(X, mask, order="C"): 

815 """Take masked data and bring them back to 3D (space only). 

816 

817 Parameters 

818 ---------- 

819 X: numpy.ndarray 

820 Masked data. shape: (features,) 

821 

822 mask: Niimg-like object 

823 See http://nilearn.github.io/manipulating_images/input_output.html 

824 Mask. mask.ndim must be equal to 3, and dtype *must* be bool. 

825 """ 

826 

827 if mask.dtype != bool: 

828 raise TypeError("mask must be a boolean array") 

829 if X.ndim != 1: 

830 raise TypeError("X must be a 1-dimensional array") 

831 n_features = mask.sum() 

832 if X.shape[0] != n_features: 

833 raise TypeError('X must be of shape (samples, %d).' % n_features) 

834 

835 data = np.zeros( 

836 (mask.shape[0], mask.shape[1], mask.shape[2]), 

837 dtype=X.dtype, order=order) 

838 data[mask] = X 

839 return data

842def _unmask_4d(X, mask, order="C"): 

843 """Take masked data and bring them back to 4D. 

844 

845 Parameters 

846 ---------- 

847 X: numpy.ndarray 

848 Masked data. shape: (samples, features) 

849 

850 mask: numpy.ndarray 

851 Mask. mask.ndim must be equal to 4, and dtype *must* be bool. 

852 

853 Returns 

854 ------- 

855 data: numpy.ndarray 

856 Unmasked data.

861 raise TypeError("mask must be a boolean array") 

862 if X.ndim != 2: 

863 raise TypeError("X must be a 2-dimensional array")

865 if X.shape[1] != n_features: 

866 raise TypeError('X must be of shape (samples, %d).' % n_features)

870 return data

873def unmask(X, mask_img, order="F"): 

874 """Take masked data and bring them back into 3D/4D 

875 

876 This function can be applied to a list of masked data. 

877 

878 Parameters 

879 ---------- 

880 X: numpy.ndarray (or list of) 

881 Masked data. shape: (samples #, features #). 

882 If X is one-dimensional, it is assumed that samples# == 1. 

883 mask_img: niimg: Niimg-like object 

884 See http://nilearn.github.io/manipulating_images/input_output.html 

885 Must be 3-dimensional. 

886 

887 Returns 

888 ------- 

889 data: nibabel.Nift1Image object 

890 Unmasked data. Depending on the shape of X, data can have 

891 different shapes: 

892 

893 - X.ndim == 2: 

894 Shape: (mask.shape[0], mask.shape[1], mask.shape[2], X.shape[0]) 

895 - X.ndim == 1: 

896 Shape: (mask.shape[0], mask.shape[1], mask.shape[2]) 

897 """ 

898 # Handle lists. This can be a list of other lists / arrays, or a list or 

899 # numbers. In the latter case skip. 

900 if isinstance(X, list) and not isinstance(X[0], numbers.Number): 

901 ret = [] 

902 for x in X: 

903 ret.append(unmask(x, mask_img, order=order)) # 1-level recursion 

904 return ret 

905 

906 # The code after this block assumes that X is an ndarray; ensure this 

907 X = np.asanyarray(X) 

908 

909 mask_img = _utils.check_niimg_3d(mask_img) 

910 mask, affine = _load_mask_img(mask_img) 

911 

912 if np.ndim(X) == 2: 

913 unmasked = _unmask_4d(X, mask, order=order) 

914 elif np.ndim(X) == 1: 

915 unmasked = _unmask_3d(X, mask, order=order) 

916 else: 

917 raise TypeError("Masked data X must be 2D or 1D array; " 

918 "got shape: %s" % str(X.shape)) 

919 

920 return new_img_like(mask_img, unmasked, affine)
