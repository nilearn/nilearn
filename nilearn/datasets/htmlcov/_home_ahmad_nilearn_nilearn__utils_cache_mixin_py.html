26def _check_memory(memory, verbose=0): 

27 """Function to ensure an instance of a joblib.Memory object. 

28 

29 Parameters 

30 ---------- 

31 memory : None or instance of joblib.Memory or str 

32 Used to cache the masking process. 

33 If a str is given, it is the path to the caching directory. 

34 

35 verbose : int, optional 

36 Verbosity level. Default=0. 

37 

38 Returns 

39 ------- 

40 memory : instance of joblib.Memory. 

41 

42 """ 

43 if memory is None: 

44 memory = Memory(location=None, verbose=verbose) 

45 if isinstance(memory, str): 

46 cache_dir = memory 

47 if nilearn.EXPAND_PATH_WILDCARDS: 

48 cache_dir = os.path.expanduser(cache_dir) 

49 

50 # Perform some verifications on given path. 

51 split_cache_dir = os.path.split(cache_dir) 

52 if (len(split_cache_dir) > 1 and 

53 (not os.path.exists(split_cache_dir[0]) and 

54 split_cache_dir[0] != '')): 

55 if (not nilearn.EXPAND_PATH_WILDCARDS and 

56 cache_dir.startswith("~")): 

57 # Maybe the user want to enable expanded user path. 

58 error_msg = ("Given cache path parent directory doesn't " 

59 "exists, you gave '{0}'. Enabling " 

60 "nilearn.EXPAND_PATH_WILDCARDS could solve " 

61 "this issue.".format(split_cache_dir[0])) 

62 elif memory.startswith("~"): 

63 # Path built on top of expanded user path doesn't exist. 

64 error_msg = ("Given cache path parent directory doesn't " 

65 "exists, you gave '{0}' which was expanded " 

66 "as '{1}' but doesn't exist either. Use " 

67 "nilearn.EXPAND_PATH_WILDCARDS to deactivate " 

68 "auto expand user path (~) behavior." 

69 .format(split_cache_dir[0], 

70 os.path.dirname(memory))) 

71 else: 

72 # The given cache base path doesn't exist. 

73 error_msg = ("Given cache path parent directory doesn't " 

74 "exists, you gave '{0}'." 

75 .format(split_cache_dir[0])) 

76 raise ValueError(error_msg) 

77 

78 memory = Memory(location=cache_dir, verbose=verbose) 

79 return memory

82def _safe_cache(memory, func, **kwargs): 

83 """A wrapper for mem.cache that flushes the cache if the version 

84 number of nibabel has changed. 

85 

86 """ 

87 ''' Workaround for 

88 https://github.com/scikit-learn-contrib/imbalanced-learn/issues/482 

89 joblib throws a spurious warning with newer scikit-learn. 

90 This code uses the recommended method first and the deprecated one 

91 if that fails, ensuring th warning is not generated in any case. 

92 ''' 

93 try: 

94 location = os.path.join(memory.location, 'joblib') 

95 except AttributeError: 

96 location = memory.location 

97 except TypeError: 

98 location = None 

99 

100 if location is None or location in __CACHE_CHECKED: 

101 return memory.cache(func, **kwargs) 

102 

103 version_file = os.path.join(location, 'module_versions.json') 

104 

105 versions = dict() 

106 if os.path.exists(version_file): 

107 with open(version_file, 'r') as _version_file: 

108 versions = json.load(_version_file) 

109 

110 modules = (nibabel, ) 

111 # Keep only the major + minor version numbers 

112 my_versions = dict((m.__name__, LooseVersion(m.__version__).version[:2]) 

113 for m in modules) 

114 commons = set(versions.keys()).intersection(set(my_versions.keys())) 

115 collisions = [m for m in commons if versions[m] != my_versions[m]] 

116 

117 # Flush cache if version collision 

118 if len(collisions) > 0: 

119 if nilearn.CHECK_CACHE_VERSION: 

120 warnings.warn("Incompatible cache in %s: " 

121 "different version of nibabel. Deleting " 

122 "the cache. Put nilearn.CHECK_CACHE_VERSION " 

123 "to false to avoid this behavior." 

124 % location) 

125 try: 

126 tmp_dir = (os.path.split(location)[:-1] 

127 + ('old_%i' % os.getpid(), )) 

128 tmp_dir = os.path.join(*tmp_dir) 

129 # We use rename + unlink to be more robust to race 

130 # conditions 

131 os.rename(location, tmp_dir) 

132 shutil.rmtree(tmp_dir) 

133 except OSError: 

134 # Another process could have removed this dir 

135 pass 

136 

137 try: 

138 os.makedirs(location) 

139 except OSError: 

140 # File exists? 

141 pass 

142 else: 

143 warnings.warn("Incompatible cache in %s: " 

144 "old version of nibabel." % location) 

145 

146 # Write json files if configuration is different 

147 if versions != my_versions: 

148 with open(version_file, 'w') as _version_file: 

149 json.dump(my_versions, _version_file) 

150 

151 __CACHE_CHECKED[location] = True 

152 

153 return memory.cache(func, **kwargs)

156class _ShelvedFunc(object): 

157 """Work around for Python 2, for which pickle fails on instance method""" 

158 def __init__(self, func): 

159 self.func = func 

160 self.func_name = func.__name__ + '_shelved'

163 return self.func.call_and_shelve(*args, **kwargs)

166def cache(func, memory, func_memory_level=None, memory_level=None, 

167 shelve=False, **kwargs): 

168 """Return a joblib.Memory object. 

169 

170 The memory_level determines the level above which the wrapped 

171 function output is cached. By specifying a numeric value for 

172 this level, the user can to control the amount of cache memory 

173 used. This function will cache the function call or not 

174 depending on the cache level. 

175 

176 Parameters 

177 ---------- 

178 func : function 

179 The function which output is to be cached. 

180 

181 memory : instance of joblib.Memory or string 

182 Used to cache the function call. 

183 

184 func_memory_level : int, optional 

185 The memory_level from which caching must be enabled for the wrapped 

186 function. 

187 

188 memory_level : int, optional 

189 The memory_level used to determine if function call must 

190 be cached or not (if user_memory_level is equal of greater than 

191 func_memory_level the function is cached). 

192 

193 shelve : bool, optional 

194 Whether to return a joblib MemorizedResult, callable by a .get() 

195 method, instead of the return value of func. 

196 Default=False. 

197 

198 kwargs : keyword arguments, optional 

199 The keyword arguments passed to memory.cache. 

200 

201 Returns 

202 ------- 

203 mem : joblib.MemorizedFunc, wrapped in _ShelvedFunc if shelving 

204 Object that wraps the function func to cache its further call. 

205 This object may be a no-op, if the requested level is lower 

206 than the value given to _cache()). 

207 For consistency, a callable object is always returned. 

208 

209 """ 

210 verbose = kwargs.get('verbose', 0) 

211 

212 # memory_level and func_memory_level must be both None or both integers. 

213 memory_levels = [memory_level, func_memory_level] 

214 both_params_integers = all(isinstance(lvl, int) for lvl in memory_levels) 

215 both_params_none = all(lvl is None for lvl in memory_levels) 

216 

217 if not (both_params_integers or both_params_none): 

218 raise ValueError('Reference and user memory levels must be both None ' 

219 'or both integers.') 

220 

221 if memory is not None and (func_memory_level is None or 

222 memory_level >= func_memory_level): 

223 if isinstance(memory, str): 

224 memory = Memory(location=memory, verbose=verbose) 

225 if not isinstance(memory, MEMORY_CLASSES): 

226 raise TypeError("'memory' argument must be a string or a " 

227 "joblib.Memory object. " 

228 "%s %s was given." % (memory, type(memory))) 

229 if (memory.location is None and memory_level is not None 

230 and memory_level > 1): 

231 warnings.warn("Caching has been enabled (memory_level = %d) " 

232 "but no Memory object or path has been provided" 

233 " (parameter memory). Caching deactivated for " 

234 "function %s." % 

235 (memory_level, func.__name__), 

236 stacklevel=2) 

237 else: 

238 memory = Memory(location=None, verbose=verbose) 

239 cached_func = _safe_cache(memory, func, **kwargs) 

240 if shelve: 

241 cached_func = _ShelvedFunc(cached_func) 

242 return cached_func

245class CacheMixin(object): 

246 """Mixin to add caching to a class. 

247 

248 This class is a thin layer on top of joblib.Memory, that mainly adds a 

249 "caching level", similar to a "log level". 

250 

251 Notes 

252 ----- 

253 Usage: to cache the results of a method, wrap it in self._cache() 

254 defined by this class. Caching is performed only if the user-specified 

255 cache level (self._memory_level) is greater than the value given as a 

256 parameter to self._cache(). See _cache() documentation for details. 

257 

258 """ 

259 def _cache(self, func, func_memory_level=1, shelve=False, **kwargs): 

260 """Return a joblib.Memory object. 

261 

262 The memory_level determines the level above which the wrapped 

263 function output is cached. By specifying a numeric value for 

264 this level, the user can to control the amount of cache memory 

265 used. This function will cache the function call or not 

266 depending on the cache level. 

267 

268 Parameters 

269 ---------- 

270 func : function 

271 The function the output of which is to be cached. 

272 

273 func_memory_level : int, optional 

274 The memory_level from which caching must be enabled for the wrapped 

275 function. Default=1. 

276 

277 shelve : bool, optional 

278 Whether to return a joblib MemorizedResult, callable by a .get() 

279 method, instead of the return value of func. Default=False. 

280 

281 Returns 

282 ------- 

283 mem : joblib.MemorizedFunc, wrapped in _ShelvedFunc if shelving 

284 Object that wraps the function func to cache its further call. 

285 This object may be a no-op, if the requested level is lower 

286 than the value given to _cache()). 

287 For consistency, a callable object is always returned. 

288 

289 """ 

290 verbose = getattr(self, 'verbose', 0) 

291 

292 # Creates attributes if they don't exist 

293 # This is to make creating them in __init__() optional. 

294 if not hasattr(self, "memory_level"): 

295 self.memory_level = 0 

296 if not hasattr(self, "memory"): 

297 self.memory = Memory(location=None, verbose=verbose) 

298 self.memory = _check_memory(self.memory, verbose=verbose) 

299 

300 # If cache level is 0 but a memory object has been provided, set 

301 # memory_level to 1 with a warning. 

302 if self.memory_level == 0 and self.memory.location is not None: 

303 warnings.warn("memory_level is currently set to 0 but " 

304 "a Memory object has been provided. " 

305 "Setting memory_level to 1.") 

306 self.memory_level = 1 

307 

308 return cache(func, self.memory, func_memory_level=func_memory_level, 

309 memory_level=self.memory_level, shelve=shelve, 

310 **kwargs)
