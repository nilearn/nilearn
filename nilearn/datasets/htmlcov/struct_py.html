25def fetch_icbm152_2009(data_dir=None, url=None, resume=True, verbose=1): 

26 """Download and load the ICBM152 template (dated 2009). 

27

54 white matter and cerebrospinal fluid. Values are file paths. 

55 "eye_mask", "face_mask", "mask": use these images to mask out 

56 parts of mri images. Values are file paths. 

57 

58 References 

59 ---------- 

60 .. [1] VS Fonov, AC Evans, K Botteron, CR Almli, RC McKinstry, DL Collins

67 Organization for Human Brain Mapping 2009 Annual Meeting. 

68

70 "ANIMAL+INSECT: Improved Cortical Structure Segmentation",

74 -----

76 http://www.bic.mni.mcgill.ca/ServicesAtlases/ICBM152NLin2009

89 keys = ("csf", "gm", "wm",

91 "eye_mask", "face_mask", "mask") 

92 filenames = [(os.path.join("mni_icbm152_nlin_sym_09a", name), url, opts)

94 "mni_icbm152_csf_tal_nlin_sym_09a.nii.gz", 

95 "mni_icbm152_gm_tal_nlin_sym_09a.nii.gz",

102 

103 "mni_icbm152_t1_tal_nlin_sym_09a_eye_mask.nii.gz", 

104 "mni_icbm152_t1_tal_nlin_sym_09a_face_mask.nii.gz", 

105 "mni_icbm152_t1_tal_nlin_sym_09a_mask.nii.gz")] 

106 

107 dataset_name = 'icbm152_2009'

172def fetch_icbm152_brain_gm_mask(data_dir=None, threshold=0.2, resume=True, 

173 verbose=1): 

174 """Downloads ICBM152 template first, then loads 'gm' mask image. 

175 

176 .. versionadded:: 0.2.5 

177 

178 Parameters 

179 ---------- 

180 data_dir : str, optional 

181 Path of the data directory. Used to force storage in a specified 

182 location. Defaults to None. 

183 

184 threshold : float, optional 

185 The parameter which amounts to include the values in the mask image. 

186 The values lies above than this threshold will be included. Defaults 

187 to 0.2 (one fifth) of values. 

188 Default=0.2. 

189 

190 resume : bool, optional 

191 If True, try resuming partially downloaded data. 

192 Default=True. 

193 

194 verbose : int, optional 

195 Verbosity level (0 means no message). Default=1. 

196 

197 Returns 

198 ------- 

199 gm_mask_img : Nifti image 

200 Corresponding to brain grey matter from ICBM152 template. 

201 

202 Notes 

203 ----- 

204 This function relies on ICBM152 templates where we particularly pick 

205 grey matter template and threshold the template at .2 to take one fifth 

206 of the values. Then, do a bit post processing such as binary closing 

207 operation to more compact mask image. 

208 

209 Note: It is advised to check the mask image with your own data processing. 

210 

211 See Also 

212 -------- 

213 nilearn.datasets.fetch_icbm152_2009: for details regarding the ICBM152 

214 template. 

215 

216 nilearn.datasets.load_mni152_template: for details about version of MNI152 

217 template and related. 

218 

219 """ 

220 # Fetching ICBM152 grey matter mask image 

221 icbm = fetch_icbm152_2009(data_dir=data_dir, resume=resume, verbose=verbose) 

222 gm = icbm['gm'] 

223 gm_img = check_niimg(gm) 

224 gm_data = niimg._safe_get_data(gm_img) 

225 

226 # getting one fifth of the values 

227 gm_mask = (gm_data > threshold) 

228 

229 gm_mask = ndimage.binary_closing(gm_mask, iterations=2) 

230 gm_mask_img = new_img_like(gm_img, gm_mask) 

231 return gm_mask_img

234def fetch_oasis_vbm(n_subjects=None, dartel_version=True, data_dir=None, 

235 url=None, resume=True, verbose=1): 

236 """Download and load Oasis "cross-sectional MRI" dataset (416 subjects). 

237 

238 For more information, see [1]_ and [2]_. 

239 

240 Parameters 

241 ---------- 

242 n_subjects : int, optional 

243 The number of subjects to load. If None is given, all the 

244 subjects are used. 

245 

246 dartel_version : boolean, optional 

247 Whether or not to use data normalized with DARTEL instead of standard 

248 SPM8 normalization. Default=True. 

249 

250 data_dir : string, optional 

251 Path of the data directory. Used to force data storage in a specified 

252 location. Default: None 

253 

254 url : string, optional 

255 Override download URL. Used for test only (or if you setup a mirror of 

256 the data). 

257 

258 resume : bool, optional 

259 If true, try resuming download if possible. Default=True. 

260 

261 verbose : int, optional 

262 Verbosity level (0 means no message). Default=1. 

263 

264 Returns 

265 ------- 

266 data : Bunch 

267 Dictionary-like object, the interest attributes are : 

268 

269 - 'gray_matter_maps': string list 

270 Paths to nifti gray matter density probability maps 

271 - 'white_matter_maps' string list 

272 Paths to nifti white matter density probability maps 

273 - 'ext_vars': np.recarray 

274 Data from the .csv file with information about selected subjects 

275 - 'data_usage_agreement': string 

276 Path to the .txt file containing the data usage agreement. 

277 

278 References 

279 ---------- 

280 .. [1] http://www.oasis-brains.org/ 

281 

282 .. [2] Open Access Series of Imaging Studies (OASIS): Cross-sectional MRI 

283 Data in Young, Middle Aged, Nondemented, and Demented Older Adults. 

284 Marcus, D. S and al., 2007, Journal of Cognitive Neuroscience. 

285 

286 Notes 

287 ----- 

288 In the DARTEL version, original Oasis data have been preprocessed 

289 with the following steps: 

290 

291 1. Dimension swapping (technically required for subsequent steps) 

292 2. Brain Extraction 

293 3. Segmentation with SPM8 

294 4. Normalization using DARTEL algorithm 

295 5. Modulation 

296 6. Replacement of NaN values with 0 in gray/white matter density maps. 

297 7. Resampling to reduce shape and make it correspond to the shape of 

298 the non-DARTEL data (fetched with dartel_version=False). 

299 8. Replacement of values < 1e-4 with zeros to reduce the file size. 

300 

301 In the non-DARTEL version, the following steps have been performed instead: 

302 

303 1. Dimension swapping (technically required for subsequent steps) 

304 2. Brain Extraction 

305 3. Segmentation and normalization to a template with SPM8 

306 4. Modulation 

307 5. Replacement of NaN values with 0 in gray/white matter density maps. 

308 

309 An archive containing the gray and white matter density probability maps 

310 for the 416 available subjects is provided. Gross outliers are removed and 

311 filtered by this data fetcher (DARTEL: 13 outliers; non-DARTEL: 1 outlier) 

312 Externals variates (age, gender, estimated intracranial volume, 

313 years of education, socioeconomic status, dementia score) are provided 

314 in a CSV file that is a copy of the original Oasis CSV file. The current 

315 downloader loads the CSV file and keeps only the lines corresponding to 

316 the subjects that are actually demanded. 

317 

318 The Open Access Structural Imaging Series (OASIS) is a project 

319 dedicated to making brain imaging data openly available to the public. 

320 Using data available through the OASIS project requires agreeing with 

321 the Data Usage Agreement that can be found at 

322 http://www.oasis-brains.org/app/template/UsageAgreement.vm 

323 

324 """ 

325 # check number of subjects 

326 if n_subjects is None: 

327 n_subjects = 403 if dartel_version else 415 

328 if dartel_version: # DARTEL version has 13 identified outliers 

329 if n_subjects > 403: 

330 warnings.warn('Only 403 subjects are available in the ' 

331 'DARTEL-normalized version of the dataset. ' 

332 'All of them will be used instead of the wanted %d' 

333 % n_subjects) 

334 n_subjects = 403 

335 else: # all subjects except one are available with non-DARTEL version 

336 if n_subjects > 415: 

337 warnings.warn('Only 415 subjects are available in the ' 

338 'non-DARTEL-normalized version of the dataset. ' 

339 'All of them will be used instead of the wanted %d' 

340 % n_subjects) 

341 n_subjects = 415 

342 if n_subjects < 1: 

343 raise ValueError("Incorrect number of subjects (%d)" % n_subjects)

359 url_csv = url + "/oasis_cross-sectional.csv" 

360 url_dua = url + "/data_usage_agreement.txt" 

361 if dartel_version: 

362 url_images = url + "/archive_dartel.tgz" 

363 else: 

364 url_images = url + "/archive.tgz" 

365

367 

368 # missing subjects create shifts in subjects ids 

369 missing_subjects = [8, 24, 36, 48, 89, 93, 100, 118, 128, 149, 154, 

370 171, 172, 175, 187, 194, 196, 215, 219, 225, 242, 

371 245, 248, 251, 252, 257, 276, 297, 306, 320, 324, 

372 334, 347, 360, 364, 391, 393, 412, 414, 427, 436] 

373 

374 if dartel_version: 

375 # DARTEL produces outliers that are hidden by nilearn API 

376 removed_outliers = [27, 57, 66, 83, 122, 157, 222, 269, 282, 287, 

377 309, 428] 

378 missing_subjects = sorted(missing_subjects + removed_outliers) 

379 file_names_gm = [ 

380 (os.path.join( 

381 "OAS1_%04d_MR1", 

382 "mwrc1OAS1_%04d_MR1_mpr_anon_fslswapdim_bet.nii.gz") 

383 % (s, s), 

384 url_images, opts) 

385 for s in range(1, 457) if s not in missing_subjects][:n_subjects] 

386 file_names_wm = [ 

387 (os.path.join( 

388 "OAS1_%04d_MR1",

390 % (s, s), 

391 url_images, opts) 

392 for s in range(1, 457) if s not in missing_subjects] 

393 else: 

394 # only one gross outlier produced, hidden by nilearn API 

395 removed_outliers = [390] 

396 missing_subjects = sorted(missing_subjects + removed_outliers) 

397 file_names_gm = [ 

398 (os.path.join( 

399 "OAS1_%04d_MR1", 

400 "mwc1OAS1_%04d_MR1_mpr_anon_fslswapdim_bet.nii.gz")

452def fetch_surf_fsaverage(mesh='fsaverage5', data_dir=None): 

453 """Download a Freesurfer fsaverage surface. 

454 See [1]_. 

455 

456 Parameters 

457 ---------- 

458 mesh : str, optional 

459 Which mesh to fetch. Default='fsaverage5'. 

460 

461 - 'fsaverage5': the low-resolution fsaverage5 mesh (10242 nodes) 

462 - 'fsaverage5_sphere': the low-resolution fsaverage5 spheres (10242 nodes) 

463 - 'fsaverage': the high-resolution fsaverage mesh (163842 nodes) 

464 (high-resolution fsaverage will result in more computation time and memory usage) 

465 

466 data_dir : str, optional 

467 Path of the data directory. Used to force data storage in a specified 

468 location. 

469 

470 Returns 

471 ------- 

472 data : sklearn.datasets.base.Bunch 

473 Dictionary-like object, the interest attributes are : 

474 - 'pial_left': Gifti file, left hemisphere pial surface mesh 

475 - 'pial_right': Gifti file, right hemisphere pial surface mesh 

476 - 'infl_left': Gifti file, left hemisphere inflated pial surface mesh 

477 - 'infl_right': Gifti file, right hemisphere inflated pial 

478 surface mesh 

479 - 'sulc_left': Gifti file, left hemisphere sulcal depth data 

480 - 'sulc_right': Gifti file, right hemisphere sulcal depth data 

481 

482 References 

483 ---------- 

484 .. [1] Fischl et al, (1999). High-resolution intersubject averaging and a 

485 coordinate system for the cortical surface. Hum Brain Mapp 8, 272-284. 

486 

487 """ 

488 meshes = {'fsaverage5': _fetch_surf_fsaverage5, 

489 'fsaverage5_sphere': _fetch_surf_fsaverage5_sphere, 

490 'fsaverage': _fetch_surf_fsaverage} 

491 if mesh not in meshes: 

492 raise ValueError( 

493 "'mesh' should be one of {}; {!r} was provided".format( 

494 list(meshes.keys()), mesh)) 

495 return meshes[mesh](data_dir=data_dir)

498def _fetch_surf_fsaverage(data_dir=None): 

499 """Helper function to ship fsaverage (highest resolution) surfaces 

500 and sulcal information with Nilearn. 

501 

502 The source of the data is downloaded from nitrc. 

503 

504 """

507 file_names = ["{}_{}".format(part, hemi)

517 for name in file_names}
