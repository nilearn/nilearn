14def largest_connected_component(volume): 

15 """Return the largest connected component of a 3D array. 

16 

17 Parameters 

18 ----------- 

19 volume : numpy.ndarray 

20 3D boolean array indicating a volume. 

21 

22 Returns 

23 -------- 

24 volume : numpy.ndarray 

25 3D boolean array with only one connected component. 

26 

27 See Also 

28 -------- 

29 nilearn.image.largest_connected_component_img : To simply operate the 

30 same manipulation directly on Nifti images. 

31 

32 Notes 

33 ----- 

34 **Handling big-endian in given numpy.ndarray** 

35 This function changes the existing byte-ordering information to new byte 

36 order, if the given volume has non-native data type. This operation 

37 is done inplace to avoid big-endian issues with scipy ndimage module. 

38 

39 """ 

40 if (hasattr(volume, "get_data") or hasattr( 

41 volume, "get_fdata") or isinstance(volume, str)): 

42 raise ValueError('Please enter a valid numpy array. For images use\ 

43 largest_connected_component_img') 

44 # Get the new byteorder to handle issues like "Big-endian buffer not 

45 # supported on little-endian compiler" with scipy ndimage label. 

46 if not volume.dtype.isnative: 

47 volume.dtype = volume.dtype.newbyteorder('N') 

48 

49 # We use asarray to be able to work with masked arrays. 

50 volume = np.asarray(volume) 

51 labels, label_nb = ndimage.label(volume) 

52 if not label_nb: 

53 raise ValueError('No non-zero values: no connected components') 

54 if label_nb == 1: 

55 return volume.astype(bool) 

56 label_count = np.bincount(labels.ravel().astype(int)) 

57 # discard the 0 label 

58 label_count[0] = 0 

59 return labels == label_count.argmax()

73def _peak_local_max(image, min_distance=10, threshold_abs=0, threshold_rel=0.1, 

74 num_peaks=np.inf): 

75 """Find peaks in an image, and return them as coordinates or a boolean array. 

76 

77 Peaks are the local maxima in a region of `2 * min_distance + 1` 

78 (i.e. peaks are separated by at least `min_distance`). 

79 

80 Parameters 

81 ---------- 

82 image : ndarray of floats 

83 Input image. 

84 

85 min_distance : int, optional 

86 Minimum number of pixels separating peaks in a region of `2 * 

87 min_distance + 1` (i.e. peaks are separated by at least 

88 `min_distance`). To find the maximum number of peaks, use 

89 `min_distance=1`. 

90 Default=10. 

91 

92 threshold_abs : float, optional 

93 Minimum intensity of peaks. Default=0. 

94 

95 threshold_rel : float, optional 

96 Minimum intensity of peaks calculated as `max(image) * threshold_rel`. 

97 Default=0.1. 

98 

99 num_peaks : int, optional 

100 Maximum number of peaks. When the number of peaks exceeds `num_peaks`, 

101 return `num_peaks` peaks based on highest peak intensity. 

102 Default=np.inf. 

103 

104 Returns 

105 ------- 

106 output : ndarray or ndarray of bools 

107 Boolean array shaped like `image`, with peaks represented by True 

108 values. 

109 

110 Notes 

111 ----- 

112 If peaks are flat (i.e. multiple adjacent pixels have identical 

113 intensities), the coordinates of all such pixels are returned. 

114 

115 The peak local maximum function returns the coordinates of local peaks 

116 (maxima) in a image. A maximum filter is used for finding local maxima. 

117 This operation dilates the original image. After comparison between 

118 dilated and original image, peak_local_max function returns the 

119 coordinates of peaks where dilated image = original. 

120 

121 This code is mostly adapted from scikit image 0.11.3 release. 

122 Location of file in scikit image: peak_local_max function in 

123 skimage.feature.peak 

124 

125 """ 

126 out = np.zeros_like(image, dtype=bool) 

127 

128 if np.all(image == image.flat[0]): 

129 return out 

130 

131 image = image.copy() 

132 

133 size = 2 * min_distance + 1 

134 image_max = ndimage.maximum_filter(image, size=size, mode='constant') 

135 

136 mask = (image == image_max) 

137 image *= mask 

138 

139 # find top peak candidates above a threshold 

140 peak_threshold = max(np.max(image.ravel()) * threshold_rel, threshold_abs) 

141 

142 # get coordinates of peaks 

143 coordinates = np.argwhere(image > peak_threshold) 

144 

145 if coordinates.shape[0] > num_peaks: 

146 intensities = image.flat[np.ravel_multi_index(coordinates.transpose(), 

147 image.shape)] 

148 idx_maxsort = np.argsort(intensities)[::-1] 

149 coordinates = coordinates[idx_maxsort][:num_peaks] 

150 

151 nd_indices = tuple(coordinates.T) 

152 out[nd_indices] = True 

153 return out
