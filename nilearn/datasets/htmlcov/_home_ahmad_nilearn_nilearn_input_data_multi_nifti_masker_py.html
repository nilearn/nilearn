23class MultiNiftiMasker(NiftiMasker, CacheMixin): 

24 """Class for masking of Niimg-like objects. 

25 

26 MultiNiftiMasker is useful when dealing with image sets from multiple 

27 subjects. Use case: integrates well with decomposition by MultiPCA and 

28 CanICA (multi-subject models) 

29 

30 Parameters 

31 ---------- 

32 mask_img : Niimg-like object 

33 See http://nilearn.github.io/manipulating_images/input_output.html 

34 Mask of the data. If not given, a mask is computed in the fit step. 

35 Optional parameters can be set using mask_args and mask_strategy to 

36 fine tune the mask extraction. 

37 

38 smoothing_fwhm : float, optional 

39 If smoothing_fwhm is not None, it gives the size in millimeters of 

40 the spatial smoothing to apply to the signal. 

41 

42 standardize : {False, True, 'zscore', 'psc'}, optional 

43 Strategy to standardize the signal. 

44 'zscore': the signal is z-scored. Timeseries are shifted 

45 to zero mean and scaled to unit variance. 

46 'psc': Timeseries are shifted to zero mean value and scaled 

47 to percent signal change (as compared to original mean signal). 

48 True : the signal is z-scored. Timeseries are shifted 

49 to zero mean and scaled to unit variance. 

50 False : Do not standardize the data. 

51 Default=False. 

52 

53 standardize_confounds : boolean, optional 

54 If standardize_confounds is True, the confounds are z-scored: 

55 their mean is put to 0 and their variance to 1 in the time dimension. 

56 Default=True. 

57 

58 high_variance_confounds : boolean, optional 

59 If True, high variance confounds are computed on provided image with 

60 :func:`nilearn.image.high_variance_confounds` and default parameters 

61 and regressed out. Default=False. 

62 

63 detrend : boolean, optional 

64 This parameter is passed to signal.clean. Please see the related 

65 documentation for details. Default=False. 

66 

67 low_pass : None or float, optional 

68 This parameter is passed to signal.clean. Please see the related 

69 documentation for details 

70 

71 high_pass : None or float, optional 

72 This parameter is passed to signal.clean. Please see the related 

73 documentation for details 

74 

75 t_r : float, optional 

76 This parameter is passed to signal.clean. Please see the related 

77 documentation for details 

78 

79 target_affine : 3x3 or 4x4 matrix, optional 

80 This parameter is passed to image.resample_img. Please see the 

81 related documentation for details. 

82 

83 target_shape : 3-tuple of integers, optional 

84 This parameter is passed to image.resample_img. Please see the 

85 related documentation for details. 

86 

87 mask_strategy : {'background', 'epi' or 'template'}, optional 

88 The strategy used to compute the mask: use 'background' if your 

89 images present a clear homogeneous background, 'epi' if they 

90 are raw EPI images, or you could use 'template' which will 

91 extract the gray matter part of your data by resampling the MNI152 

92 brain mask for your data's field of view. 

93 Depending on this value, the mask will be computed from 

94 masking.compute_background_mask, masking.compute_epi_mask or 

95 masking.compute_brain_mask. Default is 'background'. 

96 

97 mask_args : dict, optional 

98 If mask is None, these are additional parameters passed to 

99 masking.compute_background_mask or masking.compute_epi_mask 

100 to fine-tune mask computation. Please see the related documentation 

101 for details. 

102 

103 dtype : {dtype, "auto"}, optional 

104 Data type toward which the data should be converted. If "auto", the 

105 data will be converted to int32 if dtype is discrete and float32 if it 

106 is continuous. 

107 

108 memory : instance of joblib.Memory or string, optional 

109 Used to cache the masking process. 

110 By default, no caching is done. If a string is given, it is the 

111 path to the caching directory. 

112 

113 memory_level : integer, optional 

114 Rough estimator of the amount of memory used by caching. Higher value 

115 means more memory for caching. Default=0. 

116 

117 n_jobs : integer, optional 

118 The number of CPUs to use to do the computation. -1 means 

119 'all CPUs', -2 'all CPUs but one', and so on. Default=1. 

120 

121 verbose : integer, optional 

122 Indicate the level of verbosity. By default, nothing is printed. 

123 Default=0. 

124 

125 Attributes 

126 ---------- 

127 `mask_img_` : nibabel.Nifti1Image object 

128 The mask of the data. 

129 

130 `affine_` : 4x4 numpy.ndarray 

131 Affine of the transformed image. 

132 

133 See Also 

134 -------- 

135 nilearn.image.resample_img: image resampling 

136 nilearn.masking.compute_epi_mask: mask computation 

137 nilearn.masking.apply_mask: mask application on image 

138 nilearn.signal.clean: confounds removal and general filtering of signals 

139 

140 """

174 def fit(self, imgs=None, y=None): 

175 """Compute the mask corresponding to the data 

176 

177 Parameters 

178 ---------- 

179 imgs : list of Niimg-like objects 

180 See http://nilearn.github.io/manipulating_images/input_output.html 

181 Data on which the mask must be calculated. If this is a list, 

182 the affine is considered the same for all. 

183 

184 """ 

185 

186 # Load data (if filenames are given, load them) 

187 if self.verbose > 0: 

188 print("[%s.fit] Loading data from %s" % ( 

189 self.__class__.__name__, 

190 _utils._repr_niimgs(imgs, shorten=False))) 

191 # Compute the mask if not given by the user 

192 if self.mask_img is None: 

193 if self.verbose > 0: 

194 print("[%s.fit] Computing mask" % self.__class__.__name__) 

195 if not isinstance(imgs, collections.abc.Iterable) \ 

196 or isinstance(imgs, str): 

197 raise ValueError("[%s.fit] For multiple processing, you should" 

198 " provide a list of data " 

199 "(e.g. Nifti1Image objects or filenames)." 

200 "%r is an invalid input" 

201 % (self.__class__.__name__, imgs)) 

202 

203 mask_args = (self.mask_args if self.mask_args is not None 

204 else {}) 

205 if self.mask_strategy == 'background': 

206 compute_mask = masking.compute_multi_background_mask 

207 elif self.mask_strategy == 'epi': 

208 compute_mask = masking.compute_multi_epi_mask 

209 elif self.mask_strategy == 'template': 

210 compute_mask = masking.compute_multi_gray_matter_mask 

211 else: 

212 raise ValueError("Unknown value of mask_strategy '%s'. " 

213 "Acceptable values are 'background', 'epi' " 

214 "and 'template'.") 

215 

216 self.mask_img_ = self._cache( 

217 compute_mask, ignore=['n_jobs', 'verbose', 'memory'])( 

218 imgs, 

219 target_affine=self.target_affine, 

220 target_shape=self.target_shape, 

221 n_jobs=self.n_jobs, 

222 memory=self.memory, 

223 verbose=max(0, self.verbose - 1), 

224 **mask_args) 

225 else: 

226 if imgs is not None: 

227 warnings.warn('[%s.fit] Generation of a mask has been' 

228 ' requested (imgs != None) while a mask has' 

229 ' been provided at masker creation. Given mask' 

230 ' will be used.' % self.__class__.__name__) 

231 self.mask_img_ = _utils.check_niimg_3d(self.mask_img) 

232 

233 # If resampling is requested, resample the mask as well. 

234 # Resampling: allows the user to change the affine, the shape or both. 

235 if self.verbose > 0: 

236 print("[%s.transform] Resampling mask" % self.__class__.__name__) 

237 self.mask_img_ = self._cache(image.resample_img)( 

238 self.mask_img_, 

239 target_affine=self.target_affine, 

240 target_shape=self.target_shape, 

241 interpolation='nearest', copy=False) 

242 if self.target_affine is not None: 

243 self.affine_ = self.target_affine 

244 else: 

245 self.affine_ = self.mask_img_.affine 

246 # Load data in memory 

247 get_data(self.mask_img_) 

248 return self

253 Parameters

270 'all cpus'. Default=1.

272 Returns

275 List of signal for each element per subject. 

276 shape: list of (number of scans, number of elements)

280 if not hasattr(self, 'mask_img_'): 

281 raise ValueError('It seems that %s has not been fitted. ' 

282 'You must call fit() before calling transform().'

285 if self.target_affine is None: 

286 # Force resampling on first image

299 # Ignore the mask-computing params: they are not useful and will
