23def to_matrix_vector(transform): 

24 """Split an homogeneous transform into its matrix and vector components. 

25 

26 The transformation must be represented in homogeneous coordinates. 

27 It is split into its linear transformation matrix and translation vector 

28 components. 

29 

30 This function does not normalize the matrix. This means that for it to be 

31 the inverse of from_matrix_vector, transform[-1, -1] must equal 1, and 

32 transform[-1, :-1] must equal 0. 

33 

34 Parameters 

35 ---------- 

36 transform : numpy.ndarray 

37 Homogeneous transform matrix. Example: a (4, 4) transform representing 

38 linear transformation and translation in 3 dimensions. 

39 

40 Returns 

41 ------- 

42 matrix, vector : numpy.ndarray 

43 The matrix and vector components of the transform matrix. For 

44 an (N, N) transform, matrix will be (N-1, N-1) and vector will be 

45 a 1D array of shape (N-1,). 

46 

47 See Also 

48 -------- 

49 from_matrix_vector 

50 

51 """ 

52 ndimin = transform.shape[0] - 1 

53 ndimout = transform.shape[1] - 1 

54 matrix = transform[0:ndimin, 0:ndimout] 

55 vector = transform[0:ndimin, ndimout] 

56 return matrix, vector

64 

65 Parameters

69 

70 vector : numpy.ndarray

72

74 ------- 

75 xform: numpy.ndarray 

76 An (N+1, N+1) transform matrix. 

77 

78 See Also 

79 --------

91def coord_transform(x, y, z, affine): 

92 """ Convert the x, y, z coordinates from one image space to another 

93 space. 

94 

95 Parameters 

96 ---------- 

97 x : number or ndarray (any shape) 

98 The x coordinates in the input space. 

99 

100 y : number or ndarray (same shape as x) 

101 The y coordinates in the input space. 

102 

103 z : number or ndarray 

104 The z coordinates in the input space. 

105 

106 affine : 2D 4x4 ndarray 

107 Affine that maps from input to output space. 

108 

109 Returns 

110 ------- 

111 x : number or ndarray (same shape as input) 

112 The x coordinates in the output space. 

113 

114 y : number or ndarray (same shape as input) 

115 The y coordinates in the output space. 

116 

117 z : number or ndarray (same shape as input) 

118 The z coordinates in the output space. 

119 

120 Warning: The x, y and z have their output space (e.g. MNI) coordinate 

121 ordering, not 3D numpy image ordering. 

122 

123 Examples 

124 -------- 

125 Transform data from coordinates to brain space. The "affine" matrix 

126 can be found as the ".affine" attribute of a nifti image, or using 

127 the "get_affine()" method for older nibabel installations:: 

128 

129 >>> from nilearn import datasets, image 

130 >>> niimg = datasets.load_mni152_template() 

131 >>> # Find the MNI coordinates of the voxel (50, 50, 50) 

132 >>> image.coord_transform(50, 50, 50, niimg.affine) 

133 (-10.0, -26.0, 28.0) 

134 

135 """ 

136 squeeze = (not hasattr(x, '__iter__')) 

137 return_number = isinstance(x, numbers.Number) 

138 x = np.asanyarray(x) 

139 shape = x.shape 

140 coords = np.c_[np.atleast_1d(x).flat, 

141 np.atleast_1d(y).flat, 

142 np.atleast_1d(z).flat, 

143 np.ones_like(np.atleast_1d(z).flat)].T 

144 x, y, z, _ = np.dot(affine, coords) 

145 if return_number: 

146 return x.item(), y.item(), z.item() 

147 if squeeze: 

148 return x.squeeze(), y.squeeze(), z.squeeze() 

149 return np.reshape(x, shape), np.reshape(y, shape), np.reshape(z, shape)

152def get_bounds(shape, affine): 

153 """Return the world-space bounds occupied by an array given an affine. 

154 

155 The coordinates returned correspond to the **center** of the corner voxels. 

156 

157 Parameters 

158 ---------- 

159 shape : tuple 

160 shape of the array. Must have 3 integer values. 

161 

162 affine : numpy.ndarray 

163 affine giving the linear transformation between voxel coordinates 

164 and world-space coordinates. 

165 

166 Returns 

167 ------- 

168 coord : list of tuples 

169 coord[i] is a 2-tuple giving minimal and maximal coordinates along 

170 i-th axis. 

171 

172 """ 

173 adim, bdim, cdim = shape 

174 adim -= 1 

175 bdim -= 1 

176 cdim -= 1 

177 # form a collection of vectors for each 8 corners of the box 

178 box = np.array([[0., 0, 0, 1], 

179 [adim, 0, 0, 1], 

180 [0, bdim, 0, 1], 

181 [0, 0, cdim, 1], 

182 [adim, bdim, 0, 1], 

183 [adim, 0, cdim, 1], 

184 [0, bdim, cdim, 1], 

185 [adim, bdim, cdim, 1]]).T 

186 box = np.dot(affine, box)[:3] 

187 return list(zip(box.min(axis=-1), box.max(axis=-1)))

190def get_mask_bounds(img): 

191 """ Return the world-space bounds occupied by a mask. 

192 

193 Parameters 

194 ---------- 

195 img : Niimg-like object 

196 See http://nilearn.github.io/manipulating_images/input_output.html 

197 The image to inspect. Zero values are considered as 

198 background. 

199 

200 Returns 

201 -------- 

202 xmin, xmax, ymin, ymax, zmin, zmax : floats 

203 The world-space bounds (field of view) occupied by the 

204 non-zero values in the image 

205 

206 Notes 

207 ----- 

208 The image should have only one connect component. 

209 

210 The affine should be diagonal or diagonal-permuted, use 

211 reorder_img to ensure that it is the case. 

212 

213 """ 

214 img = _utils.check_niimg_3d(img) 

215 mask = _utils.numpy_conversions._asarray(_get_data(img), dtype=bool) 

216 affine = img.affine 

217 (xmin, xmax), (ymin, ymax), (zmin, zmax) = get_bounds(mask.shape, affine) 

218 slices = ndimage.find_objects(mask) 

219 if len(slices) == 0: 

220 warnings.warn("empty mask", stacklevel=2) 

221 else: 

222 x_slice, y_slice, z_slice = slices[0] 

223 x_width, y_width, z_width = mask.shape 

224 xmin, xmax = (xmin + x_slice.start*(xmax - xmin)/x_width, 

225 xmin + x_slice.stop *(xmax - xmin)/x_width) 

226 ymin, ymax = (ymin + y_slice.start*(ymax - ymin)/y_width, 

227 ymin + y_slice.stop *(ymax - ymin)/y_width) 

228 zmin, zmax = (zmin + z_slice.start*(zmax - zmin)/z_width, 

229 zmin + z_slice.stop *(zmax - zmin)/z_width) 

230 

231 return xmin, xmax, ymin, ymax, zmin, zmax

304def resample_img(img, target_affine=None, target_shape=None, 

305 interpolation='continuous', copy=True, order="F", 

306 clip=True, fill_value=0, force_resample=False): 

307 """Resample a Niimg-like object 

308 

309 Parameters 

310 ---------- 

311 img : Niimg-like object 

312 See http://nilearn.github.io/manipulating_images/input_output.html 

313 Image(s) to resample. 

314 

315 target_affine : numpy.ndarray, optional 

316 If specified, the image is resampled corresponding to this new affine. 

317 target_affine can be a 3x3 or a 4x4 matrix. (See notes) 

318 

319 target_shape : tuple or list, optional 

320 If specified, the image will be resized to match this new shape. 

321 len(target_shape) must be equal to 3. 

322 If target_shape is specified, a target_affine of shape (4, 4) 

323 must also be given. (See notes) 

324 

325 interpolation : str, optional 

326 Can be 'continuous', 'linear', or 'nearest'. Indicates the resample 

327 method. Default='continuous'. 

328 

329 copy : bool, optional 

330 If True, guarantees that output array has no memory in common with 

331 input array. 

332 In all cases, input images are never modified by this function. 

333 Default=True. 

334 

335 order : "F" or "C", optional 

336 Data ordering in output array. This function is slightly faster with 

337 Fortran ordering. Default='F'. 

338 

339 clip : bool, optional 

340 If True (default) all resampled image values above max(img) and 

341 under min(img) are clipped to min(img) and max(img). Note that 

342 0 is added as an image value for clipping, and it is the padding 

343 value when extrapolating out of field of view. 

344 If False no clip is preformed. 

345 Default=True. 

346 

347 fill_value : float, optional 

348 Use a fill value for points outside of input volume. Default=0. 

349 

350 force_resample : bool, optional 

351 Intended for testing, this prevents the use of a padding optimzation. 

352 Default=False. 

353 

354 Returns 

355 ------- 

356 resampled : nibabel.Nifti1Image 

357 input image, resampled to have respectively target_shape and 

358 target_affine as shape and affine. 

359 

360 See Also 

361 -------- 

362 nilearn.image.resample_to_img 

363 

364 Notes 

365 ----- 

366 **BoundingBoxError** 

367 If a 4x4 transformation matrix (target_affine) is given and all of the 

368 transformed data points have a negative voxel index along one of the 

369 axis, then none of the data will be visible in the transformed image 

370 and a BoundingBoxError will be raised. 

371 

372 If a 4x4 transformation matrix (target_affine) is given and no target 

373 shape is provided, the resulting image will have voxel coordinate 

374 (0, 0, 0) in the affine offset (4th column of target affine) and will 

375 extend far enough to contain all the visible data and a margin of one 

376 voxel. 

377 

378 **3x3 transformation matrices** 

379 If a 3x3 transformation matrix is given as target_affine, it will be 

380 assumed to represent the three coordinate axes of the target space. In 

381 this case the affine offset (4th column of a 4x4 transformation matrix) 

382 as well as the target_shape will be inferred by resample_img, such that 

383 the resulting field of view is the closest possible (with a margin of 

384 1 voxel) bounding box around the transformed data. 

385 

386 In certain cases one may want to obtain a transformed image with the 

387 closest bounding box around the data, which at the same time respects 

388 a voxel grid defined by a 4x4 affine transformation matrix. In this 

389 case, one resamples the image using this function given the target 

390 affine and no target shape. One then uses crop_img on the result. 

391 

392 **NaNs and infinite values** 

393 This function handles gracefully NaNs and infinite values in the input 

394 data, however they make the execution of the function much slower. 

395 

396 **Handling non-native endian in given Nifti images** 

397 This function automatically changes the byte-ordering information 

398 in the image dtype to new byte order. From non-native to native, which 

399 implies that if the given image has non-native endianess then the output 

400 data in Nifti image will have native dtype. This is only the case when 

401 if the given target_affine (transformation matrix) is diagonal and 

402 homogenous. 

403 

404 """ 

405 from .image import new_img_like # avoid circular imports 

406 

407 # Do as many checks as possible before loading data, to avoid potentially 

408 # costly calls before raising an exception. 

409 if target_shape is not None and target_affine is None: 

410 raise ValueError("If target_shape is specified, target_affine should" 

411 " be specified too.") 

412 

413 if target_shape is not None and not len(target_shape) == 3: 

414 raise ValueError('The shape specified should be the shape of ' 

415 'the 3D grid, and thus of length 3. %s was specified' 

416 % str(target_shape)) 

417 

418 if target_shape is not None and target_affine.shape == (3, 3): 

419 raise ValueError("Given target shape without anchor vector: " 

420 "Affine shape should be (4, 4) and not (3, 3)") 

421 

422 if interpolation == 'continuous': 

423 interpolation_order = 3 

424 elif interpolation == 'linear': 

425 interpolation_order = 1 

426 elif interpolation == 'nearest': 

427 interpolation_order = 0 

428 else: 

429 message = ("interpolation must be either 'continuous', 'linear' " 

430 "or 'nearest' but it was set to '{0}'").format(interpolation) 

431 raise ValueError(message) 

432 

433 if isinstance(img, str): 

434 # Avoid a useless copy 

435 input_img_is_string = True 

436 else: 

437 input_img_is_string = False

442

445 if copy and not input_img_is_string: 

446 img = _utils.copy_img(img) 

447 return img 

448 if target_affine is affine and target_shape is shape: 

449 return img 

450 if target_affine is not None: 

451 target_affine = np.asarray(target_affine) 

452 

453 if (np.all(np.array(target_shape) == shape[:3]) and 

454 np.allclose(target_affine, affine)): 

455 if copy and not input_img_is_string: 

456 img = _utils.copy_img(img) 

457 return img

460 # The value of "copy" is of no importance: output is always a separate 

461 # array. 

462 data = _get_data(img) 

463 

464 # Get a bounding box for the transformed data 

465 # Embed target_affine in 4x4 shape if necessary 

466 if target_affine.shape == (3, 3): 

467 missing_offset = True 

468 target_affine_tmp = np.eye(4) 

469 target_affine_tmp[:3, :3] = target_affine 

470 target_affine = target_affine_tmp 

471 else: 

472 missing_offset = False 

473 target_affine = target_affine.copy() 

474 transform_affine = np.linalg.inv(target_affine).dot(affine) 

475 (xmin, xmax), (ymin, ymax), (zmin, zmax) = get_bounds( 

476 data.shape[:3], transform_affine)

478 # if target_affine is (3, 3), then calculate 

479 # offset from bounding box and update bounding box 

480 # to be in the voxel coordinates of the calculated 4x4 affine 

481 if missing_offset: 

482 offset = target_affine[:3, :3].dot([xmin, ymin, zmin]) 

483 target_affine[:3, 3] = offset 

484 (xmin, xmax), (ymin, ymax), (zmin, zmax) = (

487 # if target_shape is not given (always the case with 3x3

490 # box by a margin of 1 voxel

509

511 # Make sure that we have a list here 

512 if isinstance(target_shape, np.ndarray): 

513 target_shape = target_shape.tolist() 

514 target_shape = tuple(target_shape) 

515 

516 if LooseVersion(scipy.__version__) < LooseVersion('0.20'):

518 # that caused instability.

520 

521 if interpolation == 'continuous' and data.dtype.kind == 'i': 

522 # cast unsupported data types to closest support dtype

529 else: 

530 resampled_data_dtype = data.dtype 

531 

532 # Since the release of 0.17, resampling nifti images have some issues 

533 # when affine is passed as 1D array and if data is of non-native 

534 # endianess. 

535 # See issue https://github.com/nilearn/nilearn/issues/1445. 

536 # If affine is passed as 1D, scipy uses _nd_image.zoom_shift rather 

537 # than _geometric_transform (2D) where _geometric_transform is able 

538 # to swap byte order in scipy later than 0.15 for nonnative endianess. 

539 

540 # We convert to 'native' order to not have any issues either with 

541 # 'little' or 'big' endian data dtypes (non-native endians). 

542 if len(A.shape) == 1 and not resampled_data_dtype.isnative: 

543 resampled_data_dtype = resampled_data_dtype.newbyteorder('N') 

544 

545 # Code is generic enough to work for both 3D and 4D images 

546 other_shape = data_shape[3:] 

547 resampled_data = np.zeros(list(target_shape) + other_shape, 

548 order=order, dtype=resampled_data_dtype) 

549 

550 all_img = (slice(None), ) * 3 

551 

552 # if (A == I OR some combination of permutation(I) and sign-flipped(I)) AND 

553 # all(b == integers): 

554 if (np.all(np.eye(3) == A) and all(bt == np.round(bt) for bt in b) and 

555 not force_resample): 

556 # TODO: also check for sign flips 

557 # TODO: also check for permutations of I 

558 

559 # ... special case: can be solved with padding alone 

560 # crop source image and keep N voxels offset before/after volume 

561 cropped_img, offsets = crop_img(img, pad=False, return_offset=True) 

562 

563 # TODO: flip axes that are flipped 

564 # TODO: un-shuffle permuted dimensions 

565 

566 # offset the original un-cropped image indices by the relative 

567 # translation, b. 

568 indices = [(int(off.start - dim_b), int(off.stop - dim_b)) 

569 for off, dim_b in zip(offsets[:3], b[:3])] 

570 

571 # If image are not fully overlapping, place only portion of image. 

572 slices = [] 

573 for dimsize, index in zip(resampled_data.shape, indices): 

574 slices.append(slice(np.max((0, index[0])), 

575 np.min((dimsize, index[1])))) 

576 slices = tuple(slices) 

577 

578 # ensure the source image being placed isn't larger than the dest 

579 subset_indices = tuple(slice(0, s.stop-s.start) for s in slices) 

580 resampled_data[slices] = _get_data(cropped_img)[subset_indices] 

581 else: 

582 # If A is diagonal, ndimage.affine_transform is clever enough to use a 

583 # better algorithm. 

584 if np.all(np.diag(np.diag(A)) == A): 

585 if LooseVersion(scipy.__version__) < LooseVersion('0.18'): 

586 # Before scipy 0.18, ndimage.affine_transform was applying a 

587 # different logic to the offset for diagonal affine 

588 b = np.dot(linalg.inv(A), b) 

589 A = np.diag(A) 

590 # Iterate over a set of 3D volumes, as the interpolation problem is 

591 # separable in the extra dimensions. This reduces the 

592 # computational cost 

593 for ind in np.ndindex(*other_shape): 

594 _resample_one_img(data[all_img + ind], A, b, target_shape, 

595 interpolation_order, 

596 out=resampled_data[all_img + ind], 

597 copy=not input_img_is_string, 

598 fill_value=fill_value) 

599 

600 if clip: 

601 # force resampled data to have a range contained in the original data 

602 # preventing ringing artefact 

603 # We need to add zero as a value considered for clipping, as it 

604 # appears in padding images. 

605 vmin = min(data.min(), 0) 

606 vmax = max(data.max(), 0) 

607 resampled_data.clip(vmin, vmax, out=resampled_data) 

608 

609 return new_img_like(img, resampled_data, target_affine)

612def resample_to_img(source_img, target_img, 

613 interpolation='continuous', copy=True, order='F', 

614 clip=False, fill_value=0, force_resample=False): 

615 """Resample a Niimg-like source image on a target Niimg-like image 

616 (no registration is performed: the image should already be aligned). 

617 

618 .. versionadded:: 0.2.4 

619 

620 Parameters

683def reorder_img(img, resample=None): 

684 """Returns an image with the affine diagonal (by permuting axes). 

685 The orientation of the new image will be RAS (Right, Anterior, Superior). 

686 If it is impossible to get xyz ordering by permuting the axes, a 

687 'ValueError' is raised. 

688 

689 Parameters 

690 ----------- 

691 img : Niimg-like object 

692 See http://nilearn.github.io/manipulating_images/input_output.html 

693 Image to reorder. 

694 

695 resample : None or string in {'continuous', 'linear', 'nearest'}, optional 

696 If resample is None (default), no resampling is performed, the 

697 axes are only permuted. 

698 Otherwise resampling is performed and 'resample' will 

699 be passed as the 'interpolation' argument into 

700 resample_img. 

701 

702 """ 

703 from .image import new_img_like 

704 

705 img = _utils.check_niimg(img) 

706 # The copy is needed in order not to modify the input img affine 

707 # see https://github.com/nilearn/nilearn/issues/325 for a concrete bug 

708 affine = img.affine.copy() 

709 A, b = to_matrix_vector(affine) 

710 

711 if not np.all((np.abs(A) > 0.001).sum(axis=0) == 1): 

712 # The affine is not nearly diagonal 

713 if resample is None: 

714 raise ValueError('Cannot reorder the axes: ' 

715 'the image affine contains rotations') 

716 else: 

717 # Identify the voxel size using a QR decomposition of the 

718 # affine 

719 Q, R = np.linalg.qr(affine[:3, :3]) 

720 target_affine = np.diag(np.abs(np.diag(R))[ 

721 np.abs(Q).argmax(axis=1)]) 

722 return resample_img(img, target_affine=target_affine, 

723 interpolation=resample) 

724 

725 axis_numbers = np.argmax(np.abs(A), axis=0) 

726 data = _get_data(img) 

727 while not np.all(np.sort(axis_numbers) == axis_numbers): 

728 first_inversion = np.argmax(np.diff(axis_numbers)<0) 

729 axis1 = first_inversion + 1 

730 axis2 = first_inversion 

731 data = np.swapaxes(data, axis1, axis2) 

732 order = np.array((0, 1, 2, 3)) 

733 order[axis1] = axis2 

734 order[axis2] = axis1 

735 affine = affine.T[order].T 

736 A, b = to_matrix_vector(affine) 

737 axis_numbers = np.argmax(np.abs(A), axis=0) 

738 

739 # Now make sure the affine is positive 

740 pixdim = np.diag(A).copy() 

741 if pixdim[0] < 0: 

742 b[0] = b[0] + pixdim[0]*(data.shape[0] - 1) 

743 pixdim[0] = -pixdim[0] 

744 slice1 = slice(None, None, -1) 

745 else: 

746 slice1 = slice(None, None, None) 

747 if pixdim[1] < 0: 

748 b[1] = b[1] + pixdim[1]*(data.shape[1] - 1) 

749 pixdim[1] = -pixdim[1] 

750 slice2 = slice(None, None, -1) 

751 else: 

752 slice2 = slice(None, None, None) 

753 if pixdim[2] < 0: 

754 b[2] = b[2] + pixdim[2]*(data.shape[2] - 1) 

755 pixdim[2] = -pixdim[2] 

756 slice3 = slice(None, None, -1) 

757 else: 

758 slice3 = slice(None, None, None) 

759 data = data[slice1, slice2, slice3] 

760 affine = from_matrix_vector(np.diag(pixdim), b) 

761 

762 return new_img_like(img, data, affine)
