33class NiftiMapsMasker(BaseMasker, CacheMixin): 

34 """Class for masking of Niimg-like objects. 

35 

36 NiftiMapsMasker is useful when data from overlapping volumes should be 

37 extracted (contrarily to NiftiLabelsMasker). Use case: Summarize brain 

38 signals from large-scale networks obtained by prior PCA or ICA. 

39 

40 Note that, Inf or NaN present in the given input images are automatically 

41 put to zero rather than considered as missing data. 

42 

43 Parameters 

44 ---------- 

45 maps_img : 4D niimg-like object 

46 See http://nilearn.github.io/manipulating_images/input_output.html 

47 Set of continuous maps. One representative time course per map is 

48 extracted using least square regression. 

49 

50 mask_img : 3D niimg-like object, optional 

51 See http://nilearn.github.io/manipulating_images/input_output.html 

52 Mask to apply to regions before extracting signals. 

53 

54 allow_overlap : boolean, optional 

55 If False, an error is raised if the maps overlaps (ie at least two 

56 maps have a non-zero value for the same voxel). Default=True. 

57 

58 smoothing_fwhm : float, optional 

59 If smoothing_fwhm is not None, it gives the full-width half maximum in 

60 millimeters of the spatial smoothing to apply to the signal. 

61 

62 standardize : {False, True, 'zscore', 'psc'}, optional 

63 Strategy to standardize the signal. 

64 'zscore': the signal is z-scored. Timeseries are shifted 

65 to zero mean and scaled to unit variance. 

66 'psc': Timeseries are shifted to zero mean value and scaled 

67 to percent signal change (as compared to original mean signal). 

68 True : the signal is z-scored. Timeseries are shifted 

69 to zero mean and scaled to unit variance. 

70 False : Do not standardize the data. 

71 Default=False. 

72 

73 standardize_confounds : boolean, optional 

74 If standardize_confounds is True, the confounds are z-scored: 

75 their mean is put to 0 and their variance to 1 in the time dimension. 

76 Default=True. 

77 

78 high_variance_confounds : boolean, optional 

79 If True, high variance confounds are computed on provided image with 

80 :func:`nilearn.image.high_variance_confounds` and default parameters 

81 and regressed out. Default=False. 

82 

83 detrend : boolean, optional 

84 This parameter is passed to signal.clean. Please see the related 

85 documentation for details. Default=False. 

86 

87 low_pass : None or float, optional 

88 This parameter is passed to signal.clean. Please see the related 

89 documentation for details 

90 

91 high_pass : None or float, optional 

92 This parameter is passed to signal.clean. Please see the related 

93 documentation for details 

94 

95 t_r : float, optional 

96 This parameter is passed to signal.clean. Please see the related 

97 documentation for details 

98 

99 dtype : {dtype, "auto"}, optional 

100 Data type toward which the data should be converted. If "auto", the 

101 data will be converted to int32 if dtype is discrete and float32 if it 

102 is continuous. 

103 

104 resampling_target : {"data", "mask", "maps", None}, optional. 

105 Gives which image gives the final shape/size. For example, if 

106 `resampling_target` is "mask" then maps_img and images provided to 

107 fit() are resampled to the shape and affine of mask_img. "None" means 

108 no resampling: if shapes and affines do not match, a ValueError is 

109 raised. Default="data". 

110 

111 memory : joblib.Memory or str, optional 

112 Used to cache the region extraction process. 

113 By default, no caching is done. If a string is given, it is the 

114 path to the caching directory. 

115 

116 memory_level : int, optional 

117 Aggressiveness of memory caching. The higher the number, the higher 

118 the number of functions that will be cached. Zero means no caching. 

119 Default=0. 

120 

121 verbose : integer, optional 

122 Indicate the level of verbosity. By default, nothing is printed. 

123 Default=0. 

124 

125 Notes 

126 ----- 

127 If resampling_target is set to "maps", every 3D image processed by 

128 transform() will be resampled to the shape of maps_img. It may lead to a 

129 very large memory consumption if the voxel number in maps_img is large. 

130 

131 See also 

132 -------- 

133 nilearn.input_data.NiftiMasker 

134 nilearn.input_data.NiftiLabelsMasker 

135 

136 """

183 def fit(self, X=None, y=None): 

184 """Prepare signal extraction from regions. 

185 

186 All parameters are unused, they are for scikit-learn compatibility. 

187 

188 """ 

189 # Load images 

190 logger.log("loading regions from %s" % 

191 _utils._repr_niimgs(self.maps_img, 

192 shorten=(not self.verbose)), 

193 verbose=self.verbose) 

194 self.maps_img_ = _utils.check_niimg(self.maps_img, dtype=self.dtype, atleast_4d=True) 

195 self.maps_img_ = image.clean_img(self.maps_img_, detrend=False, 

196 standardize=False, 

197 ensure_finite=True) 

198 if self.mask_img is not None: 

199 logger.log("loading mask from %s" % 

200 _utils._repr_niimgs(self.mask_img, 

201 shorten=(not self.verbose)), 

202 verbose=self.verbose) 

203 self.mask_img_ = _utils.check_niimg_3d(self.mask_img) 

204 else: 

205 self.mask_img_ = None 

206 

207 # Check shapes and affines or resample. 

208 if self.resampling_target is None and self.mask_img_ is not None: 

209 _check_same_fov(mask=self.mask_img_, maps=self.maps_img_, 

210 raise_error=True) 

211 

212 elif self.resampling_target == "mask" and self.mask_img_ is not None: 

213 if self.verbose > 0: 

214 print("Resampling maps") 

215 self.maps_img_ = image.resample_img( 

216 self.maps_img_, 

217 target_affine=self.mask_img_.affine, 

218 target_shape=self.mask_img_.shape, 

219 interpolation="continuous", 

220 copy=True) 

221 

222 elif self.resampling_target == "maps" and self.mask_img_ is not None: 

223 if self.verbose > 0: 

224 print("Resampling mask") 

225 self.mask_img_ = image.resample_img( 

226 self.mask_img_, 

227 target_affine=self.maps_img_.affine, 

228 target_shape=self.maps_img_.shape[:3], 

229 interpolation="nearest", 

230 copy=True) 

231 

232 return self

246 def transform_single_imgs(self, imgs, confounds=None): 

247 """Extract signals from a single 4D niimg. 

248 

249 Parameters 

250 ---------- 

251 imgs : 3D/4D Niimg-like object 

252 See http://nilearn.github.io/manipulating_images/input_output.html 

253 Images to process. It must boil down to a 4D image with scans 

254 number as last dimension. 

255 

256 confounds : CSV file or array-like, optional 

257 This parameter is passed to signal.clean. Please see the related 

258 documentation for details. 

259 shape: (number of scans, number of confounds) 

260 

261 Returns 

262 ------- 

263 region_signals : 2D numpy.ndarray 

264 Signal for each map. 

265 shape: (number of scans, number of maps) 

266 

267 """ 

268 # We handle the resampling of maps and mask separately because the 

269 # affine of the maps and mask images should not impact the extraction 

270 # of the signal. 

271 

272 if not hasattr(self, '_resampled_maps_img_'): 

273 self._resampled_maps_img_ = self.maps_img_ 

274 if not hasattr(self, '_resampled_mask_img_'): 

275 self._resampled_mask_img_ = self.mask_img_ 

276 

277 if self.resampling_target is None: 

278 imgs_ = _utils.check_niimg(imgs, atleast_4d=True) 

279 images = dict(maps=self.maps_img_, data=imgs_) 

280 if self.mask_img_ is not None: 

281 images['mask'] = self.mask_img_ 

282 _check_same_fov(raise_error=True, **images) 

283 else: 

284 if self.resampling_target == "data": 

285 imgs_ = _utils.check_niimg(imgs, atleast_4d=True) 

286 ref_img = imgs_ 

287 elif self.resampling_target == "mask": 

288 self._resampled_mask_img_ = self.mask_img_ 

289 ref_img = self.mask_img_ 

290 elif self.resampling_target == "maps": 

291 self._resampled_maps_img_ = self.maps_img_ 

292 ref_img = self.maps_img_ 

293 

294 if not _check_same_fov(ref_img, self._resampled_maps_img_): 

295 if self.verbose > 0: 

296 print("Resampling maps") 

297 self._resampled_maps_img_ = self._cache(image.resample_img)( 

298 self.maps_img_, interpolation="continuous", 

299 target_shape=ref_img.shape[:3], 

300 target_affine=ref_img.affine) 

301 

302 if (self.mask_img_ is not None 

303 and not _check_same_fov(ref_img, self.mask_img_)): 

304 if self.verbose > 0: 

305 print("Resampling mask") 

306 self._resampled_mask_img_ = self._cache(image.resample_img)( 

307 self.mask_img_, interpolation="nearest", 

308 target_shape=ref_img.shape[:3], 

309 target_affine=ref_img.affine) 

310 

311 if not self.allow_overlap: 

312 # Check if there is an overlap. 

313 

314 # If float, we set low values to 0 

315 data = get_data(self._resampled_maps_img_) 

316 dtype = data.dtype 

317 if dtype.kind == 'f': 

318 data[data < np.finfo(dtype).eps] = 0. 

319 

320 # Check the overlaps 

321 if np.any(np.sum(data > 0., axis=3) > 1): 

322 raise ValueError( 

323 'Overlap detected in the maps. The overlap may be ' 

324 'due to the atlas itself or possibly introduced by ' 

325 'resampling' 

326 ) 

327 

328 target_shape = None 

329 target_affine = None 

330 if self.resampling_target != 'data': 

331 target_shape = self._resampled_maps_img_.shape[:3] 

332 target_affine = self._resampled_maps_img_.affine 

333 

334 params = get_params(NiftiMapsMasker, self, 

335 ignore=['resampling_target']) 

336 params['target_shape'] = target_shape 

337 params['target_affine'] = target_affine 

338 

339 region_signals, labels_ = self._cache( 

340 filter_and_extract, ignore=['verbose', 'memory', 'memory_level'])( 

341 # Images 

342 imgs, _ExtractionFunctor(self._resampled_maps_img_, 

343 self._resampled_mask_img_), 

344 # Pre-treatments 

345 params, 

346 confounds=confounds, 

347 dtype=self.dtype, 

348 # Caching 

349 memory=self.memory, 

350 memory_level=self.memory_level, 

351 # kwargs 

352 verbose=self.verbose) 

353 self.labels_ = labels_ 

354 return region_signals
