33def _check_same_fov(*args, **kwargs): 

34 """Returns True if provided images has the same field of view (shape and 

35 affine). Returns False or raise an error elsewhere, depending on the 

36 `raise_error` argument. This function can take an unlimited number of 

37 images as arguments or keyword arguments and raise a user-friendly 

38 ValueError if asked. 

39 

40 Parameters 

41 ---------- 

42 args : images 

43 Images to be checked. Images passed without keywords will be labelled 

44 as img_#1 in the error message (replace 1 with the appropriate index). 

45 

46 kwargs : images 

47 Images to be checked. In case of error, images will be reference by 

48 their keyword name in the error message. 

49 

50 raise_error : boolean, optional 

51 If True, an error will be raised in case of error. 

52 

53 """ 

54 raise_error = kwargs.pop('raise_error', False) 

55 for i, arg in enumerate(args): 

56 kwargs['img_#%i' % i] = arg 

57 errors = [] 

58 for (a_name, a_img), (b_name, b_img) in itertools.combinations( 

59 kwargs.items(), 2): 

60 if not a_img.shape[:3] == b_img.shape[:3]: 

61 errors.append((a_name, b_name, 'shape')) 

62 if not np.allclose(a_img.affine, b_img.affine): 

63 errors.append((a_name, b_name, 'affine')) 

64 if len(errors) > 0 and raise_error: 

65 raise ValueError('Following field of view errors were detected:\n' + 

66 '\n'.join(['- %s and %s do not have the same %s' % e 

67 for e in errors])) 

68 return (len(errors) == 0)

80def _iter_check_niimg(niimgs, ensure_ndim=None, atleast_4d=False, 

81 target_fov=None, dtype=None, 

82 memory=Memory(location=None), 

83 memory_level=0, verbose=0): 

84 """Iterate over a list of niimgs and do sanity checks and resampling 

85 

86 Parameters 

87 ---------- 

88 niimgs : list of niimg or glob pattern 

89 Image to iterate over. 

90 

91 ensure_ndim : integer, optional 

92 If specified, an error is raised if the data does not have the 

93 required dimension. 

94 

95 atleast_4d : boolean, optional 

96 If True, any 3D image is converted to a 4D single scan. 

97 Default=False. 

98 

99 target_fov : tuple of affine and shape, optional 

100 If specified, images are resampled to this field of view. 

101 

102 dtype : {dtype, "auto"}, optional 

103 Data type toward which the data should be converted. If "auto", the 

104 data will be converted to int32 if dtype is discrete and float32 if it 

105 is continuous. 

106 

107 memory : instance of joblib.Memory or string, optional 

108 Used to cache the masking process. 

109 By default, no caching is done. If a string is given, it is the 

110 path to the caching directory. 

111 Default=Memory(location=None). 

112 

113 memory_level : integer, optional 

114 Rough estimator of the amount of memory used by caching. Higher value 

115 means more memory for caching. Default=0. 

116 

117 verbose : integer, optional 

118 Indicate the level of verbosity. By default, nothing is printed. 

119 Default=0. 

120 

121 See also 

122 -------- 

123 check_niimg, check_niimg_3d, check_niimg_4d 

124 

125 """ 

126 # If niimgs is a string, use glob to expand it to the matching filenames. 

127 niimgs = _resolve_globbing(niimgs) 

128 

129 ref_fov = None 

130 resample_to_first_img = False 

131 ndim_minus_one = ensure_ndim - 1 if ensure_ndim is not None else None 

132 if target_fov is not None and target_fov != "first": 

133 ref_fov = target_fov 

134 i = -1 

135 for i, niimg in enumerate(niimgs): 

136 try: 

137 niimg = check_niimg( 

138 niimg, ensure_ndim=ndim_minus_one, atleast_4d=atleast_4d, 

139 dtype=dtype) 

140 if i == 0: 

141 ndim_minus_one = len(niimg.shape) 

142 if ref_fov is None: 

143 ref_fov = (niimg.affine, niimg.shape[:3]) 

144 resample_to_first_img = True 

145 

146 if not _check_fov(niimg, ref_fov[0], ref_fov[1]): 

147 if target_fov is not None: 

148 from nilearn import image # we avoid a circular import 

149 if resample_to_first_img: 

150 warnings.warn('Affine is different across subjects.' 

151 ' Realignement on first subject ' 

152 'affine forced') 

153 niimg = cache(image.resample_img, memory, 

154 func_memory_level=2, 

155 memory_level=memory_level)( 

156 niimg, target_affine=ref_fov[0], 

157 target_shape=ref_fov[1]) 

158 else: 

159 raise ValueError( 

160 "Field of view of image #%d is different from " 

161 "reference FOV.\n" 

162 "Reference affine:\n%r\nImage affine:\n%r\n" 

163 "Reference shape:\n%r\nImage shape:\n%r\n" 

164 % (i, ref_fov[0], niimg.affine, ref_fov[1], 

165 niimg.shape)) 

166 yield niimg 

167 except DimensionError as exc: 

168 # Keep track of the additional dimension in the error 

169 exc.increment_stack_counter() 

170 raise 

171 except TypeError as exc: 

172 img_name = '' 

173 if isinstance(niimg, str): 

174 img_name = " (%s) " % niimg 

175 

176 exc.args = (('Error encountered while loading image #%d%s' 

177 % (i, img_name),) + exc.args) 

178 raise 

179 

180 # Raising an error if input generator is empty. 

181 if i == -1: 

182 raise ValueError("Input niimgs list is empty.")

185def check_niimg(niimg, ensure_ndim=None, atleast_4d=False, dtype=None, 

186 return_iterator=False, 

187 wildcards=True): 

188 """Check that niimg is a proper 3D/4D niimg. Turn filenames into objects. 

189 

190 Parameters 

191 ---------- 

192 niimg : Niimg-like object 

193 See http://nilearn.github.io/manipulating_images/input_output.html 

194 If niimg is a string, consider it as a path to Nifti image and 

195 call nibabel.load on it. The '~' symbol is expanded to the user home 

196 folder. 

197 If it is an object, check if the affine attribute present and that 

198 nilearn.image.get_data returns a result, raise TypeError otherwise. 

199 

200 ensure_ndim : integer {3, 4}, optional 

201 Indicate the dimensionality of the expected niimg. An 

202 error is raised if the niimg is of another dimensionality. 

203 

204 atleast_4d : boolean, optional 

205 Indicates if a 3d image should be turned into a single-scan 4d niimg. 

206 Default=False. 

207 

208 dtype : {None, dtype, "auto"}, optional 

209 Data type toward which the data should be converted. If "auto", the 

210 data will be converted to int32 if dtype is discrete and float32 if it 

211 is continuous. If None, data will not be converted to a new data type. 

212 Default=None. 

213 

214 return_iterator : boolean, optional 

215 Returns an iterator on the content of the niimg file input. 

216 Default=False. 

217 

218 wildcards : boolean, optional 

219 Use niimg as a regular expression to get a list of matching input 

220 filenames. 

221 If multiple files match, the returned list is sorted using an ascending 

222 order. 

223 If no file matches the regular expression, a ValueError exception is 

224 raised. 

225 Default=True. 

226 

227 Returns 

228 ------- 

229 result : 3D/4D Niimg-like object 

230 Result can be nibabel.Nifti1Image or the input, as-is. It is guaranteed 

231 that the returned object has an affine attribute and that its data can 

232 be retrieved with nilearn.image.get_data. 

233 

234 Notes 

235 ----- 

236 In nilearn, special care has been taken to make image manipulation easy. 

237 This method is a kind of pre-requisite for any data processing method in 

238 nilearn because it checks if data have a correct format and loads them if 

239 necessary. 

240 

241 Its application is idempotent. 

242 

243 See also 

244 -------- 

245 _iter_check_niimg, check_niimg_3d, check_niimg_4d 

246 

247 """ 

248 from ..image import new_img_like # avoid circular imports 

249 

250 if isinstance(niimg, str): 

251 if wildcards and ni.EXPAND_PATH_WILDCARDS: 

252 # Ascending sorting + expand user path 

253 filenames = sorted(glob.glob(os.path.expanduser(niimg))) 

254 

255 # processing filenames matching globbing expression 

256 if len(filenames) >= 1 and glob.has_magic(niimg): 

257 niimg = filenames # iterable case 

258 # niimg is an existing filename 

259 elif [niimg] == filenames: 

260 niimg = filenames[0] 

261 # No files found by glob 

262 elif glob.has_magic(niimg): 

263 # No files matching the glob expression, warn the user 

264 message = ("No files matching the entered niimg expression: " 

265 "'%s'.\n You may have left wildcards usage " 

266 "activated: please set the global constant " 

267 "'nilearn.EXPAND_PATH_WILDCARDS' to False to " 

268 "deactivate this behavior.") % niimg 

269 raise ValueError(message) 

270 else: 

271 raise ValueError("File not found: '%s'" % niimg) 

272 elif not os.path.exists(niimg): 

273 raise ValueError("File not found: '%s'" % niimg) 

274 

275 # in case of an iterable 

276 if hasattr(niimg, "__iter__") and not isinstance(niimg, str): 

277 if return_iterator: 

278 return _iter_check_niimg(niimg, ensure_ndim=ensure_ndim, 

279 dtype=dtype) 

280 return concat_niimgs(niimg, ensure_ndim=ensure_ndim, dtype=dtype) 

281 

282 # Otherwise, it should be a filename or a SpatialImage, we load it 

283 niimg = load_niimg(niimg, dtype=dtype) 

284 

285 if ensure_ndim == 3 and len(niimg.shape) == 4 and niimg.shape[3] == 1: 

286 # "squeeze" the image. 

287 data = _safe_get_data(niimg) 

288 affine = niimg.affine 

289 niimg = new_img_like(niimg, data[:, :, :, 0], affine) 

290 if atleast_4d and len(niimg.shape) == 3: 

291 data = _get_data(niimg).view() 

292 data.shape = data.shape + (1, ) 

293 niimg = new_img_like(niimg, data, niimg.affine) 

294 

295 if ensure_ndim is not None and len(niimg.shape) != ensure_ndim: 

296 raise DimensionError(len(niimg.shape), ensure_ndim) 

297 

298 if return_iterator: 

299 return (_index_img(niimg, i) for i in range(niimg.shape[3])) 

300 

301 return niimg

383def concat_niimgs(niimgs, dtype=np.float32, ensure_ndim=None, 

384 memory=Memory(location=None), memory_level=0, 

385 auto_resample=False, verbose=0): 

386 """Concatenate a list of 3D/4D niimgs of varying lengths. 

387 

388 The niimgs list can contain niftis/paths to images of varying dimensions 

389 (i.e., 3D or 4D) as well as different 3D shapes and affines, as they 

390 will be matched to the first image in the list if auto_resample=True. 

391 

392 Parameters 

393 ---------- 

394 niimgs : iterable of Niimg-like objects or glob pattern 

395 See http://nilearn.github.io/manipulating_images/input_output.html 

396 Niimgs to concatenate. 

397 

398 dtype : numpy dtype, optional 

399 The dtype of the returned image. Default=np.float32. 

400 

401 ensure_ndim : integer, optional 

402 Indicate the dimensionality of the expected niimg. An 

403 error is raised if the niimg is of another dimensionality. 

404 

405 auto_resample : boolean, optional 

406 Converts all images to the space of the first one. 

407 Default=False. 

408 

409 verbose : int, optional 

410 Controls the amount of verbosity (0 means no messages). 

411 Default=0. 

412 

413 memory : instance of joblib.Memory or string, optional 

414 Used to cache the resampling process. 

415 By default, no caching is done. If a string is given, it is the 

416 path to the caching directory. 

417 Default=Memory(location=None). 

418 

419 memory_level : integer, optional 

420 Rough estimator of the amount of memory used by caching. Higher value 

421 means more memory for caching. Default=0. 

422 

423 Returns 

424 ------- 

425 concatenated : nibabel.Nifti1Image 

426 A single image. 

427 

428 See Also 

429 -------- 

430 nilearn.image.index_img 

431 

432 """ 

433 from ..image import new_img_like # avoid circular imports 

434 

435 target_fov = 'first' if auto_resample else None 

436 

437 # We remove one to the dimensionality because of the list is one dimension. 

438 ndim = None 

439 if ensure_ndim is not None: 

440 ndim = ensure_ndim - 1 

441 

442 # If niimgs is a string, use glob to expand it to the matching filenames. 

443 niimgs = _resolve_globbing(niimgs) 

444 

445 # First niimg is extracted to get information and for new_img_like 

446 first_niimg = None 

447 

448 iterator, literator = itertools.tee(iter(niimgs)) 

449 try: 

450 first_niimg = check_niimg(next(literator), ensure_ndim=ndim) 

451 except StopIteration: 

452 raise TypeError('Cannot concatenate empty objects') 

453 except DimensionError as exc: 

454 # Keep track of the additional dimension in the error 

455 exc.increment_stack_counter() 

456 raise 

457 

458 # If no particular dimensionality is asked, we force consistency wrt the 

459 # first image 

460 if ndim is None: 

461 ndim = len(first_niimg.shape) 

462 

463 if ndim not in [3, 4]: 

464 raise TypeError('Concatenated images must be 3D or 4D. You gave a ' 

465 'list of %dD images' % ndim) 

466 

467 lengths = [first_niimg.shape[-1] if ndim == 4 else 1] 

468 for niimg in literator: 

469 # We check the dimensionality of the niimg 

470 try: 

471 niimg = check_niimg(niimg, ensure_ndim=ndim) 

472 except DimensionError as exc: 

473 # Keep track of the additional dimension in the error 

474 exc.increment_stack_counter() 

475 raise 

476 lengths.append(niimg.shape[-1] if ndim == 4 else 1) 

477 

478 target_shape = first_niimg.shape[:3] 

479 if dtype == None: 

480 dtype = _get_data(first_niimg).dtype 

481 data = np.ndarray(target_shape + (sum(lengths), ), 

482 order="F", dtype=dtype) 

483 cur_4d_index = 0 

484 for index, (size, niimg) in enumerate(zip(lengths, _iter_check_niimg( 

485 iterator, atleast_4d=True, target_fov=target_fov, 

486 memory=memory, memory_level=memory_level))): 

487 

488 if verbose > 0: 

489 if isinstance(niimg, str): 

490 nii_str = "image " + niimg 

491 else: 

492 nii_str = "image #" + str(index) 

493 print("Concatenating {0}: {1}".format(index + 1, nii_str)) 

494 

495 data[..., cur_4d_index:cur_4d_index + size] = _get_data(niimg) 

496 cur_4d_index += size 

497 

498 return new_img_like(first_niimg, data, first_niimg.affine, copy_header=True)
