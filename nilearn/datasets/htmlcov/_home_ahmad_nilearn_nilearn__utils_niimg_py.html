29def _safe_get_data(img, ensure_finite=False, copy_data=False): 

30 """ Get the data in the image without having a side effect on the 

31 Nifti1Image object 

32 

33 Parameters 

34 ---------- 

35 img: Nifti image/object 

36 Image to get data. 

37 

38 ensure_finite: bool 

39 If True, non-finite values such as (NaNs and infs) found in the 

40 image will be replaced by zeros. 

41 

42 copy_data: bool, default is False 

43 If true, the returned data is a copy of the img data. 

44 

45 Returns 

46 ------- 

47 data: numpy array 

48 nilearn.image.get_data return from Nifti image. 

49 """ 

50 if copy_data: 

51 img = copy.deepcopy(img) 

52 

53 # typically the line below can double memory usage 

54 # that's why we invoke a forced call to the garbage collector 

55 gc.collect() 

56 

57 data = _get_data(img) 

58 if ensure_finite: 

59 non_finite_mask = np.logical_not(np.isfinite(data)) 

60 if non_finite_mask.sum() > 0: # any non_finite_mask values? 

61 warn( 

62 "Non-finite values detected. " 

63 "These values will be replaced with zeros." 

64 ) 

65 data[non_finite_mask] = 0 

66 

67 return data

70def _get_target_dtype(dtype, target_dtype): 

71 """Returns a new dtype if conversion is needed 

72 

73 Parameters 

74 ---------- 

75 

76 dtype: dtype 

77 Data type of the original data 

78 

79 target_dtype: {None, dtype, "auto"} 

80 If None, no conversion is required. If a type is provided, the 

81 function will check if a conversion is needed. The "auto" mode will 

82 automatically convert to int32 if dtype is discrete and float32 if it 

83 is continuous. 

84 

85 Returns 

86 ------- 

87 

88 dtype: dtype 

89 The data type toward which the original data should be converted. 

90 """ 

91 

92 if target_dtype is None: 

93 return None 

94 if target_dtype == 'auto': 

95 if dtype.kind == 'i': 

96 target_dtype = np.int32 

97 else: 

98 target_dtype = np.float32 

99 if target_dtype == dtype: 

100 return None 

101 return target_dtype

104def load_niimg(niimg, dtype=None): 

105 """Load a niimg, check if it is a nibabel SpatialImage and cast if needed 

106 

107 Parameters: 

108 ----------- 

109 

110 niimg: Niimg-like object 

111 See http://nilearn.github.io/manipulating_images/input_output.html 

112 Image to load. 

113 

114 dtype: {dtype, "auto"} 

115 Data type toward which the data should be converted. If "auto", the 

116 data will be converted to int32 if dtype is discrete and float32 if it 

117 is continuous. 

118 

119 Returns: 

120 -------- 

121 img: image 

122 A loaded image object. 

123 """ 

124 from ..image import new_img_like # avoid circular imports 

125 

126 if isinstance(niimg, str): 

127 # data is a filename, we load it 

128 niimg = nibabel.load(niimg) 

129 elif not isinstance(niimg, nibabel.spatialimages.SpatialImage): 

130 raise TypeError("Data given cannot be loaded because it is" 

131 " not compatible with nibabel format:\n" 

132 + _repr_niimgs(niimg, shorten=True)) 

133 

134 dtype = _get_target_dtype(_get_data(niimg).dtype, dtype) 

135 

136 if dtype is not None: 

137 # Copyheader and set dtype in header if header exists 

138 if niimg.header is not None: 

139 niimg = new_img_like(niimg, _get_data(niimg).astype(dtype), 

140 niimg.affine, copy_header=True) 

141 niimg.header.set_data_dtype(dtype) 

142 else: 

143 niimg = new_img_like(niimg, _get_data(niimg).astype(dtype), 

144 niimg.affine) 

145 

146 return niimg

149def _is_binary_niimg(niimg): 

150 """Returns whether a given niimg is binary or not. 

151 

152 Parameters 

153 ---------- 

154 niimg: Niimg-like object 

155 See http://nilearn.github.io/manipulating_images/input_output.html 

156 Image to test. 

157 

158 Returns 

159 ------- 

160 is_binary: Boolean 

161 True if binary, False otherwise. 

162 

163 """ 

164 niimg = load_niimg(niimg) 

165 data = _safe_get_data(niimg, ensure_finite=True) 

166 unique_values = np.unique(data) 

167 if len(unique_values) != 2: 

168 return False 

169 return sorted(list(unique_values)) == [0,1]

193def _repr_niimgs(niimgs, shorten=True): 

194 """ Pretty printing of niimg or niimgs. 

195 

196 Parameters 

197 ---------- 

198 niimgs: image or collection of images 

199 nibabel SpatialImage to repr. 

200 

201 shorten: boolean, optional, default is True 

202 If True, filenames with more than 20 characters will be 

203 truncated, and lists of more than 3 file names will be 

204 printed with only first and last element. 

205 

206 Returns 

207 ------- 

208 repr: str 

209 String representation of the image. 

210 """ 

211 # Maximum number of elements to be displayed 

212 # Note: should be >= 3 to make sense... 

213 list_max_display = 3 

214 # Simple string case 

215 if isinstance(niimgs, (str, Path)): 

216 return _short_repr(niimgs, shorten=shorten) 

217 # Collection case 

218 if isinstance(niimgs, collections.abc.Iterable): 

219 if shorten and len(niimgs) > list_max_display: 

220 return '[%s]' % ',\n ...\n '.join(_repr_niimgs(niimg, shorten=shorten) for niimg in [niimgs[0], niimgs[-1]]) 

221 elif len(niimgs) > list_max_display: 

222 return '[%s]' % ',\n '.join(_repr_niimgs(niimg, shorten=shorten) for niimg in niimgs) 

223 else: 

224 return '[%s]' % ', '.join(_repr_niimgs(niimg, shorten=shorten) for niimg in niimgs) 

225 # Nibabel objects have a 'get_filename' 

226 try: 

227 filename = niimgs.get_filename() 

228 if filename is not None: 

229 return "%s('%s')" % (niimgs.__class__.__name__, 

230 _short_repr(filename, 

231 shorten=shorten)) 

232 else: 

233 # No shortening in this case 

234 return "%s(\nshape=%s,\naffine=%s\n)" % \ 

235 (niimgs.__class__.__name__, 

236 repr(niimgs.shape), 

237 repr(niimgs.affine)) 

238 except: 

239 pass 

240 return _short_repr(repr(niimgs), shorten=shorten)

243def _short_repr(niimg_rep, shorten=True, truncate=20): 

244 """Gives a shorten version on niimg representation 

245 """ 

246 # Make sure truncate has a reasonable value 

247 truncate = max(truncate, 10) 

248 path_to_niimg = Path(niimg_rep) 

249 if not shorten: 

250 return str(path_to_niimg) 

251 # If the name of the file itself is larger than 

252 # truncate, then shorten the name only 

253 if len(path_to_niimg.name) > truncate: 

254 return path_to_niimg.name[: (truncate - 2)] + '...' 

255 # Else add some folder structure if available 

256 else: 

257 rep = path_to_niimg.name 

258 if len(path_to_niimg.parts) > 1: 

259 for p in path_to_niimg.parts[::-1][1:]: 

260 if len(rep) + len(p) < truncate - 3: 

261 rep = str(Path(p, rep)) 

262 else: 

263 rep = str(Path("...", rep)) 

264 break 

265 return rep 

266 return niimg_rep
